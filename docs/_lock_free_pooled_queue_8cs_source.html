120 public bool TryEnqueue(T item)
121 {
122 var node = _pool.Get(item);
123 while (true)
124 {
125 var oldTail = _tail;
126 var oldNext = oldTail.Next;
127
128 if (oldTail != _tail) // Tail changed while we were reading
129 {
130 continue;
131 }
132
133 if (oldNext == null)
134 {
135 // Try to link the new node at the end
136 if (Interlocked.CompareExchange(ref oldTail.Next, node, null) == null)
137 {
138 // Success – try to advance the tail (non‑critical)
139 Interlocked.CompareExchange(ref _tail, node, oldTail);
140 Interlocked.Increment(ref _count);
141 return true;
142 }
143 }
144 else
145 {
146 // Tail is lagging – help advance it
147 Interlocked.CompareExchange(ref _tail, oldNext, oldTail);
148 }
149 }
150 }
151
155 public bool TryDequeue(out T item)
156 {
157 while (true)
158 {
159 var head = _head;
160 var tail = _tail;
161 var next = head.Next;
162
163 if (head != _head) // Head changed
164 {
165 continue;
166 }
167
168 if (head == tail)
169 {
170 if (next == null)
171 {
172 item = default!;
173 return false;
174 }
175 // Tail is lagging – help advance it
176 Interlocked.CompareExchange(ref _tail, next, tail);
177 }
178 else
179 {
180 item = next!.Value;
181 if (Interlocked.CompareExchange(ref _head, next, head) == head)
182 {
183 Interlocked.Decrement(ref _count);
184 _pool.Return(head); // Return old dummy node to pool
185 return true;
186 }
187 }
188 }
