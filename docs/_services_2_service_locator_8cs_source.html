95 public bool TryGetService<TService>([NotNullWhen(true)] out TService? service)
96 where TService : class
97 {
98 if (!_collection.TryGetDescriptor(typeof(TService), out var descriptor))
99 {
100 service = null;
101 return false;
102 }
103
104 if (descriptor.Lifetime == ServiceLifetime.Scoped)
105 {
106 throw new InvalidOperationException($"Cannot resolve scoped service '{typeof(TService)}' from root locator. Use CreateScope() to obtain a scope.");
107 }
108
109 service = (TService)ResolveCore(descriptor);
110 return true;
111 }

158 private object ResolveCore(ServiceDescriptor descriptor)
159 {
160 return descriptor.Lifetime switch
161 {
162 ServiceLifetime.Singleton => GetOrCreateSingleton(descriptor),
163 ServiceLifetime.Transient => CreateTransient(descriptor),
164 _ => throw new InvalidOperationException($"Unexpected lifetime '{descriptor.Lifetime}'")
165 };
166 }
167
173 private object GetOrCreateSingleton(ServiceDescriptor descriptor)
174 {
175 // If an explicit instance was provided, use it.
176 if (descriptor.Instance != null)
177 {
178 return descriptor.Instance;
179 }
180 return _singletonCache.GetOrAdd(descriptor.ServiceType, (_, d) => CreateServiceInstance(d), descriptor);
181 }
182
188 [MethodImpl(MethodImplOptions.AggressiveInlining)]
189 private object CreateTransient(ServiceDescriptor descriptor)
190 => CreateServiceInstance(descriptor);
191
198 private object CreateServiceInstance(ServiceDescriptor descriptor)
199 {
200 if (descriptor.ImplementationType == null)
201 {
202 throw new InvalidOperationException($"No implementation type for {descriptor.ServiceType}");
203 }
204
205 var factory = _factoryCache.GetOrAdd(descriptor.ImplementationType, CreateFactory);
206 var instance = factory(this);
207
208 // Track only singletons (transients are not tracked).
209 if (descriptor.Lifetime == ServiceLifetime.Singleton)
210 {
211 _collection.TrackSingletonDisposable(instance);
212 }
213
214 return instance;
215 }

223 internal Func<IServiceLocator, object> CreateFactory(Type type)
224 {
225 var constructor = type.GetConstructors(BindingFlags.Public | BindingFlags.Instance).FirstOrDefault()
226 ?? throw new InvalidOperationException($"No public constructor for {type}");
227
228 var parameters = constructor.GetParameters();
229 if (parameters.Length == 0)
230 {
231 return _ => Activator.CreateInstance(type)!;
232 }
233
234 var paramExpr = Expression.Parameter(typeof(IServiceLocator), "locator");
235 var args = new Expression[parameters.Length];
236 for (int i = 0; i < parameters.Length; i++)
237 {
238 var paramType = parameters[i].ParameterType;
239 // Emit: (TService)locator.GetService<TService>() (using reflection-free call via expression)
240 var getServiceMethod = typeof(IServiceLocator).GetMethod(nameof(GetService))!.MakeGenericMethod(paramType);
241 args[i] = Expression.Call(paramExpr, getServiceMethod);
242 }
243
244 var newExpr = Expression.New(constructor, args);
245 var lambda = Expression.Lambda<Func<IServiceLocator, object>>(newExpr, paramExpr);
246 return lambda.Compile();
247 }
248 }
