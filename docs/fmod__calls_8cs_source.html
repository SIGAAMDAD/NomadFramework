137 private static string GetPlatformLibraryName(string baseName)
138 {
139 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
140 {
141 return $"{baseName}.dll";
142 }
143 else if (RuntimeInformation.IsOSPlatform(OSPlatform.OSX))
144 {
145 return $"{baseName}.dylib";
146 }
147 else
148 {
149 if (IsAndroid())
150 {
151 return $"lib{baseName}.so";
152 }
153
154 string arch = RuntimeInformation.ProcessArchitecture switch
155 {
156 Architecture.X64 => "x86_64",
157 Architecture.Arm64 => "aarch64",
158 Architecture.Armv6 => "aarch32", // TODO: check if this is ACTUALLY what it is
159 Architecture.X86 => "i386",
160 _ => throw new Exception("...what the fuck are you running this on")
161 };
162
163 return $"lib{baseName}-{arch}.so";
164 }
165 }

171 private static bool IsAndroid()
172 {
173 try
174 {
175 return File.Exists("/system/build.prop")

177 || Environment.GetEnvironmentVariable("ANDROID_ROOT") != null;
178 }

180 {
181 return false;
182 }
