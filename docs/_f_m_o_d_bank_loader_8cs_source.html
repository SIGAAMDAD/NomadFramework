69 public Result<TResource> Load<TResource, TId>( TId id ) {
70 if ( id is not string path ) {
71 throw new InvalidCastException();
72 }
73 try {
74 Console.WriteLine( $"Attempting to load bank file '{path}'..." );
75 FMODValidator.ValidateCall( _fmodSystem.StudioSystem.loadBankFile( path, FMOD.Studio.LOAD_BANK_FLAGS.NORMAL, out var bank ) );
76 FMODValidator.ValidateCall( bank.getID( out var guid ) );
77 _logger.PrintLine( $"FMODBankLoader.LoadBank: loaded bank '{path}'" );
78 _guidRepository.AddBankId( path, guid );
79
80 var resource = new FMODBankResource( bank );
81 if ( resource is TResource result ) {
82 return Result<TResource>.Success( result );
83 } else {
84 throw new InvalidCastException();
85 }
86 } catch ( FMODException e ) {
87 _logger.PrintError( $"FMODBankLoader.LoadBank: failed to load bank '{path}' - {e.Error}" );
88 throw;
89 }
90 }
