114 void ISaveWriterService.BeginSave( string name, GameVersion gameVersion ) {
115 var filepath = _slotRepository.AddSaveFile( name, false );
116
117 _writer = _fileSystem.OpenWrite( new MemoryFileWriteConfig { FilePath = AtomicWriterService.GetAtomicPathName() } ) as IMemoryFileWriteStream;
118 if ( _writer == null ) {
119 _category.PrintError( $"Couldn't create save file {filepath}!" );
120 return;
121 }
122
123 _category.PrintLine( $"Writing save data to {filepath}..." );
124
125 {
126 var header = new SaveHeader( name, gameVersion, _sections.Count, Checksum.Empty );
127 header.Serialize( _writer );
128 }
129 }
130
131 /*
132 ===============
133 EndSave
134 ===============
135 */
141 void ISaveWriterService.EndSave( string name, GameVersion gameVersion ) {
142 ArgumentGuard.ThrowIfNull( _writer );
143
144 int sectionCount = _sections.Count;
145 foreach ( var section in _sections ) {
146 section.Value.Dispose();
147 }
148
149 var checksum = Checksum.Compute( _writer.Buffer!.Span );
150
151 _writer.Seek( 0, System.IO.SeekOrigin.Begin );
152 var header = new SaveHeader( name, gameVersion, sectionCount, checksum );
153 header.Serialize( _writer );
154
155 _writer.Seek( 0, System.IO.SeekOrigin.End );
156
157 if ( _config.DebugLogging ) {
158 _category.PrintLine( $"\tMagic: {Constants.HEADER_MAGIC}" );
159 _category.PrintLine( $"Finalized save data file:" );
160 _category.PrintLine( $"\tSectionCount: {header.SectionCount}" );
161 _category.PrintLine( $"\tSaveName: {header.Name}" );
162 _category.PrintLine( $"\tChecksum64: {header.Checksum.Value}" );
163 _category.PrintLine( $"\tGameVersion: {header.Version.ToInt()}" );
164 }
165
166 // clear the memory buffer
167 var filepath = _slotRepository.AddSaveFile( name, false );
168 _atomicWriter.FinalizeSaveData( filepath, _writer );
169
170 _sections.Clear();
171 }
172
173 /*
174 ===============
175 AddSection
176 ===============
177 */
184 public ISaveSectionWriter AddSection( string sectionId ) {
185 ArgumentGuard.ThrowIfNull( _writer );
186
187 if ( _sections.ContainsKey( sectionId ) ) {
188 throw new DuplicateSectionException( sectionId );
189 }
190 var writer = new SaveSectionWriter( in _config, _logger, _category, sectionId, _writer );
191 _sections[sectionId] = writer;
192
193 if ( _config.LogSerializationTree ) {
194 _category.PrintLine( "$\t[Section] (NAME) {sectionId}" );
195 }
196
197 return writer;
198 }
