142 public T AddComponent<T>(Action<T>? initializer = null)
143 where T : IComponent, new()
144 {
145 T component;
146 if (typeof(Component).IsAssignableFrom(typeof(T)))
147 {
148 component = (T)(IComponent)_gameObject.AddComponent(typeof(T));
149 }
150 else
151 {
152 component = new T();

311 private void RegisterAttachedUnityComponents()
312 {
313 MonoBehaviour[] behaviours = _gameObject.GetComponents<MonoBehaviour>();
314 for (int i = 0; i < behaviours.Length; i++)
315 {
316 if (behaviours[i] is IComponent component)

342 internal static IGameObject WrapGameObject(GameObject gameObject)
343 {
344 MonoBehaviour[] behaviours = gameObject.GetComponents<MonoBehaviour>();
345 for (int i = 0; i < behaviours.Length; i++)
346 {
347 if (behaviours[i] is IGameObject wrapped)
348 {
349 return wrapped;
350 }
351 }
352
353 return new UnityGameObject(gameObject);

362 internal static bool TryGetBackingGameObject(IGameObject? gameObject, out GameObject unityGameObject)
363 {
364 if (gameObject is UnityGameObject wrapper)
365 {
366 unityGameObject = wrapper.GameObject;
367 return true;
368 }
369
370 if (gameObject is Component component)
371 {
372 unityGameObject = component.gameObject;
373 return true;
374 }
375
376 unityGameObject = null!;
377 return false;
378 }
