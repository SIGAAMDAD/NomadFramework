590 private T Convert<F, T>(in F from, AnyType type)
591 {
592 if (_type == AnyType.String && type != AnyType.String)
593 {
594 _value = type switch
595 {
596 AnyType.Boolean => new Union { Boolean = bool.Parse(_value.String!) },
597 AnyType.Int8 => new Union { Int8 = sbyte.Parse(_value.String!, CultureInfo.CurrentCulture) },
598 AnyType.Int16 => new Union { Int16 = short.Parse(_value.String!, CultureInfo.CurrentCulture) },
599 AnyType.Int32 => new Union { Int32 = int.Parse(_value.String!, CultureInfo.CurrentCulture) },
600 AnyType.Int64 => new Union { Int64 = long.Parse(_value.String!, CultureInfo.CurrentCulture) },
601 AnyType.UInt8 => new Union { UInt8 = byte.Parse(_value.String!, CultureInfo.CurrentCulture) },
602 AnyType.UInt16 => new Union { UInt16 = ushort.Parse(_value.String!, CultureInfo.CurrentCulture) },
603 AnyType.UInt32 => new Union { UInt32 = uint.Parse(_value.String!, CultureInfo.CurrentCulture) },
604 AnyType.UInt64 => new Union { UInt64 = ulong.Parse(_value.String!, CultureInfo.CurrentCulture) },
605 AnyType.Float32 => new Union { Float32 = float.Parse(_value.String!, CultureInfo.CurrentCulture) },
606 AnyType.Float64 => new Union { Float64 = double.Parse(_value.String!, CultureInfo.CurrentCulture) },
607 _ => throw new InvalidCastException(nameof(type))
608 };
609 }
610 else
611 {
612 _value = type switch
613 {
614 AnyType.Boolean => new Union { Boolean = Unsafe.As<F, bool>(ref Unsafe.AsRef(in from)) },
615 AnyType.Int8 => new Union { Int8 = Unsafe.As<F, sbyte>(ref Unsafe.AsRef(in from)) },
616 AnyType.Int16 => new Union { Int16 = Unsafe.As<F, short>(ref Unsafe.AsRef(in from)) },
617 AnyType.Int32 => new Union { Int32 = Unsafe.As<F, int>(ref Unsafe.AsRef(in from)) },
618 AnyType.Int64 => new Union { Int64 = Unsafe.As<F, long>(ref Unsafe.AsRef(in from)) },
619 AnyType.UInt8 => new Union { UInt8 = Unsafe.As<F, byte>(ref Unsafe.AsRef(in from)) },
620 AnyType.UInt16 => new Union { UInt16 = Unsafe.As<F, ushort>(ref Unsafe.AsRef(in from)) },
621 AnyType.UInt32 => new Union { UInt32 = Unsafe.As<F, uint>(ref Unsafe.AsRef(in from)) },
622 AnyType.UInt64 => new Union { UInt64 = Unsafe.As<F, ulong>(ref Unsafe.AsRef(in from)) },
623 AnyType.Float32 => new Union { Float32 = Unsafe.As<F, float>(ref Unsafe.AsRef(in from)) },
624 AnyType.Float64 => new Union { Float64 = Unsafe.As<F, double>(ref Unsafe.AsRef(in from)) },
625 AnyType.String => new Union { String = from!.ToString() },
626 _ => throw new InvalidCastException(nameof(type))
627 };
628 }
629 _type = type;
630 return Unsafe.As<F, T>(ref Unsafe.AsRef(in from));
