612 private T Convert<F, T>(in F from, AnyType type)
613 {
614 if (_type == AnyType.String && type != AnyType.String)
615 {
616 _value = type switch
617 {
618 AnyType.Boolean => new Union { Boolean = bool.Parse(_value.String!) },
619 AnyType.Int8 => new Union { Int8 = sbyte.Parse(_value.String!, CultureInfo.CurrentCulture) },
620 AnyType.Int16 => new Union { Int16 = short.Parse(_value.String!, CultureInfo.CurrentCulture) },
621 AnyType.Int32 => new Union { Int32 = int.Parse(_value.String!, CultureInfo.CurrentCulture) },
622 AnyType.Int64 => new Union { Int64 = long.Parse(_value.String!, CultureInfo.CurrentCulture) },
623 AnyType.UInt8 => new Union { UInt8 = byte.Parse(_value.String!, CultureInfo.CurrentCulture) },
624 AnyType.UInt16 => new Union { UInt16 = ushort.Parse(_value.String!, CultureInfo.CurrentCulture) },
625 AnyType.UInt32 => new Union { UInt32 = uint.Parse(_value.String!, CultureInfo.CurrentCulture) },
626 AnyType.UInt64 => new Union { UInt64 = ulong.Parse(_value.String!, CultureInfo.CurrentCulture) },
627 AnyType.Float32 => new Union { Float32 = float.Parse(_value.String!, CultureInfo.CurrentCulture) },
628 AnyType.Float64 => new Union { Float64 = double.Parse(_value.String!, CultureInfo.CurrentCulture) },
629 _ => throw new InvalidCastException(nameof(type))
630 };
631 }
632 else
633 {
634 _value = type switch
635 {
636 AnyType.Boolean => new Union { Boolean = Unsafe.As<F, bool>(ref Unsafe.AsRef(in from)) },
637 AnyType.Int8 => new Union { Int8 = Unsafe.As<F, sbyte>(ref Unsafe.AsRef(in from)) },
638 AnyType.Int16 => new Union { Int16 = Unsafe.As<F, short>(ref Unsafe.AsRef(in from)) },
639 AnyType.Int32 => new Union { Int32 = Unsafe.As<F, int>(ref Unsafe.AsRef(in from)) },
640 AnyType.Int64 => new Union { Int64 = Unsafe.As<F, long>(ref Unsafe.AsRef(in from)) },
641 AnyType.UInt8 => new Union { UInt8 = Unsafe.As<F, byte>(ref Unsafe.AsRef(in from)) },
642 AnyType.UInt16 => new Union { UInt16 = Unsafe.As<F, ushort>(ref Unsafe.AsRef(in from)) },
643 AnyType.UInt32 => new Union { UInt32 = Unsafe.As<F, uint>(ref Unsafe.AsRef(in from)) },
644 AnyType.UInt64 => new Union { UInt64 = Unsafe.As<F, ulong>(ref Unsafe.AsRef(in from)) },
645 AnyType.Float32 => new Union { Float32 = Unsafe.As<F, float>(ref Unsafe.AsRef(in from)) },
646 AnyType.Float64 => new Union { Float64 = Unsafe.As<F, double>(ref Unsafe.AsRef(in from)) },
647 AnyType.String => new Union { String = from!.ToString() },
648 _ => throw new InvalidCastException(nameof(type))
649 };
