74 public static InternString Intern(string? str)
75 {
76 if (string.IsNullOrEmpty(str))
77 {
78 return InternString.Empty;
79 }
80 if (_global._stringToIds.TryGetValue(str, out ulong existingId))
81 {
82 return new InternString(existingId);
83 }
84
85 lock (_global._newStringLock)
86 {
87 if (_global._stringToIds.TryGetValue(str, out existingId))
88 {
89 return new InternString(existingId);
90 }
91 ulong newId = _global._nextId++;
92 _global._stringToIds[str] = newId;
93 _global._idToString[newId] = str;
94 return new InternString(newId);
95 }
