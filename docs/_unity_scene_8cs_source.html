180 public T? FindChild<T>( string childName )
181 where T : class, IGameObject {
182 if ( string.IsNullOrWhiteSpace( childName ) ) {
183 return null;
184 }
185
186 RefreshChildren();
187
188 string[] segments = childName.Split( '/' );
189 IGameObject? current = null;
190 for ( int i = 0; i < _children.Count; i++ ) {
191 if ( _children[i].Name == segments[0] ) {
192 current = _children[i];
193 break;
194 }
195 }
196
197 if ( current == null ) {
198 return null;
199 }
200
201 if ( segments.Length == 1 ) {
202 return current as T;
203 }
204
205 string remainingPath = string.Join( "/", segments, 1, segments.Length - 1 );
206 return current.FindChild<T>( remainingPath );
207 }
