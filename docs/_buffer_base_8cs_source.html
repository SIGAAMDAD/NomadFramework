288 private unsafe void MemCpy(ReadOnlySpan<byte> src, int srcOffset, Span<byte> dst, int dstOffset, int length)
289 {
290 if (src.Length == 0 || dst.Length == 0)
291 {
292 return;
293 }
294
295 // Validate all indices and lengths
296 RangeGuard.ThrowIfLessThan(srcOffset, 0, nameof(srcOffset));
297 RangeGuard.ThrowIfLessThan(dstOffset, 0, nameof(dstOffset));
298 RangeGuard.ThrowIfLessThan(length, 0, nameof(length));
299
300 // Ensure the source region is entirely within the source span
301 RangeGuard.ThrowIfGreaterThan(srcOffset + length, src.Length, nameof(srcOffset));
302
303 // Ensure the destination region is entirely within the destination span
304 RangeGuard.ThrowIfGreaterThan(dstOffset + length, dst.Length, nameof(dstOffset));
305
306 // Compute the remaining space in destination (optional, but used in MemoryCopy call)
307 long destSize = dst.Length - dstOffset;
308
309 fixed (byte* srcPtr = &src[srcOffset])
310 fixed (byte* dstPtr = &dst[dstOffset])
311 {
312 System.Buffer.MemoryCopy(srcPtr, dstPtr, destSize, length);
313 }
314 }
