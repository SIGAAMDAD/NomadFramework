117 private unsafe ImmutableDictionary<string, AnimationFramesData> AllocFrames( AnimatedSprite2D animatedSprite, int index ) {
118 var spriteFrames = animatedSprite.SpriteFrames;
119 var animationNames = spriteFrames.GetAnimationNames();
120
121 var animations = new Dictionary<string, AnimationFramesData>( animationNames.Length );
122 int[] frameCounts = new int[ animationNames.Length ];
123 bool[] loops = new bool[ animationNames.Length ];
124 for ( int i = 0; i < animationNames.Length; i++ ) {
125 var animationName = animationNames[ i ];
126
127 int frameCount = spriteFrames.GetFrameCount( animationName );
128 frameCounts[ i ] = frameCount;
129
130 loops[ i ] = spriteFrames.GetAnimationLoop( animationName );
131
132 var rids = new Rid[ frameCount ];
133 var regions = new Rect2[ frameCount ];
134 var durations = new float[ frameCount ];
135 for ( int f = 0; f < frameCount; f++ ) {
136 var texture = spriteFrames.GetFrameTexture( animationName, f );
137 rids[ i ] = texture.GetRid();
138 regions[ i ] = GetTextureRegion( texture );
139 durations[ i ] = spriteFrames.GetFrameDuration( animationName, f );
140 }
141
142 fixed ( float* durationsPtr = durations )
143 fixed ( Rid* ridsPtr = rids )
144 fixed ( Rect2* regionsPtr = regions )
145 fixed ( bool* loopPtr = &loops[ i ] )
146 fixed ( int* frameCountPtr = &frameCounts[ i ] ) {
147 animations[ animationNames[ i ] ] = new AnimationFramesData(
148 durationsPtr,
149 ridsPtr,
150 regionsPtr,
151 loopPtr,
152 frameCountPtr
153 );
154 }
155
156 _framesBatchData.Loops[ index ] = loops;
157 _framesBatchData.FrameDurations[ index ] = durations;
158 _framesBatchData.TextureRids[ index ] = rids;
159 _framesBatchData.TextureRegions[ index ] = regions;
160 _framesBatchData.FrameCounts[ index ] = frameCounts;
161 }
162 return animations.ToImmutableDictionary();
163 }

255 private unsafe void UpdateFrameData( int* currentFrame, float* frameTimer, bool* playing, bool backwards, float speedScale, int index, float delta ) {
256 fixed ( bool* loopsPtr = _framesBatchData.Loops[ index ] )
257 fixed ( int* frameCountPtr = _framesBatchData.FrameCounts[ index ] )
258 fixed ( float* frameDurationsPtr = _framesBatchData.FrameDurations[ index ] ) {
259 *frameTimer += delta * speedScale;
260 while ( *frameTimer >= frameDurationsPtr[ *currentFrame ] ) {
261 *frameTimer -= frameDurationsPtr[ *currentFrame ];
262 if ( backwards ) {
263 (*currentFrame)--;
264 } else {
265 (*currentFrame)++;
266 }
267 if ( *currentFrame >= frameCountPtr[ index ] ) {
268 if ( loopsPtr[ index ] ) {
269 *currentFrame = 0;
270 } else {
271 *currentFrame = frameCountPtr[ index ] - 1;
272 *playing = false;
273 break;
274 }
275 }
276 }
