56 public ConfigFileWriter( string configFile, ILoggerCategory logger, ICVarSystemService cvarSystem, IFileSystem fileSystem, ICollection<CVarGroup> groups ) {
57 ArgumentGuard.ThrowIfNullOrEmpty( configFile );
58
59 _cvarSystem = cvarSystem;
60 logger.PrintLine( $"ConfigFileWriter: writing cvar config to {configFile}...." );
61
62 _writer = null;
63
64 try {
65 string? directoryPath = Path.GetDirectoryName( configFile );
66 if ( !string.IsNullOrWhiteSpace( directoryPath ) && !fileSystem.DirectoryExists( directoryPath ) ) {
67 fileSystem.CreateDirectory( directoryPath );
68 }
69
70 using ( _writer = fileSystem.OpenWrite( new FileWriteConfig { FilePath = configFile, Format = StreamFormat.Utf8 } ) as IFileWriteStream ) {
71 WriteHeader();
72 foreach ( var group in groups ) {
73 WriteGroup( group );
74 }
75 _writer.Flush();
76 }
77 } catch ( Exception e ) {
78 _writer?.Close();
79 logger.PrintError( $"...Error writing configuration file {configFile}: {e.Message}" );
80 throw;
81 }
82 }
83

92 private void WriteHeader() {
93 _writer.WriteLine( "; ===========================================================================" );
94 _writer.WriteLine( "; Configuration File - Auto-generated by Nomad.CVars System" );
95 _writer.WriteLine( $"; Generated on: {DateTime.Now:yyyy-MM-dd HH:mm:ss}" );
96 _writer.WriteLine( "; ===========================================================================" );
97 _writer.WriteLine( string.Empty );
98 }
99
100 /*
101 ===============
102 WriteVariable
103 ===============
104 */
109 private readonly void WriteVariable( ICVar cvar ) {
110 ArgumentGuard.ThrowIfNull( cvar );
111 switch ( cvar.Type ) {
112 case CVarType.Boolean:
113 _writer.WriteLine( $"{cvar.Name}={cvar.GetBooleanValue()}" );
114 break;
115 case CVarType.Int:
116 _writer.WriteLine( $"{cvar.Name}={cvar.GetIntegerValue()}" );
117 break;
118 case CVarType.UInt:
119 _writer.WriteLine( $"{cvar.Name}={cvar.GetUIntegerValue()}" );
120 break;
121 case CVarType.String:
122 _writer.WriteLine( $"{cvar.Name}=\"{cvar.GetStringValue()}\"" );
123 break;
124 case CVarType.Decimal:
125 _writer.WriteLine( $"{cvar.Name}={cvar.GetDecimalValue()}" );
126 break;
127 }
128 }
129
130 /*
131 ===============
132 WriteGroup
133 ===============
134 */
139 private void WriteGroup( ICVarGroup group ) {
140 ArgumentGuard.ThrowIfNull( group );
141
142 _writer.WriteLine( string.Empty );
143 _writer.WriteLine( $"[{group.Name}]" );
144
145 foreach ( var cvar in group.CVars ) {
146 WriteVariable( _cvarSystem.GetCVar( cvar.Name ) );
147 }
148 }
149 };
