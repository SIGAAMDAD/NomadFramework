115 private void OnNetConnectionStatusChanged( SteamNetConnectionStatusChangedCallback_t pCallback ) {
116 CSteamID senderId = pCallback.m_info.m_identityRemote.GetSteamID();
117
118 var connection = _connectionCache.GetOrAdd( senderId, f => new SteamNetConnection( pCallback.m_hConn ) );
119 switch ( pCallback.m_info.m_eState ) {
120 case ESteamNetworkingConnectionState.k_ESteamNetworkingConnectionState_Connected:
121 connection.SetStatus( SteamConnectionStatus.Connected );
122 break;
123 case ESteamNetworkingConnectionState.k_ESteamNetworkingConnectionState_Connecting:
124 connection.SetStatus( SteamConnectionStatus.Opened );
125 OnIncomingConnection( senderId );
126 break;
127 case ESteamNetworkingConnectionState.k_ESteamNetworkingConnectionState_Dead:
128 case ESteamNetworkingConnectionState.k_ESteamNetworkingConnectionState_ProblemDetectedLocally:
129 case ESteamNetworkingConnectionState.k_ESteamNetworkingConnectionState_ClosedByPeer:
130 connection.SetStatus( SteamConnectionStatus.Closed );
131 _clientDisconnected.Publish( (ulong)senderId );
132 break;
133 }
