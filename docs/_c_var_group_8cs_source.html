68 public void AddCVar( ICVar cvar ) {
69 ArgumentGuard.ThrowIfNull( cvar );
70
71 if ( _cvars.Contains( cvar ) ) {
72 _logger.PrintWarning( $"CVarGroup.AddCVar: CVar '{cvar.Name}' already in group '{Name}'!" );
73 return;
74 }
75 _cvars.Add( cvar );
76 }
77
78 /*

82 */
87 public void RemoveCVar( ICVar cvar ) {
88 ArgumentGuard.ThrowIfNull( cvar );
89
90 if ( !_cvars.Contains( cvar ) ) {
91 _logger.PrintWarning( $"CVarGroup.RemoveCVar: CVar '{cvar.Name}' not in group '{Name}'!" );
92 return;
