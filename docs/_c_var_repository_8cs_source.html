74 public ICVar<T> AddCVar<T>( CVarCreateInfo<T> createInfo, IGameEventRegistryService eventFactory ) {
75 ArgumentGuard.ThrowIfNullOrEmpty( createInfo.Name );
76 ArgumentGuard.ThrowIfNullOrEmpty( createInfo.Description );
77
78 var name = new InternString( createInfo.Name );
79 if ( _cvars.TryGetValue( name, out ICVar? var ) ) {
80 if ( var is ICVar<T> value ) {
81 return value;
82 }
83 throw new InvalidCastException( $"CVar {createInfo.Name} found in CVarSystem cache isn't a valid CVar object!" );
84 }
85
86 ICVar<T> cvar = new CVar<T>( eventFactory, createInfo );
87 _cvars.TryAdd( name, cvar );
88 return cvar;
89 }

134 public bool TryFind<T>( string name, out ICVar<T>? cvar ) {
135 ArgumentGuard.ThrowIfNullOrEmpty( name );
136
137 if ( _cvars.TryGetValue( new InternString( name ), out ICVar? var ) && var is ICVar<T> typedVar ) {

204 public ICVar<T>? GetCVar<T>( string name ) {
205 ArgumentGuard.ThrowIfNullOrEmpty( name );
206
207 if ( !_cvars.TryGetValue( new InternString( name ), out var cvar ) ) {
208 _logger.PrintError( $"CVarSystem.GetCVar: no cvar found for name '{name}'!" );
209 return null;
210 }
211 return cvar.Type == typeof(T).GetCVarType() ? cvar as ICVar<T> : throw new CVarTypeMismatchException( typeof( T ), cvar.Type.GetSystemType() );
212 }

228 _logger.PrintError( $"CVarSystem.GetCVar: no cvar found for name '{name}'!" );
229 return null;
230 }
231 return cvar;
232 }

238 */
244 public ICVar<T>[] GetCVarsWithValueType<T>() {
245 List<ICVar<T>> cvars = new List<ICVar<T>>();
246
247 foreach ( var cvar in _cvars ) {
248 if ( cvar.Value is ICVar<T> typedVar ) {
249 cvars.Add( typedVar );
250 }
251 }
