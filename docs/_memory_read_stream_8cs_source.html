579 public string ReadString() {
580 StateGuard.ThrowIfDisposed( isDisposed, this );
581
582 int byteCount = Read7BitEncodedInt();
583 if ( byteCount == 0 ) {
584 return string.Empty;
585 }
586
587 string value;
588 unsafe {
589 fixed ( byte* src = buffer!.GetSlice( (int)position, byteCount ) ) {
590 value = Encoding.UTF8.GetString( src, byteCount );
591 }
592 }
593 position += byteCount;
594 return value;
595 }
596
597 /*
598 ===============
599 Read7BitEncodedInt
600 ===============
601 */
607 public int Read7BitEncodedInt() {
608 StateGuard.ThrowIfDisposed( isDisposed, this );
609
610 int value = 0;
611 int shift = 0;
612 byte b;
613
614 do {
615 b = ReadUInt8();
616 value |= (b & 0x7F) << shift;
617 shift += 7;
618 if ( shift > 35 ) {
619 throw new FormatException( "Invalid 7-bit encoded integer formatting in stream." );
620 }
621 } while ( (b & 0x80) != 0 );
622
623 return value;
624 }

636 [MethodImpl( MethodImplOptions.AggressiveInlining )]
637 private T Read<T>( int size )
638 where T : unmanaged {
639 StateGuard.ThrowIfDisposed( isDisposed, this );
640 if ( position + size > length ) {
641 throw new EndOfStreamException();
642 }
643
644 T value;
645 unsafe {
646 fixed ( void* ptr = buffer!.GetSlice( (int)position, size ) ) {
647 value = Unsafe.ReadUnaligned<T>( ptr );
648 }
649 }
650 position += size;
651 return value;
