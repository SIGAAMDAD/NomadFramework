115 public float CalculateActualPriority(
116 float startTime,
117 Vector2 position,
118 float basePriority,
119 SoundCategory category,
120 float? lastTimePlayed,
121 int consecutiveStealCount
122 ) {
123 Vector2 listener = _listenerService.ActiveListener;
124 float dx = position.X - listener.X;
125 float dy = position.Y - listener.Y;
126 float distance = MathF.Sqrt( dx * dx + dy * dy );
127
128 float priority = basePriority * category.Config.PriorityScale * CalculateDistanceFactor( distance );
129
130 if ( lastTimePlayed.HasValue ) {
131 float timeSinceLast = startTime - lastTimePlayed.Value;
132 float timePenalty = timeSinceLast >= 0.5f ? 0f : 1f - (timeSinceLast * 2f);
133 priority *= 1f - (timePenalty * 0.5f);
134 }
135
136 float freqPenalty = consecutiveStealCount >= 5 ? 0.5f : consecutiveStealCount * 0.1f;
137 priority *= 1f - (freqPenalty * _frequencyPenalty);
138
139 return Math.Clamp( priority, 0.01f, 1.0f );
140 }
141
142 /*
143 ===============
144 CalculateDistanceFactor
145 ===============
146 */
152 public float CalculateDistanceFactor( float distance ) {
153 if ( distance <= _distanceFalloffStart ) {
154 return 1.0f;
155 }
156 if ( distance >= _distanceFalloffEnd ) {
157 return 0.1f;
158 }
159 float t = (distance - _distanceFalloffStart) * _invDistanceFalloffRange;
160 return 1.0f - (t * 0.9f); // 1.0 -> 0.1 linearly
161 }

163 /*
164 ===============
165 CalculateTimePenalty
166 ===============
167 */
174 private static float CalculateTimePenalty( float startTime, float? lastTimePlayed ) {
175 if ( !lastTimePlayed.HasValue ) {
176 return 0.0f;
177 }
178
179 float timeSinceLast = startTime - lastTimePlayed.Value;
180 const float protectionTime = 0.5f;
181
182 return timeSinceLast > protectionTime ?
183 0.0f
184 :
185 1.0f - (timeSinceLast / protectionTime);
186 }

199 return Math.Clamp( consecutiveStealCount * 0.1f, 0.0f, 0.5f );
200 }
