61 InputStateService

67 public InputStateService() {
68 long totalBytes = 0;
69 long pressedBytes = PadBytes( sizeof( ulong ) * DEVICE_SLOT_COUNT * WORDS_PER_DEVICE, Core.Constants.WORDSIZE );
70 long axis1DBytes = PadBytes( sizeof( float ) * DEVICE_SLOT_COUNT * CONTROL_COUNT, Core.Constants.WORDSIZE );
71 long axis2DBytes = PadBytes( sizeof( float ) * 2 * DEVICE_SLOT_COUNT * CONTROL_COUNT, Core.Constants.WORDSIZE );
72 totalBytes = pressedBytes + axis1DBytes + axis2DBytes;
73
74#if NET6_0_OR_GREATER
75 _memory = NativeMemory.AlignedAlloc( (nuint)totalBytes, Core.Constants.WORDSIZE );
76#else
77 _memory = (void*)Marshal.AllocHGlobal( (int)totalBytes );
78#endif
79 _pressedBits = (ulong*)_memory;
80 _axis1D = (float*)((byte*)_memory + pressedBytes);
81 _axis2D = (Vector2*)((byte*)_memory + pressedBytes + axis1DBytes);
82
83 new Span<byte>( _memory, (int)totalBytes ).Clear();
84 }
85
86 public void Dispose() {
87 if ( _isDisposed ) {
88 return;
89 }
90
91 if ( _memory != null ) {
92#if NET6_0_OR_GREATER
93 NativeMemory.AlignedFree( _memory );
94#else
95 Marshal.FreeHGlobal( (nint)_memory );
96#endif
97 }
98
99 _isDisposed = true;
100 GC.SuppressFinalize( this );
101 }

118 /*
119 ===============
120 IsPressed
121 ===============
122 */
129 [MethodImpl( MethodImplOptions.AggressiveInlining )]
130 public bool IsPressed( InputDeviceSlot slot, InputControlId control ) {
131 int controlIndex = (int)control;
132 ulong* words = GetPressedWords( slot );
133 return (words[controlIndex >> 6] & (1UL << (controlIndex & 63))) != 0UL;
134 }
135
136 /*
137 ===============
138 SetPressed
139 ===============
140 */
147 [MethodImpl( MethodImplOptions.AggressiveInlining )]
148 public void SetPressed( InputDeviceSlot slot, InputControlId control, bool pressed ) {
149 int controlIndex = (int)control;
150 ulong* words = GetPressedWords( slot );
151 ulong mask = 1UL << (controlIndex & 63);
152
153 if ( pressed ) {
154 words[controlIndex >> 6] |= mask;
155 } else {
156 words[controlIndex >> 6] &= ~mask;
157 }
158 }
