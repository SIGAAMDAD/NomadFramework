60 [MethodImpl( MethodImplOptions.AggressiveInlining )]
61 public bool IsPressed( InputDeviceSlot slot, InputControlId control ) {
62 int controlIndex = (int)control;
63 int baseWord = ((int)slot * WORDS_PER_DEVICE) + (controlIndex >> 6);
64 ulong mask = 1UL << (controlIndex & 63);
65 return (_pressedBits[baseWord] & mask) != 0UL;
66 }
