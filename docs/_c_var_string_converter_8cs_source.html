63 private static bool TryParseBool( string s, out object? result ) {
64 if ( bool.TryParse( s, out bool value ) ) {
65 result = value;
66 return true;
67 }
68 result = null;
69 return false;
70 }

86 private static bool TryParsePrimitive<T>( string s, out object? result, NumberStyles style, TryParseUnmanaged<T> parser )
87 where T : unmanaged {
88 if ( parser.Invoke( s, style, CultureInfo.InvariantCulture, out var value ) ) {
89 result = value;
90 return true;

96 /*

98 TryParse
99 ===============
100 */
109 public static bool TryParse( string str, Type targetType, out object? result ) {
110 if ( _parsers.TryGetValue( targetType, out var parser ) ) {
111 return parser.Invoke( str, out result );
112 }
113 if ( targetType.IsEnum ) {
114 return EnumTryParse( str, targetType, out result );
115 }
116 throw new NotSupportedException( $"CVars do not support type {targetType.Name}" );
117 }

119 /*
120 ===============
121 EnumTryParse
122 ===============
123 */
131 private static bool EnumTryParse( string str, Type targetType, out object? result ) {
132 // we only use Enum.IsDefined here because if we don't there's a bug where even if the integral value itself exceeds the enum, it'll still get parsed as valid. So we need a manual
133 // bounds check here to ensure that never happens.
134 if ( Enum.TryParse( targetType, str, ignoreCase: true, out result ) && Enum.IsDefined( targetType, result ) ) {
135 return true;
136 }
137 if ( TryParse( str, Enum.GetUnderlyingType( targetType ), out object? underlying ) && Enum.IsDefined( targetType, underlying ) ) {
138 result = Enum.ToObject( targetType, underlying );
139 return true;
140 }
141 result = null;
142 return false;
143 }
