239 /*
240 ===============
241 SetActionBindings
242 ===============
243 */
251 public bool SetActionBindings( string mappingName, string actionId, ImmutableArray<InputBindingDefinition> bindings ) {
252 ThrowIfDisposed();
253
254 if ( string.IsNullOrWhiteSpace( mappingName ) || string.IsNullOrWhiteSpace( actionId ) ) {
255 return false;
256 }
257 if ( !_loadedMappings.TryGetValue( mappingName, out var mapping ) ) {
258 return false;
259 }
260
261 var builder = mapping.Actions.ToBuilder();
262 for ( int i = 0; i < builder.Count; i++ ) {
263 if ( builder[ i ].Id.Equals( actionId, StringComparison.Ordinal ) ) {
264 builder[ i ] = new InputActionDefinition(
265 builder[ i ].Name,
266 builder[ i ].Id,
267 builder[ i ].ValueType,
268 bindings.Clone()
269 );
270
271 _loadedMappings = _loadedMappings.SetItem( mappingName, new LoadedBindMapping( mappingName, builder.ToImmutable() ) );
272 ValidateActiveMappings();
273 RebuildCaches();
274 return true;
275 }
276 }
277
278 if ( !_bindMappings.TryGetValue( mappingName, out var mergedMapping ) ) {
279 return false;
280 }
281
282 for ( int i = 0; i < mergedMapping.Length; i++ ) {
283 if ( !mergedMapping[ i ].Id.Equals( actionId, StringComparison.Ordinal ) ) {
284 continue;
285 }
286
287 builder.Add( new InputActionDefinition( mergedMapping[ i ].Name, mergedMapping[ i ].Id, mergedMapping[ i ].ValueType, bindings.Clone() ) );
288 _loadedMappings = _loadedMappings.SetItem( mappingName, new LoadedBindMapping( mappingName, builder.ToImmutable() ) );
289 ValidateActiveMappings();
290 RebuildCaches();
291 return true;
292 }
293
294 return false;
295 }

320 ===============
321 LoadAllBindMappings
322 ===============
323 */
327 private void LoadAllBindMappings() {
328 var mappingBuilder = ImmutableDictionary.CreateBuilder<string, LoadedBindMapping>( StringComparer.OrdinalIgnoreCase );
329 var files = _fileSystem.GetFiles( Constants.BINDINGS_DIRECTORY, "*.json", true );
330
331 for ( int i = 0; i < files.Count; i++ ) {
332 string filePath = files[i];
333 if ( string.Equals( filePath, _defaultsPath, StringComparison.OrdinalIgnoreCase ) ) {
334 continue;
335 }
336 if ( !_loader.LoadBindDatabase( filePath, out var mappingBindings ) ) {
337 continue;
338 }
339
340 string mappingName = GetBindMappingName( filePath );
341 if ( mappingBuilder.ContainsKey( mappingName ) ) {
342 throw new InvalidOperationException( $"Duplicate binding mapping name '{mappingName}' discovered while loading '{filePath}'." );
343 }
344
345 _category.PrintLine( $"Loaded bind mapping '{mappingName}'..." );
346
347 mappingBuilder[mappingName] = new LoadedBindMapping( mappingName, mappingBindings );
348 }
349
350 _loadedMappings = mappingBuilder.ToImmutable();
351 RefreshActiveMappingsFromSettings();
352 ValidateActiveMappings();
353 RebuildCaches();
354 }

384 /*
385 ===============
386 MergeActions
387 ===============
388 */
395 private static ImmutableArray<InputActionDefinition> MergeActions( IEnumerable<ImmutableArray<InputActionDefinition>> sources ) {
396 var actionIndices = new Dictionary<string, int>( StringComparer.Ordinal );
397 var builder = ImmutableArray.CreateBuilder<InputActionDefinition>();
398
399 foreach ( var source in sources ) {
400 for ( int i = 0; i < source.Length; i++ ) {
401 var action = source[i];
402 if ( !actionIndices.TryGetValue( action.Id, out int actionIndex ) ) {
403 actionIndices.Add( action.Id, builder.Count );
404 builder.Add( action );
405 continue;
406 }
407
408 var existing = builder[actionIndex];
409 if ( existing.ValueType != action.ValueType ) {
410 throw new InvalidOperationException( $"Binding action '{action.Id}' has conflicting value types across binding databases." );
411 }
412
413 builder[actionIndex] = new InputActionDefinition(
414 existing.Name,
415 existing.Id,
416 existing.ValueType,
417 OverrideBindingsByScheme( existing.Bindings, action.Bindings )
418 );
419 }
420 }
421
422 return builder.ToImmutable();
423 }

425 /*
426 ===============
427 OverrideBindingsByScheme
428 ===============
429 */
436 private static ImmutableArray<InputBindingDefinition> OverrideBindingsByScheme( ImmutableArray<InputBindingDefinition> left, ImmutableArray<InputBindingDefinition> right ) {
437 if ( right.IsDefaultOrEmpty ) {
438 return left;
439 }
440
441 var overriddenSchemes = ImmutableHashSet.CreateBuilder<InputScheme>();
442 for ( int i = 0; i < right.Length; i++ ) {
443 overriddenSchemes.Add( right[ i ].Scheme );
444 }
445
446 var builder = ImmutableArray.CreateBuilder<InputBindingDefinition>( left.Length + right.Length );
447
448 for ( int i = 0; i < left.Length; i++ ) {
449 if ( overriddenSchemes.Contains( left[ i ].Scheme ) ) {
450 continue;
451 }
452 builder.Add( left[ i ] );
453 }
454 for ( int i = 0; i < right.Length; i++ ) {
455 if ( ContainsBinding( builder, right[ i ] ) ) {
456 continue;
457 }
458 builder.Add( right[ i ] );
459 }
460
461 return builder.ToImmutable();
462 }

509 private void ValidateActiveMappings() {
510 var builder = _activeMappings.ToBuilder();
511 foreach ( var pair in _activeMappings ) {
512 if ( !_loadedMappings.TryGetValue( pair.Value, out var mapping ) || !mapping.Schemes.Contains( pair.Key ) ) {

519 /*
520 ===============
521 RebuildCaches
522 ===============
523 */
527 private void RebuildCaches() {
528 var mergedMappings = ImmutableDictionary.CreateBuilder<string, ImmutableArray<InputActionDefinition>>( StringComparer.OrdinalIgnoreCase );
529 foreach ( var pair in _loadedMappings ) {
530 mergedMappings[pair.Key] = MergeActions( _defaultBindings, pair.Value.Actions );
531 }
532
533 _bindMappings = mergedMappings.ToImmutable();
534
535 var sources = new List<ImmutableArray<InputActionDefinition>> { _defaultBindings };
536 var includedMappings = new HashSet<string>( StringComparer.OrdinalIgnoreCase );
537 foreach ( var pair in _activeMappings ) {
538 if ( !includedMappings.Add( pair.Value ) ) {
539 continue;
540 }
541 if ( _loadedMappings.TryGetValue( pair.Value, out var mapping ) ) {
542 sources.Add( mapping.Actions );
543 }
544 }
545
546 _allBindings = MergeActions( sources );
547 }

614 private static ImmutableHashSet<InputScheme> GetSchemes( ImmutableArray<InputActionDefinition> actions ) {
615 var builder = ImmutableHashSet.CreateBuilder<InputScheme>();
616 for ( int i = 0; i < actions.Length; i++ ) {
617 for ( int j = 0; j < actions[ i ].Bindings.Length; j++ ) {
618 builder.Add( actions[ i ].Bindings[ j ].Scheme );
