146 public IGameEvent<TArgs> GetEvent<TArgs>( string name, string nameSpace, EventFlags flags = EventFlags.Default )
147 where TArgs : struct
148 {
149 var internedName = new InternString( name );
150 var internedNameSpace = new InternString( nameSpace );
151 var key = new EventKey(
152 name: internedName,
153 nameSpace: internedNameSpace,
154 argsType: typeof( TArgs )
155 );
156
157 if ( _eventCache.TryGetValue( key, out IGameEvent? value ) ) {
158 return (IGameEvent<TArgs>)value;
159 }
160
161 value = new GameEvent<TArgs>( internedNameSpace, internedName, _logger, flags );
162 _eventCache.TryAdd( key, value );
163 return (IGameEvent<TArgs>)value;
164 }

179 public bool TryGetEvent<TArgs>( string name, string nameSpace, out IGameEvent<TArgs>? gameEvent )
180 where TArgs : struct
181 {
182 var key = new EventKey(
183 name: new InternString( name ),
184 nameSpace: new InternString( nameSpace ),
185 argsType: typeof( TArgs )
186 );
187 if ( _eventCache.TryGetValue( key, out IGameEvent? @event ) ) {
188 if ( @event is IGameEvent<TArgs> typedEvent ) {
189 gameEvent = typedEvent;
190 return true;
191 }
192 }
193 gameEvent = null;
194 return false;
195 }

214 nameSpace: new InternString( nameSpace ),
215 argsType: typeof( TArgs )
216 );
217 if ( _eventCache.TryRemove( key, out IGameEvent? eventObj ) ) {
