158 private void OnCommandExecuted( in TextEnteredEventArgs textEntered ) {
159 string textCommand = _commandBuilder.ArgumentCount > 0 ? _commandBuilder.GetArgumentAt( 0 ) : string.Empty;
160
161 if ( !string.IsNullOrEmpty( textCommand ) && _commandService.TryGetCommand( textCommand, out ConsoleCommand consoleCommand ) ) {
162 var arguments = _commandBuilder.GetArgs();
163
164 try {
165 _commandExecuted.Publish( new CommandExecutedEventArgs( consoleCommand, arguments.Length ) );
166 } catch ( Exception ex ) {
167 _logger.PrintError( $"CommandLine.OnCommandExecuted: error executing command - {ex.Message}" );
168 throw;
169 }
170 } else if ( !string.IsNullOrEmpty( textCommand ) ) {
171 _unknownCommand.Publish( new CommandExecutedEventArgs() );
172 _logger.PrintWarning( $"CommandLine.OnCommandExecuted: command '{textCommand}' not found." );
173 }
174 }
