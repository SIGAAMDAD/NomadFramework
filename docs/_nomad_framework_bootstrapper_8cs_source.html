40 public NomadFrameworkBootstrapper(IServiceRegistry registry, IServiceLocator locator)
41 {
42 ArgumentGuard.ThrowIfNull(registry);
43 ArgumentGuard.ThrowIfNull(locator);

46 _locator = locator;

53 public void Dispose()
54 {
55 if (!_isDisposed)
56 {
57 for (int i = _bootstrappers.Count - 1; i >= 0; i--)
58 {
59 _bootstrappers[i].Shutdown();
60 }
61 }
62 GC.SuppressFinalize(this);
63 _isDisposed = true;
64 }

71 public NomadFrameworkBootstrapper AddBootstrapper<T>()

93 public void Bootstrap()
94 {
95 for (int i = 0; i < _bootstrappers.Count; i++)

97 _bootstrappers[i].Initialize(_registry, _locator);
98 }
99
100 StartSystems();

106 private void StartSystems()
107 {
108 var startables = _locator.GetServices<IStartable>();
