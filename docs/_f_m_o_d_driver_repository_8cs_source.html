83 public FMODDriverRepository( ILoggerService logger, ICVarSystemService cvarSystem, FMOD.System system ) {
84 _logger = logger;
85 _system = system;
86
87 _supportedAudioDrivers = new Dictionary<FMOD.OUTPUTTYPE, string>() {
88 [FMOD.OUTPUTTYPE.AUTODETECT] = "Auto Detect",
89#if WINDOWS
90 [FMOD.OUTPUTTYPE.ASIO] = "ASIO",
91 [FMOD.OUTPUTTYPE.WASAPI] = "WasAPI",
92 [FMOD.OUTPUTTYPE.WINSONIC] = "WinSonic",
93#elif LINUX
94 [ FMOD.OUTPUTTYPE.ALSA ] = "ALSA",
95 [ FMOD.OUTPUTTYPE.PULSEAUDIO ] = "PulseAudio"
96#endif
97 }.ToImmutableDictionary();
98
99 var audioDriver = cvarSystem.GetCVarOrThrow<string>( Constants.CVars.EngineUtils.Audio.AUDIO_DRIVER );
100 audioDriver.ValueChanged.Subscribe( OnAudioDriverValueChanged );
101
102 var outputDeviceIndex = cvarSystem.GetCVarOrThrow<int>( Constants.CVars.EngineUtils.Audio.OUTPUT_DEVICE_INDEX );
103 outputDeviceIndex.ValueChanged.Subscribe( OnAudioDeviceValueChanged );
104
105 FMODValidator.ValidateCall( _system.setCallback( OnAudioOutputDeviceListChanged, FMOD.SYSTEM_CALLBACK_TYPE.DEVICELISTCHANGED ) );
106
107 GetAudioDeviceData();
108 FMODValidator.ValidateCall( _system.getOutput( out _audioDriver ) );
109 }

132 public void SetOutputDevice( int deviceIndex ) {
133 if ( deviceIndex < 0 || deviceIndex >= _devices.Length ) {
134 throw new ArgumentOutOfRangeException( nameof( deviceIndex ) );
135 }
136 var device = _devices[deviceIndex];
137
138 _logger.PrintLine( $"FMODDriverRepository.SetOutputDevice: setting output audio device to '{device.Name}'..." );
139 FMODValidator.ValidateCall( _system.setDriver( deviceIndex ) );
140 _outputDeviceIndex = deviceIndex;
141 }

143 /*
144 ===============
145 GetAudioDeviceData
146 ===============
147 */
151 private void GetAudioDeviceData() {
152 FMODValidator.ValidateCall( _system.getNumDrivers( out int numDrivers ) );
153
154 _devices = new FMODDeviceInfo[numDrivers];
155 for ( int i = 0; i < numDrivers; i++ ) {
156 FMODValidator.ValidateCall( _system.getDriverInfo( i, out string name, 256, out var guid, out int systemRate, out FMOD.SPEAKERMODE speakerMode, out int speakerChannels ) );
157 _devices[i] = new FMODDeviceInfo( name, guid, systemRate, speakerMode, speakerChannels );
158 _logger.PrintLine( $"FMODDevice.GetAudioDeviceData: found audio output device '{name}' - speakerMode = '{speakerMode}', channelCount = '{speakerChannels}'" );
159 }
160
161 _system.getDriver( out _outputDeviceIndex );
162 }

173 private void OnAudioDriverValueChanged( in CVarValueChangedEventArgs<string> args ) {
174 FMOD.OUTPUTTYPE outputType = FMOD.OUTPUTTYPE.MAX;
175
176 foreach ( var drivers in _supportedAudioDrivers ) {
177 if ( drivers.Value.Equals( args.NewValue, StringComparison.InvariantCulture ) ) {
178 outputType = drivers.Key;
179 break;
180 }
181 }
182 if ( outputType == FMOD.OUTPUTTYPE.MAX ) {
183 _logger.PrintError( $"FMODDriverRepository.OnAudioDriverValueChanged: invalid audio driver name '{args.NewValue}'" );
184 return;
185 }
186
187 _audioDriver = outputType;
188 _logger.PrintLine( $"FMODDriverRepository.OnAudioDriverValueChanged: setting audio driver API to '{_audioDriver}'..." );
189 FMODValidator.ValidateCall( _system.setOutput( _audioDriver ) );
190 }
