88 public override void _Input( InputEvent @event ) {
89 var now = DateTime.UtcNow.ToFileTimeUtc();
90 switch ( @event ) {
91 case InputEventKey keyEvent:
92 _keyboardEvent.Publish(
93 new KeyboardEvent(
94 GodotKeyToNomadKey( keyEvent.GetPhysicalKeycodeWithModifiers() ),
95 now,
96 keyEvent.Pressed
97 )
98 );
99 break;
100 case InputEventMouseButton mouseButton:
101 _mouseButtonEvent.Publish(
102 new MouseButtonEvent(
103 GodotMouseButtonToNomadMouseButton( mouseButton.ButtonIndex ),
104 now,
105 mouseButton.Pressed
106 )
107 );
108 break;
109 case InputEventMouseMotion mouseMotion:
110 _mouseMotionEvent.Publish(
111 new MouseMotionEvent(
112 now,
113 (int)mouseMotion.Position.X,
114 (int)mouseMotion.Position.Y
115 )
116 );
117 break;
118 case InputEventJoypadButton joypadButton:
119 _gamepadButtonEvent.Publish(
120 new GamepadButtonEvent(
121 GodotJoypadButtonToNomadGamepadButton( joypadButton.ButtonIndex ),
122 joypadButton.Device,
123 now,
124 joypadButton.Pressed
125 )
126 );
127 break;
128 case InputEventJoypadMotion joypadMotion:
129 _gamepadAxisEvent.Publish(
130 new GamepadAxisEvent(
131 GodotStickToNomadGamepadStick( joypadMotion.Axis ),
132 now,
133 joypadMotion.Device,
134 GodotAxisValueToVector2( joypadMotion )
135 )
136 );
137 break;
138 case InputEventScreenDrag screenDrag:
139 break;
140 case InputEventScreenTouch screenTouch:
141 break;
142 }
143 }

156 [MethodImpl( MethodImplOptions.AggressiveInlining )]
157 private static System.Numerics.Vector2 GodotAxisValueToVector2( InputEventJoypadMotion motion ) {
158 return motion.Axis switch {
159 JoyAxis.LeftX or JoyAxis.RightX => new System.Numerics.Vector2( motion.AxisValue, 0.0f ),
160 JoyAxis.LeftY or JoyAxis.RightY => new System.Numerics.Vector2( 0.0f, motion.AxisValue ),
161 _ => throw new ArgumentOutOfRangeException( nameof( motion ) ),
162 };
163 }
