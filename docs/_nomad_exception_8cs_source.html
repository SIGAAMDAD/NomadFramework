110 private static string? ResolveNomadModuleInfo()
111 {
112 var thisAssembly = typeof(NomadException).Assembly;
113
114 // skip 1 to move past this method's own fame
115 var st = new StackTrace(skipFrames: 1, fNeedFileInfo: true);
116 var frames = st.GetFrames();
117 if (frames == null)
118 {
119 return null;
120 }
121
122 for (int i = 0; i < frames.Length; i++)
123 {
124 var method = frames[i].GetMethod();
125 if (method == null)
126 {
127 continue;
128 }
129
130 var asm = method.Module.Assembly;
131 if (asm == thisAssembly)
132 {
133 continue;
134 }
135
136 var attribute = asm.GetCustomAttribute<NomadModule>();
137 if (attribute != null)
138 {
139 return attribute.Name;
140 }
141 }
142 return null;
