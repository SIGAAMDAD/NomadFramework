142 internal GodotWindowService( SceneTree sceneTree, IGameEventRegistryService eventFactory ) {
143 _window = sceneTree.Root.GetWindow();
144 _window.FocusEntered += OnFocusEntered;
145 _window.FocusExited += OnFocusExited;
146 _window.SizeChanged += OnSizeChanged;
147 _window.CloseRequested += OnCloseRequested;
148 _window.TitleChanged += OnTitleChanged;
149
150 _sizeChanged = eventFactory.GetEvent<WindowSizeChangedEventArgs>( Core.Constants.Events.EngineUtils.WINDOW_SIZE_CHANGED, Core.Constants.Events.EngineUtils.NAMESPACE );
151 _closeRequested = eventFactory.GetEvent<EmptyEventArgs>( Core.Constants.Events.EngineUtils.CLOSE_REQUESTED, Core.Constants.Events.EngineUtils.NAMESPACE );
152 _focusChanged = eventFactory.GetEvent<bool>( Core.Constants.Events.EngineUtils.FOCUS_CHANGED, Core.Constants.Events.EngineUtils.NAMESPACE );

154 _monitors = GetScreenList();

188 public IReadOnlyList<WindowResolution> GetSupportedResolutions( int monitorIndex ) {
189 var monitor = _monitors[monitorIndex];
190 var resolutions = new List<WindowResolution>();
191
192 for ( var resolution = WindowResolution.Min; resolution <= WindowResolution.Max; resolution++ ) {
193 if ( monitor.ScreenSize >= resolution ) {
194 resolutions.Add( resolution );
195 }
196 }
197
198 return [.. resolutions];
199 }

216 private static Monitor[] GetScreenList() {
217 int screenCount = DisplayServer.GetScreenCount();
218 var screens = new Monitor[screenCount];
219
220 for ( int i = 0; i < screenCount; i++ ) {
221 var screenSize = DisplayServer.ScreenGetSize( i );
222 var size = new WindowSize( screenSize.X, screenSize.Y );
223 screens[i] = new Monitor( i, (int)DisplayServer.ScreenGetRefreshRate( i ), (WindowResolution)size );
224 }
225
226 return screens;
227 }

233 private void SetWindowMode( WindowMode mode ) {
234 bool borderless = false;
235 var modeEnum = Window.ModeEnum.Windowed;
236 switch ( mode ) {
237 case WindowMode.Windowed:
238 borderless = false;
239 modeEnum = Window.ModeEnum.Windowed;
240 break;
241 case WindowMode.BorderlessWindowed:
242 borderless = true;
243 modeEnum = Window.ModeEnum.Windowed;
244 break;
245 case WindowMode.Fullscreen:
246 borderless = false;
247 modeEnum = Window.ModeEnum.Fullscreen;
248 break;
249 case WindowMode.BorderlessFullscreen:
250 borderless = true;
251 modeEnum = Window.ModeEnum.Fullscreen;
252 break;
253 case WindowMode.ExclusiveFullscreen:
254 borderless = true;
255 modeEnum = Window.ModeEnum.ExclusiveFullscreen;
256 break;
257 }
258 _window.Borderless = borderless;
259 _window.Mode = modeEnum;
260 _mode = mode;
261 }
