199 internal GodotWindowService( SceneTree sceneTree, ICVarSystemService cvarSystem, IGameEventRegistryService eventFactory ) {
200 _window = sceneTree.Root;
201 _window.FocusEntered += OnFocusEntered;
202 _window.FocusExited += OnFocusExited;
203 _window.SizeChanged += OnSizeChanged;
204 _window.CloseRequested += OnCloseRequested;
205 _window.TitleChanged += OnTitleChanged;
206
207 _sizeChanged = eventFactory.GetEvent<WindowSizeChangedEventArgs>( Core.Constants.Events.EngineUtils.WINDOW_SIZE_CHANGED, Core.Constants.Events.EngineUtils.NAMESPACE );
208 _closeRequested = eventFactory.GetEvent<EmptyEventArgs>( Core.Constants.Events.EngineUtils.CLOSE_REQUESTED, Core.Constants.Events.EngineUtils.NAMESPACE );
209 _focusChanged = eventFactory.GetEvent<bool>( Core.Constants.Events.EngineUtils.FOCUS_CHANGED, Core.Constants.Events.EngineUtils.NAMESPACE );
210
211 _monitors = GetScreenList();
212
213 // always ensure we stay in the top-left hand corner to reduce the chance of creating a window that's half off screen.
214 _window.Position = Vector2I.Zero;
215 _window.ContentScaleAspect = Window.ContentScaleAspectEnum.KeepHeight;
216 _window.ContentScaleMode = Window.ContentScaleModeEnum.Viewport;
217 _window.ContentScaleSize = new Vector2I( 800, 600 );
218
219 var maximumFramerate = cvarSystem.GetCVarOrThrow<int>( Core.Constants.CVars.EngineUtils.Display.MAX_FPS );
220 MaximumFramerate = maximumFramerate.Value;
221 maximumFramerate.ValueChanged.Subscribe( OnMaximumFramerateValueChanged );
222
223 var windowMode = cvarSystem.GetCVarOrThrow<WindowMode>( Core.Constants.CVars.EngineUtils.Display.WINDOW_MODE );
224 Mode = windowMode.Value;
225 windowMode.ValueChanged.Subscribe( OnWindowModeValueChanged );
226
227 var windowResolution = cvarSystem.GetCVarOrThrow<WindowResolution>( Core.Constants.CVars.EngineUtils.Display.WINDOW_RESOLUTION );
228 var windowSize = (WindowSize)windowResolution.Value;
229 Width = windowSize.Width;
230 Height = windowSize.Height;
231 windowResolution.ValueChanged.Subscribe( OnWindowResolutionValueChanged );
232
233 var monitor = cvarSystem.GetCVarOrThrow<int>( Core.Constants.CVars.EngineUtils.Display.MONITOR );
234 ScreenIndex = monitor.Value;
235 monitor.ValueChanged.Subscribe( OnMonitorValueChanged );
236
237 WindowService.Initialize( this );
238 }

284 public IReadOnlyList<WindowResolution> GetSupportedResolutions( int monitorIndex ) {
285 var monitor = _monitors[monitorIndex];
286 var resolutions = new List<WindowResolution>();
287
288 for ( var resolution = WindowResolution.Min; resolution <= WindowResolution.Max; resolution++ ) {
289 if ( monitor.ScreenSize >= resolution ) {
290 resolutions.Add( resolution );
291 }
292 }
293 resolutions.Add( WindowResolution.Res_Native );
294
295 return [.. resolutions];
296 }

323 private static Monitor[] GetScreenList() {
324 int screenCount = DisplayServer.GetScreenCount();
325 var screens = new Monitor[screenCount];
326
327 for ( int i = 0; i < screenCount; i++ ) {
328 var screenSize = DisplayServer.ScreenGetSize( i );
329 var size = new WindowSize( screenSize.X, screenSize.Y );
330 screens[i] = new Monitor( i, (int)DisplayServer.ScreenGetRefreshRate( i ), (WindowResolution)size );
331 }
332
333 return screens;
334 }

345 private void SetWindowMode( WindowMode mode ) {
346 bool borderless = false;
347 var modeEnum = Window.ModeEnum.Windowed;
348 switch ( mode ) {
349 case WindowMode.Windowed:
350 borderless = false;
351 modeEnum = Window.ModeEnum.Windowed;
352 break;
353 case WindowMode.BorderlessWindowed:
354 borderless = true;
355 modeEnum = Window.ModeEnum.Windowed;
356 break;
357 case WindowMode.Fullscreen:
358 borderless = false;
359 modeEnum = Window.ModeEnum.Fullscreen;
360 break;
361 case WindowMode.BorderlessFullscreen:
362 borderless = true;
363 modeEnum = Window.ModeEnum.Fullscreen;
364 break;
365 case WindowMode.ExclusiveFullscreen:
366 borderless = true;
367 modeEnum = Window.ModeEnum.ExclusiveFullscreen;
368 break;
369 }
370 _window.Borderless = borderless;
371 _window.Mode = modeEnum;
372 _mode = mode;
373 }

490 private void SetWindowSize( WindowSize size ) {
491 if ( _mode == WindowMode.BorderlessFullscreen || _mode == WindowMode.Fullscreen || _mode == WindowMode.ExclusiveFullscreen ) {
492 var baseSize = _window.ContentScaleSize;
493 float scaleX = (float)baseSize.X / size.Width;
494 float scaleY = (float)baseSize.Y / size.Height;
495 _window.ContentScaleFactor = MathF.Min( scaleX, scaleY );
496
497 var subviewport = _window.GetNode<SubViewport>( "ApplicationBootstrapper/PostProcessingContainer/PostProcessing" );
498 subviewport.Size2DOverride = new Vector2I( size.Width, size.Height );
499 } else {
500 var windowSize = new Vector2I( size.Width, size.Height );
501 _window.Size = windowSize;
502
503 var subviewport = _window.GetNode<SubViewport>( "ApplicationBootstrapper/PostProcessingContainer/PostProcessing" );
504 subviewport.Size2DOverride = new Vector2I( size.Width, size.Height );
505 }
506 }
