79 ===============

85 private unsafe EntityDataDto? AllocEntityIndex( out int index ) {
86 index = -1;
87 if ( _freeIndexCount > 0 ) {
88 index = _freeIndices[ --_freeIndexCount ];
89 } else if ( _freeIndexCount < MAX_ENTITIES ) {
90 index = _entityCount++;
91 } else {
92 return null;
93 }
94
95 fixed ( Vector2* positionPtr = &_hotBatchData.Positions[ index ] )
96 fixed ( Vector2* scalePtr = &_hotBatchData.Scales[ index ] )
97 fixed ( float* rotationPtr = &_hotBatchData.Rotations[ index ] )
98 fixed ( bool* visiblePtr = &_hotBatchData.Visibilities[ index ] )
99 fixed ( Color* modulatePtr = &_hotBatchData.Colors[ index ] )
100 fixed ( uint* visibilityLayerPtr = &_coldBatchData.VisibilityLayers[ index ] )
101 fixed ( int* zindexPtr = &_coldBatchData.ZIndexes[ index ] )
102 fixed ( int* lightmaskPtr = &_coldBatchData.LightMasks[ index ] )
103 fixed ( Rid* ridPtr = &_hotBatchData.Rids[ index ] ) {
104 return new EntityDataDto(
105 modulatePtr,
106 ridPtr,
107 zindexPtr,
108 lightmaskPtr,
109 visibilityLayerPtr,
110 positionPtr,
111 scalePtr,
112 rotationPtr,
113 visiblePtr
114 );
115 }
116 }

118 /*
119 ===============
120 CreateEntity
121 ===============
122 */
129 public IRenderEntity? CreateEntity( CanvasItem canvasItem ) {
130 switch ( canvasItem ) {
131 case TextureRect textureRect: {
132 break;
133 }
134 case Sprite2D sprite: {
135 var entityDto = AllocEntityIndex( out _ );
136 if ( !entityDto.HasValue ) {
137 return null;
138 }
139 return new RenderSprite( entityDto.Value, sprite );
140 }
141 }
142 throw new InvalidOperationException( "Invalid CanvasItem type!" );
143 }

157 public IAnimationEntity? CreateAnimator( CanvasItem canvasItem, IGameEventRegistryService eventFactory ) {
158 if ( canvasItem is AnimatedSprite2D animatedSprite ) {
159 return new RenderAnimator( animatedSprite, eventFactory );
160 }
161 throw new InvalidCastException();
162 }

164 /*
165 ===============
166 Update
167 ===============
168 */
173 public void Update( float delta ) {
174 int count = _entityCount;
175 unsafe {
176 fixed ( Rid* ridPtr = _hotBatchData.Rids )
177 fixed ( Vector2* positionPtr = _hotBatchData.Positions )
178 fixed ( Vector2* scalePtr = _hotBatchData.Scales )
179 fixed ( float* rotationPtr = _hotBatchData.Rotations )
180 fixed ( bool* visiblePtr = _hotBatchData.Visibilities ) {
181 for ( int i = 0; i < count; i++ ) {
182 RenderingServer.CanvasItemClear( ridPtr[ i ] );
183 RenderingServer.CanvasItemSetTransform( ridPtr[ i ], new Transform2D( rotationPtr[ i ], scalePtr[ i ], 0.0f, positionPtr[ i ] ) );
184 }
185 }
186 _animationService.Update( delta );
187 }
188 }
