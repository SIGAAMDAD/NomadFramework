168 private void OnVSyncModeChanged( in CVarValueChangedEventArgs<VSyncMode> args ) {
169 switch ( args.NewValue ) {
170 case VSyncMode.Disabled:
171 QualitySettings.vSyncCount = 0;
172 break;
173 case VSyncMode.Enabled:
174 case VSyncMode.Adaptive:
175 QualitySettings.vSyncCount = 1;
176 break;
177 case VSyncMode.TripleBuffered:
178 QualitySettings.vSyncCount = 1;
179 _customSettings[Core.Constants.CVars.EngineUtils.Display.VSYNC_MODE] = new Any( "TripleBuffered" );
180 break;
181 }
182 }

193 private void OnAntiAliasingChanged( in CVarValueChangedEventArgs<AntiAliasingMode> args ) {
194 switch ( args.NewValue ) {
195 case AntiAliasingMode.None:
196 QualitySettings.antiAliasing = 0;
197 break;
198 case AntiAliasingMode.MSAA_2x:
199 QualitySettings.antiAliasing = 2;
200 break;
201 case AntiAliasingMode.MSAA_4x:
202 QualitySettings.antiAliasing = 4;
203 break;
204 case AntiAliasingMode.MSAA_8x:
205 QualitySettings.antiAliasing = 8;
206 break;
207 default:
208 QualitySettings.antiAliasing = 0;
209 _customSettings[Core.Constants.CVars.EngineUtils.Display.ANTI_ALIASING] = new Any( args.NewValue.ToString() );
210 break;
211 }
