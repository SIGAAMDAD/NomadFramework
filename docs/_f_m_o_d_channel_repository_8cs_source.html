133 public FMODChannelHandle? AllocateChannel( string id, Vector2 position, SoundCategory config, float basePriority = 0.5f, bool isEssential = false ) {
134 var instance = CreateSoundInstance( id );
135 float actualPriority = CalculateActualPriority( id, position, basePriority, config );
136
137 EnforceCategoryLimits();
138
139 if ( !_channelTracker.TryAllocateChannel( id, config, basePriority, actualPriority, isEssential, out FMODChannel channel ) ) {
140 _category.PrintError( $"Couldn't allocate a channel for sound event '{id}'!" );
141 return null;
142 }
143
144 try {
145 instance.CreateInstance( out FMODChannelResource eventInstance );
146 eventInstance.Volume = _effectsVolume / 10.0f;
147 eventInstance.Position = position;
148
149 FMODValidator.ValidateCall( eventInstance.SetFinishedCallback( SoundFinishedCallback ) );
150 _channelTracker.AttachInstance( channel, eventInstance );
151 FMODValidator.ValidateCall( eventInstance.Start() );
152 _channelTracker.RecordPlayback( id );
153 return channel.Handle;
154 } catch {
155 _channelTracker.StopChannel( channel, false );
156 throw;
157 }
158 }

193 /*
194 ===============
195 UpdatePriorities
196 ===============
197 */
201 private void UpdatePriorities() {
202 Vector2 listenerPos = _listenerService.ActiveListener;
203
204 for ( int i = 0; i < _channelTracker.ActiveChannelCount; i++ ) {
205 FMODChannel channel = _channelTracker.GetActiveChannelAt( i );
206 if ( !channel.IsPlaying || channel.Category == null ) {
207 continue;
208 }
209
210 float distanceFactor = _priorityCalculator.CalculateDistanceFactor( Vector2.Distance( channel.Instance.Position, listenerPos ) );
211 channel.Volume = distanceFactor;
212 channel.CurrentPriority = channel.BasePriority * channel.Category.Config.PriorityScale * distanceFactor;
213 }
214 }

216 /*
217 ===============
218 EnforceCategoryLimits
219 ===============
220 */
224 private void EnforceCategoryLimits() {
225 foreach ( var category in _busRepository.Categories ) {
226 while ( _channelTracker.CountSoundsInCategory( category.Key ) > category.Value.Config.MaxSimultaneous ) {
227 if ( !_channelTracker.TryGetLowestPriorityChannelInCategory( category.Value.Name, out FMODChannel channel ) ) {
228 break;
229 }
230 _channelTracker.StopChannel( channel, true );
231 }
232 }
233 }

236 ===============
237 CalculateActualPriority
238 ===============
239 */
248 private float CalculateActualPriority( string id, Vector2 position, float basePriority, SoundCategory category ) {
249 float? lastTimePlayed = null;
250 int consecutiveStealCount = 0;
251
252 if ( _channelTracker.TryGetPlaybackStatistics( id, out float value, out int stealCount ) ) {
253 lastTimePlayed = value;
254 consecutiveStealCount = stealCount;
255 }
256
257 return _priorityCalculator.CalculateActualPriority(
258 _channelTracker.ElapsedSeconds,
259 position,
260 basePriority,
261 category,
262 lastTimePlayed,
263 consecutiveStealCount
264 );
265 }
