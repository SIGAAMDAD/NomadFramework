87 : base( animatedSprite ) {

91 var spriteFrames = animatedSprite.SpriteFrames;
92 var animationNames = spriteFrames.GetAnimationNames();
93
94 _currentAnimation = animatedSprite.Animation;
95
96 var animations = new Dictionary<string, AnimationData>( animationNames.Length );
97 for ( int i = 0; i < animationNames.Length; i++ ) {
98 var animationName = animationNames[ i ];
99 int frameCount = spriteFrames.GetFrameCount( animationName );
100
101 var rids = new Rid[ frameCount ];
102 var regions = new Rect2[ frameCount ];
103 var durations = new float[ frameCount ];
104 for ( int f = 0; f < frameCount; f++ ) {
105 var texture = spriteFrames.GetFrameTexture( animationName, f );
106 rids[ f ] = texture.GetRid();
107 regions[ f ] = GetTextureRegion( texture );
108 durations[ f ] = spriteFrames.GetFrameDuration( animationName, f );
109 }
110
111 fixed ( Rid* ridPtr = rids )

113 fixed ( float* durationsPtr = durations ) {
114 animations[ animationNames[ i ] ] = new AnimationData(

116 regionPtr,
117 durationsPtr,
118 frameCount,
119 ( float )spriteFrames.GetAnimationSpeed( animationName ),

121 );
122 }
123 }

138 public void Play( string animationName = "", float speedScale = 1.0f, bool backwards = false ) {
139 if ( animationName.Length == 0 || _currentAnimation == animationName ) {
140 return;
141 }
142 if ( !_animationData.TryGetValue( animationName, out var animation ) ) {
143 return;
144 }
145 _currentAnimation = animationName;
146 _currentFrame = backwards ? animation.FrameCount - 1 : 0;
147 _frameTimer = 0.0f;
148 _playing = true;
149 _speedScale = speedScale;
150
151 UpdateCurrentFrameData( in animation );
152 }
153
154 /*
155 ===============
156 Update
157 ===============
158 */
163 public override void Update( float delta ) {
164 if ( _currentAnimation.Length == 0 ) {
165 return;
166 }
167
168 var animation = _animationData[ _currentAnimation ];
169
170 _frameTimer += animation.AnimationSpeed * _speedScale * delta;
171 while ( _frameTimer >= animation.FrameDurations[ _currentFrame ] ) {
172 _frameTimer -= animation.FrameDurations[ _currentFrame ];
173
174 if ( _backwards ) {
175 _currentFrame--;
176 } else {
177 _currentFrame++;
178 }
179 if ( _currentFrame >= animation.FrameCount ) {
180 if ( animation.Loop ) {
181 _currentFrame = 0;
182 } else {
183 _currentFrame = animation.FrameCount - 1;
184 _playing = false;
185 break;
186 }
187 }
188 }
189 UpdateCurrentFrameData( in animation );
190 }
191

193 ===============
194 UpdateCurrentFrameData
195 ===============
196 */
201 private void UpdateCurrentFrameData( in AnimationData animation ) {
202 var textureRid = animation.TextureRids[ _currentFrame ];
203 var textureRegion = animation.TextureRegions[ _currentFrame ];
204
205 if ( textureRid.IsValid ) {
206 var frameSize = textureRegion.Size;
207
208 RenderingServer.CanvasItemClear( _canvasRid );
209
210 var scale = _scale;
211 if ( _flipH ) {
212 scale.X = Math.Abs( scale.X ) * -1.0f;
213 }
214 RenderingServer.CanvasItemSetTransform( _canvasRid, new Transform2D( _rotation, scale, 0.0f, _position ) );
215
216 Rect2 destRect = new Rect2(
217 -frameSize / 2,
218 frameSize
219 );
220
221 RenderingServer.CanvasItemAddTextureRectRegion( _canvasRid, destRect, textureRid, textureRegion );
222 }
