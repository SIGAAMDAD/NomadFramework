153 public override long Seek( long offset, SeekOrigin origin ) {
154 StateGuard.ThrowIfDisposed( isDisposed, this );
155 switch ( origin ) {
156 case SeekOrigin.Begin:
157 if ( offset > length ) {
158 throw new IOException( "seek (begin) offset is larger than the stream's size!" );
159 }
160 position = offset;
161 break;
162 case SeekOrigin.Current:
163 if ( position + offset > length ) {
164 throw new IOException( "seek (current) offset + position is larger than the stream's size!" );
165 }
166 position += offset;
167 break;
168 case SeekOrigin.End:
169 if ( offset > 0 ) {
170 throw new IOException( "seek (end) offset is greater than 0!" );
171 }
172 position = length + offset;
173 break;
174 default:
175 return -1;
176 }
177 return position;
178 }
