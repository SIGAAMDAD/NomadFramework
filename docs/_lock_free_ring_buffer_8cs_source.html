106 public bool TryEnqueue(T item)
107 {
108 int currentHead = _head;
109 int nextHead = (currentHead + 1) & _mask;
110
111 // Check if full (tail hasn't caught up)
112 if (nextHead == _tail)
113 {
114 return false;
115 }
116
117 _buffer[currentHead] = item;
118
119 // Ensure item is written before head is updated
120 Thread.MemoryBarrier();
121 _head = nextHead;
122 return true;
123 }

130 public bool TryDequeue(out T item)
131 {

133
134 // Check if empty
135 if (currentTail == _head)

139 }
140
141 item = _buffer[currentTail];
142 int nextTail = (currentTail + 1) & _mask;
143
144 // Ensure item is read before tail is updated
145 Thread.MemoryBarrier();
146 _tail = nextTail;
147 return true;
148 }
