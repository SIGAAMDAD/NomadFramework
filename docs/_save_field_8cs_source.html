88 string name = stream.ReadString();
89 if ( string.IsNullOrEmpty( name ) || name.Length > Constants.MAX_FIELD_NAME_LENGTH ) {
90 throw new FieldCorruptException( section, index, stream.Position, $"Field name length corrupted (0 or string overflow, {name.Length} bytes)" );
91 }
92
93 AnyType type = (AnyType)stream.ReadUInt8();
94 if ( type < AnyType.Boolean || type >= AnyType.Count ) {
95 throw new FieldCorruptException( section, index, stream.Position, $"Field type '{type}' isn't valid" );
96 }
97
98 Any value = FieldSerializerRegistry.GetSerializer( Any.GetAnyType( type ) ).Deserialize( stream );
99
100 return new SaveField( name, type, value );
