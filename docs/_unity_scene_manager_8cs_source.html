113 public IScene LoadScene( string path, LoadSceneMode mode = LoadSceneMode.Single ) {
114 USceneManager.LoadScene( path, mode == LoadSceneMode.Additive ? USceneLoadMode.Additive : USceneLoadMode.Single );
115 SyncLoadedScenes();
116
117 UnityScene scene = ResolveScene( path );
118 if ( mode == LoadSceneMode.Single ) {
119 _activeScene = scene;
120 } else if ( !_additiveScenes.Contains( scene ) ) {
121 _additiveScenes.Add( scene );
122 }
123
124 return scene;
125 }

216 private UnityScene ResolveScene( string path ) {
217 string normalizedPath = NormalizeScenePath( path );
218
219 UScene scene = USceneManager.GetSceneByPath( normalizedPath );
220 if ( !scene.IsValid() ) {
221 scene = USceneManager.GetSceneByName( Path.GetFileNameWithoutExtension( normalizedPath ) );
222 }
223
224 return GetOrCreateScene( scene, normalizedPath );
225 }
226
227 /*
228 ===============
229 SyncLoadedScenes
230 ===============
231 */
235 private void SyncLoadedScenes() {
236 _additiveScenes.Clear();
237
238 UScene activeScene = USceneManager.GetActiveScene();
239 _activeScene = activeScene.IsValid() ? GetOrCreateScene( activeScene, activeScene.path ) : null;
240
241 for ( int i = 0; i < USceneManager.sceneCount; i++ ) {
242 UScene scene = USceneManager.GetSceneAt( i );
243 if ( !scene.IsValid() ) {
244 continue;
245 }
246
247 UnityScene unityScene = GetOrCreateScene( scene, scene.path );
248 if ( _activeScene == null || scene.handle != _activeScene.Scene.handle ) {
249 _additiveScenes.Add( unityScene );
250 }
251 }
252 }

254 /*
255 ===============
256 GetOrCreateScene
257 ===============
258 */
266 private UnityScene GetOrCreateScene( UScene scene, string path ) {
267 if ( !scene.IsValid() ) {
268 throw new InvalidOperationException( "The requested Unity scene is invalid." );
269 }
270 if ( _sceneCache.TryGetValue( scene.handle, out var cachedScene ) ) {
271 return cachedScene;
272 }
273
274 var createdScene = new UnityScene( scene, path );
275 _sceneCache[scene.handle] = createdScene;
276 return createdScene;
277 }
