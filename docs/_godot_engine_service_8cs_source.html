63 public GodotEngineService( SceneTree sceneTree, IServiceRegistry serviceFactory, IServiceLocator locator ) {
64 ArgumentGuard.ThrowIfNull( sceneTree );
65 ArgumentGuard.ThrowIfNull( serviceFactory );
66 ArgumentGuard.ThrowIfNull( locator );
67
68 _locator = locator;
69 _sceneTree = sceneTree;
70 _root = sceneTree.Root;
71
72 _logger = locator.GetService<ILoggerService>();
73 _eventFactory = locator.GetService<IGameEventRegistryService>();
74 var cvarSystem = locator.GetService<ICVarSystemService>();
75
76 _loader = new GodotLoader();
77
78 _windowService = new GodotWindowService( _sceneTree, _eventFactory );
79 serviceFactory.AddSingleton( _windowService );
80 _windowService.CloseRequested.Subscribe( OnWindowCloseRequested );
81
82 _localizationService = new GodotLocalizationService();
83 serviceFactory.AddSingleton( _localizationService );
84
85 _sceneManager = new GodotSceneManager( _sceneTree, new BaseCache<PackedScene, string>( _logger, _eventFactory, _loader ) );
86 serviceFactory.AddSingleton( _sceneManager );
87
88 _displayService = new GodotDisplayService( sceneTree, _windowService, cvarSystem );
89 serviceFactory.AddSingleton( _displayService );
90
91 _logger.AddSink( new ConsoleSink() );
92
93 EngineService.Initialize( this );
94 }
95
