67 public GodotEngineService( SceneTree sceneTree, IServiceRegistry serviceFactory, IServiceLocator locator ) {
68 ArgumentGuard.ThrowIfNull( serviceFactory );
69 ArgumentGuard.ThrowIfNull( locator );
70
71 _locator = locator ?? throw new ArgumentNullException( nameof( locator ) );
72 _sceneTree = sceneTree ?? throw new ArgumentNullException( nameof( sceneTree ) );
73 _root = sceneTree.Root;
74
75 _logger = locator.GetService<ILoggerService>();
76 _eventFactory = locator.GetService<IGameEventRegistryService>();
77 var cvarSystem = locator.GetService<ICVarSystemService>();
78
79 _loader = new GodotLoader();
80
81 _inputPump = new GodotInputPump( _eventFactory );
82 _root.CallDeferred( Node.MethodName.AddChild, _inputPump );
83
84 DisplayCVars.Register( cvarSystem );
85
86 _windowService = new GodotWindowService( _sceneTree, cvarSystem, _eventFactory );
87 serviceFactory.AddSingleton( _windowService );
88 _windowService.CloseRequested.Subscribe( OnWindowCloseRequested );
89
90 _localizationService = new GodotLocalizationService();
91 serviceFactory.AddSingleton( _localizationService );
92
93 _displayService = new GodotDisplayService( sceneTree, _windowService, cvarSystem );
94 serviceFactory.AddSingleton( _displayService );
95
96 _sceneManager = new GodotSceneManager( _sceneTree, new BaseCache<PackedScene, string>( _logger, _eventFactory, _loader ) );
97 serviceFactory.AddSingleton( _sceneManager );
98 serviceFactory.AddSingleton<ISplitScreenService>( new GodotSplitScreenService() );
99
100 serviceFactory.AddSingleton<IRaycastService>( new GodotRaycastService( _sceneTree.Root.World2D ) );
101
102 _logger.AddSink( new ConsoleSink() );
103
104 serviceFactory.AddSingleton<ITimeService>( new GodotTimeService() );
105 serviceFactory.AddSingleton<IGamePauseService>( new GodotPauseService( _sceneTree, _eventFactory ) );
106
107 EngineService.Initialize( this );
108 }

119 _isDisposed = true;
120 }
