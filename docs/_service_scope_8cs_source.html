53 public void Dispose()
54 {
55 if (_isDisposed)
56 {
57 return;
58 }
59 foreach (var instance in _scopedInstances.Values)
60 {
61 if (instance is IDisposable d)

63 d.Dispose();
64 }
65 }
66 _scopedInstances.Clear();
67 _isDisposed = true;

76 public TService GetService<TService>()
77 where TService : class
78 {
79 if (TryGetService(out TService? service))
80 {
81 return service;
82 }

92 public bool TryGetService<TService>([NotNullWhen(true)] out TService? service)
93 where TService : class
94 {
95 if (!_root._collection.TryGetDescriptor(typeof(TService), out var descriptor))
96 {
97 service = null;
98 return false;
99 }
100
101 if (descriptor.Lifetime == ServiceLifetime.Scoped)
102 {
103 // Scoped: use our own cache
104 if (_scopedInstances.TryGetValue(descriptor.ServiceType, out object? cached))
105 {
106 service = (TService)cached;
107 return true;
108 }
109
110 var instance = CreateScopedInstance(descriptor);
111 _scopedInstances[descriptor.ServiceType] = instance;
112 service = (TService)instance;
113 return true;
114 }
115
116 // Singleton/Transient: delegate to root locator
117 return _root.TryGetService(out service);
118 }
119
125 public IEnumerable<TService> GetServices<TService>()
126 where TService : class
127 {
128 // Scoped services currently in cache
129 foreach (var entry in _scopedInstances)
130 {
131 if (entry.Value is TService s)
132 {
133 yield return s;
134 }
135 }
136
137 // Also include singletons from root? Typically GetServices returns all instances of a type,
138 // but here we mimic the original behavior (only singletons from root cache). Adjust if needed.
139 foreach (var s in _root.GetServices<TService>())
140 {
141 yield return s;
142 }
143 }
144
151 public TService CreateInstance<TService, TImplementation>()
152 where TService : class
153 where TImplementation : class, TService
154 {
155 return _root.CreateInstance<TService, TImplementation>();
156 }
157
163 public IServiceScope CreateScope()

172 private object CreateScopedInstance(ServiceDescriptor descriptor)
173 {
174 if (descriptor.ImplementationType == null)
175 {
176 throw new InvalidOperationException($"No implementation type for {descriptor.ServiceType}");
177 }
178
179 // Use the root's factory cache to create the instance (avoids reflection).
180 var factory = _root.CreateFactory(descriptor.ImplementationType); // we need to expose this or reuse internal
181 var instance = factory(this);
182 // Scoped instances are tracked for disposal by this scope.
183 return instance;
184 }
185 }
