61 public IServiceRegistry Register<TService, TImplementation>(ServiceLifetime lifetime)
62 where TService : class
63 where TImplementation : class, TService
64 {
65 var descriptor = lifetime switch
66 {
67 ServiceLifetime.Singleton => ServiceDescriptor.Singleton<TService, TImplementation>(),
68 ServiceLifetime.Transient => ServiceDescriptor.Transient<TService, TImplementation>(),
69 ServiceLifetime.Scoped => ServiceDescriptor.Scoped<TService, TImplementation>(),
70 _ => throw new ArgumentOutOfRangeException(nameof(lifetime))
71 };
72 _descriptors[typeof(TService)] = descriptor;
73 return this;
74 }
