137 private void PumpKeyboard( long now ) {
138 var keyboard = Keyboard.current;
139 if ( keyboard == null ) {
140 return;
141 }
142
143 foreach ( var translation in _keyboardMap ) {
144 var control = keyboard[translation.UnityKey];
145 if ( control == null ) {
146 continue;
147 }
148
149 if ( control.wasPressedThisFrame ) {
150 _system.PushKeyboardEvent(
151 new KeyboardEvent( translation.NomadKey, now, true )
152 );
153 }
154
155 if ( control.wasReleasedThisFrame ) {
156 _system.PushKeyboardEvent(
157 new KeyboardEvent( translation.NomadKey, now, false )
158 );
159 }
160 }
161 }

167 private void PumpMouse( long now ) {
168 var mouse = Mouse.current;
169 if ( mouse == null ) {
170 return;
171 }
172
173 PumpMouseButton( now, mouse.leftButton, MouseButton.Left );
174 PumpMouseButton( now, mouse.rightButton, MouseButton.Right );
175 PumpMouseButton( now, mouse.middleButton, MouseButton.Middle );
176
177 Vector2 position = mouse.position.ReadValue();
178 Vector2 delta = mouse.delta.ReadValue();
179
180 if ( delta != Vector2.zero || position != _lastMousePosition ) {
181 _system.PushMouseMotionEvent(
182 new MouseMotionEvent(
183 now,
184 Mathf.RoundToInt( position.x ),
185 Mathf.RoundToInt( position.y )
186 )
187 );
188
189 _lastMousePosition = position;
190 }
191
192 // Unity scroll is a delta, not a pressed/released wheel button,
193 // so synthesize a tap-like press/release pair.
194 Vector2 scroll = mouse.scroll.ReadValue();
195 if ( scroll.y > 0f ) {
196 EmitScrollWheel( now, MouseButton.WheelUp );
197 } else if ( scroll.y < 0f ) {
198 EmitScrollWheel( now, MouseButton.WheelDown );
199 }
200 }

245 PumpGamepadButton( now, gamepad.dpad.up, deviceId, GamepadButton.DPadUp );

247 PumpGamepadButton( now, gamepad.dpad.left, deviceId, GamepadButton.DPadLeft );
248 PumpGamepadButton( now, gamepad.dpad.right, deviceId, GamepadButton.DPadRight );
249
250 PumpGamepadButton( now, gamepad.leftShoulder, deviceId, GamepadButton.LeftShoulder );

296 _system.PushGamepadButtonEvent(

299 }

310 private void PumpStick( long now, int deviceId, GamepadStick stick, Vector2 currentValue, Dictionary<int, Vector2> cache ) {
311 if ( !cache.TryGetValue( deviceId, out var previousValue ) ||
312 (currentValue - previousValue).sqrMagnitude > (AXIS_EPSILON * AXIS_EPSILON) ) {
313 cache[deviceId] = currentValue;
314
315 _system.PushGamepadAxisEvent(
316 new GamepadAxisEvent(
317 stick,
318 now,
319 deviceId,
320 new System.Numerics.Vector2( currentValue.x, currentValue.y )
321 )
322 );
323 }
