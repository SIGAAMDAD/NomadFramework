103 public async Task<SteamLobbyData?> CreateLobby( LobbyInfo info, CancellationToken ct = default ) {
104 if ( info.MaxPlayers < 1 || info.MaxPlayers > _maxPlayers ) {
105 throw new ArgumentOutOfRangeException( nameof( info ), "LobbyInfo.MaxPlayers is less than 1 or greater than MaxPlayers!" );
106 }
107
108 ELobbyType type = info.Visibility switch {
109 LobbyVisibility.Private => ELobbyType.k_ELobbyTypePrivate,
110 LobbyVisibility.Public => ELobbyType.k_ELobbyTypePublic,
111 LobbyVisibility.FriendsOnly => ELobbyType.k_ELobbyTypeFriendsOnly,
112 _ => throw new ArgumentOutOfRangeException( nameof( info ) )
113 };
114
115 return await _lobbyCreated.Invoke(
116 result => {
117 if ( result.m_eResult != EResult.k_EResultOK ) {
118 _category.PrintError( $"SteamLobbyFactory.OnLobbyCreated: error creating lobby - {result.m_eResult}" );
119 return null;
120 }
121 _category.PrintLine( $"SteamLobbyFactory.OnLobbyFactory: created new lobby with CSteamID '{result.m_ulSteamIDLobby}'" );
122
123 CSteamID id = (CSteamID)result.m_ulSteamIDLobby;
124
125 // setup default metadata
126 SteamMatchmaking.SetLobbyOwner( id, _userData.UserID );
127 SteamMatchmaking.SetLobbyMemberLimit( id, info.MaxPlayers );
128 SteamMatchmaking.SetLobbyJoinable( id, true );
129
130 SteamMatchmaking.SetLobbyData( id, nameof( LobbyInfo.Name ), info.Name );
131 SteamMatchmaking.SetLobbyData( id, nameof( LobbyInfo.Map ), info.Map );
132 SteamMatchmaking.SetLobbyData( id, nameof( LobbyInfo.GameMode ), info.GameMode );
133 SteamMatchmaking.SetLobbyData( id, nameof( LobbyInfo.Visibility ), info.Visibility.ToString() );
134
135 foreach ( var metadata in info.Metadata ) {
136 SteamMatchmaking.SetLobbyData( id, metadata.Key, metadata.Value );
137 }
138
139 var data = new SteamLobbyData( id, info, Guid.NewGuid() );
140 data.Update();
141
142 return data;
143 },
144 () => SteamMatchmaking.CreateLobby( type, info.MaxPlayers ),
145 ct
146 );
147 }
