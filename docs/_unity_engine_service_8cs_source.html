134 public string GetOSPath( string localpath ) {
135 if ( string.IsNullOrWhiteSpace( localpath ) ) {
136 return string.Empty;
137 }
138
139 string normalizedPath = NormalizePath( localpath );
140 if ( Path.IsPathRooted( normalizedPath ) ) {
141 return normalizedPath;
142 }
143
144 if ( normalizedPath.StartsWith( "Assets/StreamingAssets", StringComparison.OrdinalIgnoreCase ) ) {
145 string relativePath = normalizedPath.Substring( "Assets/StreamingAssets".Length ).TrimStart( '/' );
146 return NormalizePath( Path.Combine( Application.streamingAssetsPath, relativePath ) );
147 }
148
149 if ( normalizedPath.StartsWith( "Assets", StringComparison.OrdinalIgnoreCase ) ) {
150 string projectRoot = System.IO.Directory.GetParent( Application.dataPath ) != null
151 ? System.IO.Directory.GetParent( Application.dataPath ).FullName
152 : Application.dataPath;
153 return NormalizePath( Path.Combine( projectRoot, normalizedPath ) );
154 }
155
156 return NormalizePath( Path.GetFullPath( normalizedPath ) );
157 }
158
164 public string GetLocalPath( string ospath ) {
165 if ( string.IsNullOrWhiteSpace( ospath ) ) {
166 return string.Empty;
167 }
168
169 string normalizedPath = NormalizePath( ospath );
170 string normalizedStreamingAssetsPath = NormalizePath( Application.streamingAssetsPath );
171 if ( normalizedPath.StartsWith( normalizedStreamingAssetsPath, StringComparison.OrdinalIgnoreCase ) ) {
172 return "Assets/StreamingAssets" + normalizedPath.Substring( normalizedStreamingAssetsPath.Length );
173 }
174
175 string normalizedDataPath = NormalizePath( Application.dataPath );
176 if ( normalizedPath.StartsWith( normalizedDataPath, StringComparison.OrdinalIgnoreCase ) ) {
177 return "Assets" + normalizedPath.Substring( normalizedDataPath.Length );
178 }
179
180 return normalizedPath;
181 }

187 public string GetStoragePath( StorageScope scope ) => scope switch {
188 StorageScope.Install => NormalizePath( System.IO.Directory.GetParent( Application.dataPath ) != null ? System.IO.Directory.GetParent( Application.dataPath ).FullName : Application.dataPath ),
189 StorageScope.StreamingAssets => NormalizePath( Application.streamingAssetsPath ),
190 StorageScope.UserData => NormalizePath( Application.persistentDataPath ),
191 StorageScope.Documents => NormalizePath( Environment.GetFolderPath( Environment.SpecialFolder.MyDocuments ) ),
192 StorageScope.Temporary => NormalizePath( Application.temporaryCachePath ),
193 _ => throw new ArgumentOutOfRangeException( nameof( scope ) ),
194 };
