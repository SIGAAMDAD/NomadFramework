175 /*
176 ===============
177 ParseLineInput
178 ===============
179 */
185 private void ParseLineInput( string text ) {
186 bool inQuotes = false;
187 bool escaped = false;
188
189 _arguments.Clear();
190 _commandBuilder.Clear();
191
192 for ( int i = 0; i < text.Length; i++ ) {
193 char c = text[i];
194
195 if ( escaped ) {
196 _commandBuilder.Append( c );
197 escaped = false;
198 continue;
199 }
200
201 switch ( c ) {
202 case '\\':
203 escaped = true;
204 continue;
205 case '\"':
206 inQuotes = !inQuotes;
207 continue;
208 default:
209 if ( char.IsWhiteSpace( c ) && !inQuotes ) {
210 if ( _commandBuilder.Length > 0 ) {
211 _arguments.Add( _commandBuilder.ToString() );
212 _commandBuilder.Clear();
213 }
214 continue;
215 }
216 break;
217 }
218
219 _commandBuilder.Append( c );
220 }
221
222 if ( _commandBuilder.Length > 0 ) {
223 _arguments.Add( _commandBuilder.ToString() );
224 }
225 }
