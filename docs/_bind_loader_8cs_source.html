68 ===============

76 public bool LoadBindDatabase( string filePath, out ImmutableArray<InputActionDefinition> binds ) {
77 IBufferHandle? fileBuffer = _fileSystem.LoadFile( filePath );
78 if ( fileBuffer == null ) {
79 binds = ImmutableArray<InputActionDefinition>.Empty;
80 _category.PrintLine( $"Couldn't load bind file '{filePath}'!" );
81 return false;
82 }
83
84 try {
85 using var stream = fileBuffer.AsStream( 0, fileBuffer.Length );
86 using var document = JsonLoader.Parse( stream );
87
88 if ( !JsonLoader.TryGetProperty( document.RootElement, "Bindings", out var bindingsElement ) ) {
89 throw new Exception( $"Bindings file '{filePath}' is missing a 'Bindings' property." );
90 }
91
92 var actionIndices = new Dictionary<string, int>( StringComparer.Ordinal );
93 var actions = new List<ActionBuilder>();
94
95 switch ( bindingsElement.ValueKind ) {
96 case JsonValueKind.Array:
97 foreach ( var actionElement in bindingsElement.EnumerateArray() ) {
98 ParseActionDefinition( actionElement, actionIndices, actions );
99 }
100 break;
101 case JsonValueKind.Object:
102 foreach ( var actionProperty in bindingsElement.EnumerateObject() ) {
103 ParseActionDefinition( actionProperty.Value, actionIndices, actions, actionProperty.Name );
104 }
105 break;
106 default:
107 throw new Exception( $"Bindings file '{filePath}' has an invalid 'Bindings' payload. Expected an array or object." );
108 }
109
110 var builder = ImmutableArray.CreateBuilder<InputActionDefinition>( actions.Count );
111 for ( int i = 0; i < actions.Count; i++ ) {
112 builder.Add( actions[i].Build() );
113 }
114 binds = builder.ToImmutable();
115 return true;
116 } finally {
117 fileBuffer.DisposeAsync().AsTask().GetAwaiter().GetResult();
118 }
119 }

134 private static void ParseActionDefinition( JsonElement actionElement, Dictionary<string, int> actionIndices, List<ActionBuilder> actions, string? fallbackName = null ) {
135 string name = GetRequiredString( actionElement, "Name", fallbackName );
136 string id = GetRequiredString( actionElement, "Id", fallbackName );
137 InputValueType valueType = JsonLoader.GetRequired<InputValueType>( actionElement, "ValueType" );
138 InputScheme scheme = JsonLoader.GetRequired<InputScheme>( actionElement, "Scheme" );
139
140 if ( !JsonLoader.TryGetProperty( actionElement, "Bindings", out JsonElement bindingsElement ) && !JsonLoader.TryGetProperty( actionElement, "Binding", out bindingsElement ) ) {
141 throw new Exception( $"Binding action '{id}' is missing a binding payload." );
142 }
143
144 if ( !actionIndices.TryGetValue( id, out int actionIndex ) ) {
145 actionIndex = actions.Count;
146 actionIndices.Add( id, actionIndex );
147 actions.Add( new ActionBuilder( name, id, valueType ) );
148 } else if ( actions[actionIndex].ValueType != valueType ) {
149 throw new Exception( $"Binding action '{id}' has conflicting value types." );
150 }
151
152 if ( bindingsElement.ValueKind == JsonValueKind.Array ) {
153 foreach ( JsonElement bindingElement in bindingsElement.EnumerateArray() ) {
154 actions[actionIndex].Bindings.Add( ParseBindingDefinition( bindingElement, scheme ) );
155 }
156 return;
157 }
158
159 if ( bindingsElement.ValueKind != JsonValueKind.Object ) {
160 throw new Exception( $"Binding action '{id}' has an invalid binding payload." );
161 }
162
163 actions[actionIndex].Bindings.Add( ParseBindingDefinition( bindingsElement, scheme ) );
164 }

178 private static InputBindingDefinition ParseBindingDefinition( JsonElement bindingElement, InputScheme scheme ) {
179 InputBindingKind kind = JsonLoader.TryGetProperty( bindingElement, "Kind", out JsonElement kindElement )
180 ? JsonLoader.Read<InputBindingKind>( kindElement, "Kind" )
181 : InferBindingKind( bindingElement );
182
183 InputBindingDefinition definition = new InputBindingDefinition {
184 Scheme = scheme,
185 Kind = kind
186 };
187
188 switch ( kind ) {
189 case InputBindingKind.Button:
190 definition.Button = ParseButtonBinding( bindingElement );
191 break;
192 case InputBindingKind.Axis1D:
193 definition.Axis1D = ParseAxis1DBinding( bindingElement );
194 break;
195 case InputBindingKind.Axis1DComposite:
196 definition.Axis1DComposite = ParseAxis1DCompositeBinding( bindingElement );
197 break;
198 case InputBindingKind.Axis2D:
199 definition.Axis2D = ParseAxis2DBinding( bindingElement );
200 break;
201 case InputBindingKind.Axis2DComposite:
202 definition.Axis2DComposite = ParseAxis2DCompositeBinding( bindingElement );
203 break;
204 case InputBindingKind.Delta2D:
205 definition.Delta2D = ParseDelta2DBinding( bindingElement );
206 break;
207 default:
208 throw new Exception( $"Unsupported binding kind '{kind}'." );
209 }
210
211 return definition;
212 }

236 ===============

244 private static ButtonBinding ParseButtonBinding( JsonElement bindingElement ) {
245 InputDeviceSlot deviceId = ParseDeviceSlot( GetRequiredString( bindingElement, "DeviceId" ) );
246 string controlName = GetRequiredString( bindingElement, "ControlId", GetOptionalString( bindingElement, "ButtonId" ) );
247
248 ImmutableArray<InputControlId>.Builder modifiers = ImmutableArray.CreateBuilder<InputControlId>();
249 if ( JsonLoader.TryGetProperty( bindingElement, "Modifiers", out JsonElement modifiersElement ) ) {
250 if ( modifiersElement.ValueKind != JsonValueKind.Array ) {
251 throw new Exception( "Binding modifiers must be an array." );
252 }
253 foreach ( JsonElement modifierElement in modifiersElement.EnumerateArray() ) {
254 if ( modifierElement.ValueKind != JsonValueKind.String ) {
255 throw new Exception( "Binding modifiers must be strings." );
256 }
257 modifiers.Add( ParseKeyboardControl( modifierElement.GetString()! ) );
258 }
259 }
260
261 return new ButtonBinding(
262 deviceId: deviceId,
263 controlId: ParseButtonControl( deviceId, controlName ),
264 modifiers: modifiers.ToImmutable()
265 );
266 }

380 private static InputDeviceSlot ParseDeviceSlot( string deviceId ) {
381 if ( deviceId.Equals( Constants.KEYBOARD_DEVICE_ID, StringComparison.OrdinalIgnoreCase ) ) {
382 return InputDeviceSlot.Keyboard;
383 }
384
385 if ( deviceId.Equals( Constants.MOUSE_BUTTON_DEVICE_ID, StringComparison.OrdinalIgnoreCase )
386 || deviceId.Equals( Constants.MOUSE_MOTION_DEVICE_ID, StringComparison.OrdinalIgnoreCase )
387 || deviceId.Equals( "Mouse", StringComparison.OrdinalIgnoreCase ) )
388 {
389 return InputDeviceSlot.Mouse;
390 }
391
392 if ( deviceId.Equals( Constants.GAMEPAD_DEVICE_ID, StringComparison.OrdinalIgnoreCase )
393 || deviceId.Equals( nameof( InputDeviceSlot.Gamepad0 ), StringComparison.OrdinalIgnoreCase ) )
394 {
395 return InputDeviceSlot.Gamepad0;
396 }
397 if ( deviceId.Equals( nameof( InputDeviceSlot.Gamepad1 ), StringComparison.OrdinalIgnoreCase ) ) {
398 return InputDeviceSlot.Gamepad1;
399 }
400 if ( deviceId.Equals( nameof( InputDeviceSlot.Gamepad2 ), StringComparison.OrdinalIgnoreCase ) ) {
401 return InputDeviceSlot.Gamepad2;
402 }
403 if ( deviceId.Equals( nameof( InputDeviceSlot.Gamepad3 ), StringComparison.OrdinalIgnoreCase ) ) {
404 return InputDeviceSlot.Gamepad3;
405 }
406 throw new Exception( $"Invalid DeviceId '{deviceId}' in bindings file." );
407 }

409 private static InputControlId ParseButtonControl( InputDeviceSlot deviceId, string controlName ) {
410 return deviceId switch {
411 InputDeviceSlot.Keyboard => ParseKeyboardControl( controlName ),
412 InputDeviceSlot.Mouse => ParseMouseButtonControl( controlName ),
413 InputDeviceSlot.Gamepad0 or InputDeviceSlot.Gamepad1 or InputDeviceSlot.Gamepad2 or InputDeviceSlot.Gamepad3 => ParseGamepadButtonControl( controlName ),
414 _ => throw new Exception( $"Unsupported button device '{deviceId}'." )

464 private static string GetRequiredString( JsonElement element, string propertyName, string? fallback = null ) {
465 if ( element.ValueKind == JsonValueKind.String ) {
466 return JsonLoader.Read<string>( element, propertyName );
467 }
468 if ( JsonLoader.TryGet( element, propertyName, out string? value ) ) {
469 return value;
470 }
471 if ( fallback != null ) {
472 return fallback;
473 }
474 throw new Exception( $"Bindings file is missing string property '{propertyName}'." );
475 }

477 private static string? GetOptionalString( JsonElement element, string propertyName ) {
478 return JsonLoader.TryGet( element, propertyName, out string? value ) ? value : null;
