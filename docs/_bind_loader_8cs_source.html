65 public bool LoadBindDatabase( string filePath, out ImmutableArray<InputActionDefinition> binds ) {
66 if ( !_fileSystem.FileExists( filePath ) ) {
67 binds = ImmutableArray<InputActionDefinition>.Empty;
68 return false;
69 }
70
71 using var document = JsonDocument.Parse(
72 _fileSystem.LoadFile( filePath ).Memory,
73 new JsonDocumentOptions {
74 CommentHandling = JsonCommentHandling.Skip,
75 AllowTrailingCommas = true
76 }
77 );
78
79 if ( !TryGetProperty( document.RootElement, "Bindings", out var bindingsElement ) ) {
80 throw new Exception( $"Bindings file '{filePath}' is missing a 'Bindings' property." );
81 }
82
83 var actionIndices = new Dictionary<string, int>( StringComparer.Ordinal );
84 var actions = new List<ActionBuilder>();
85
86 switch ( bindingsElement.ValueKind ) {
87 case JsonValueKind.Array:
88 foreach ( var actionElement in bindingsElement.EnumerateArray() ) {
89 ParseActionDefinition( actionElement, actionIndices, actions );
90 }
91 break;
92 case JsonValueKind.Object:
93 foreach ( var actionProperty in bindingsElement.EnumerateObject() ) {
94 ParseActionDefinition( actionProperty.Value, actionIndices, actions, actionProperty.Name );
95 }
96 break;
97 default:
98 throw new Exception( $"Bindings file '{filePath}' has an invalid 'Bindings' payload. Expected an array or object." );
99 }
100
101 var builder = ImmutableArray.CreateBuilder<InputActionDefinition>( actions.Count );
102 for ( int i = 0; i < actions.Count; i++ ) {
103 builder.Add( actions[i].Build() );
104 }
105 binds = builder.ToImmutable();
106 return true;
107 }

122 private static void ParseActionDefinition( JsonElement actionElement, Dictionary<string, int> actionIndices, List<ActionBuilder> actions, string? fallbackName = null ) {
123 string name = GetRequiredString( actionElement, "Name", fallbackName );
124 InputValueType valueType = ParseEnum<InputValueType>( GetRequiredString( actionElement, "ValueType" ), "ValueType" );
125 InputScheme scheme = ParseEnum<InputScheme>( GetRequiredString( actionElement, "Scheme" ), "Scheme" );
126
127 if ( !TryGetProperty( actionElement, "Bindings", out var bindingsElement ) && !TryGetProperty( actionElement, "Binding", out bindingsElement ) ) {
128 throw new Exception( $"Binding action '{name}' is missing a binding payload." );
129 }
130
131 if ( !actionIndices.TryGetValue( name, out int actionIndex ) ) {
132 actionIndex = actions.Count;
133 actionIndices.Add( name, actionIndex );
134 actions.Add( new ActionBuilder( name, valueType ) );
135 } else if ( actions[actionIndex].ValueType != valueType ) {
136 throw new Exception( $"Binding action '{name}' has conflicting value types." );
137 }
138
139 if ( bindingsElement.ValueKind == JsonValueKind.Array ) {
140 foreach ( var bindingElement in bindingsElement.EnumerateArray() ) {
141 actions[actionIndex].Bindings.Add( ParseBindingDefinition( bindingElement, scheme ) );
142 }
143 return;
144 }
145
146 if ( bindingsElement.ValueKind != JsonValueKind.Object ) {
147 throw new Exception( $"Binding action '{name}' has an invalid binding payload." );
148 }
149
150 actions[actionIndex].Bindings.Add( ParseBindingDefinition( bindingsElement, scheme ) );
151 }

165 private static InputBindingDefinition ParseBindingDefinition( JsonElement bindingElement, InputScheme scheme ) {
166 InputBindingKind kind = TryGetProperty( bindingElement, "Kind", out var kindElement )
167 ? ParseEnum<InputBindingKind>( GetRequiredString( kindElement, "Kind value" ), "Kind" )
168 : InferBindingKind( bindingElement );
169
170 var definition = new InputBindingDefinition {
171 Scheme = scheme,
172 Kind = kind
173 };
174
175 switch ( kind ) {
176 case InputBindingKind.Button:
177 definition.Button = ParseButtonBinding( bindingElement );
178 break;
179 case InputBindingKind.Axis1D:
180 definition.Axis1D = ParseAxis1DBinding( bindingElement );
181 break;
182 case InputBindingKind.Axis1DComposite:
183 definition.Axis1DComposite = ParseAxis1DCompositeBinding( bindingElement );
184 break;
185 case InputBindingKind.Axis2D:
186 definition.Axis2D = ParseAxis2DBinding( bindingElement );
187 break;
188 case InputBindingKind.Axis2DComposite:
189 definition.Axis2DComposite = ParseAxis2DCompositeBinding( bindingElement );
190 break;
191 case InputBindingKind.Delta2D:
192 definition.Delta2D = ParseDelta2DBinding( bindingElement );
193 break;
194 default:
195 throw new Exception( $"Unsupported binding kind '{kind}'." );
196 }
197
198 return definition;
199 }

232 private static ButtonBinding ParseButtonBinding( JsonElement bindingElement ) {
233 InputDeviceSlot deviceId = ParseDeviceSlot( GetRequiredString( bindingElement, "DeviceId" ) );
234 string controlName = GetRequiredString( bindingElement, "ControlId", GetOptionalString( bindingElement, "ButtonId" ) );
235
236 var modifiers = ImmutableArray.CreateBuilder<InputControlId>();
237 if ( TryGetProperty( bindingElement, "Modifiers", out var modifiersElement ) ) {
238 if ( modifiersElement.ValueKind != JsonValueKind.Array ) {
239 throw new Exception( "Binding modifiers must be an array." );
240 }
241
242 foreach ( var modifierElement in modifiersElement.EnumerateArray() ) {
243 if ( modifierElement.ValueKind != JsonValueKind.String ) {
244 throw new Exception( "Binding modifiers must be strings." );
245 }
246 modifiers.Add( ParseKeyboardControl( modifierElement.GetString()! ) );
247 }
248 }
249
250 return new ButtonBinding {
251 DeviceId = deviceId,
252 ControlId = ParseButtonControl( deviceId, controlName ),
253 Modifiers = modifiers.ToImmutable()
254 };
255 }

369 private static InputDeviceSlot ParseDeviceSlot( string deviceId ) {
370 if ( deviceId.Equals( Constants.KEYBOARD_DEVICE_ID, StringComparison.OrdinalIgnoreCase ) ) {
371 return InputDeviceSlot.Keyboard;
372 }
373
374 if ( deviceId.Equals( Constants.MOUSE_BUTTON_DEVICE_ID, StringComparison.OrdinalIgnoreCase )
375 || deviceId.Equals( Constants.MOUSE_MOTION_DEVICE_ID, StringComparison.OrdinalIgnoreCase )
376 || deviceId.Equals( "Mouse", StringComparison.OrdinalIgnoreCase ) )
377 {
378 return InputDeviceSlot.Mouse;
379 }
380
381 if ( deviceId.Equals( Constants.GAMEPAD_DEVICE_ID, StringComparison.OrdinalIgnoreCase )
382 || deviceId.Equals( nameof( InputDeviceSlot.Gamepad0 ), StringComparison.OrdinalIgnoreCase ) )
383 {
384 return InputDeviceSlot.Gamepad0;
385 }
386
387 if ( deviceId.Equals( nameof( InputDeviceSlot.Gamepad1 ), StringComparison.OrdinalIgnoreCase ) ) {
388 return InputDeviceSlot.Gamepad1;
389 }
390
391 if ( deviceId.Equals( nameof( InputDeviceSlot.Gamepad2 ), StringComparison.OrdinalIgnoreCase ) ) {
392 return InputDeviceSlot.Gamepad2;
393 }
394
395 if ( deviceId.Equals( nameof( InputDeviceSlot.Gamepad3 ), StringComparison.OrdinalIgnoreCase ) ) {
396 return InputDeviceSlot.Gamepad3;
397 }
398
399 throw new Exception( $"Invalid DeviceId '{deviceId}' in bindings file." );
400 }

457 private static string GetRequiredString( JsonElement element, string propertyName, string? fallback = null ) {
458 if ( element.ValueKind == JsonValueKind.String ) {
459 return element.GetString() ?? throw new Exception( $"Encountered null string while reading '{propertyName}'." );
460 }
461 if ( TryGetProperty( element, propertyName, out var property ) && property.ValueKind == JsonValueKind.String ) {
462 return property.GetString() ?? throw new Exception( $"Encountered null string while reading '{propertyName}'." );
463 }
464 if ( fallback != null ) {
465 return fallback;
466 }
467 throw new Exception( $"Bindings file is missing string property '{propertyName}'." );
468 }

470 private static string? GetOptionalString( JsonElement element, string propertyName ) {
471 if ( TryGetProperty( element, propertyName, out var property ) && property.ValueKind == JsonValueKind.String ) {
472 return property.GetString();
473 }
474 return null;
475 }

477 private static float GetOptionalSingle( JsonElement element, string propertyName, float defaultValue ) {
478 if ( !TryGetProperty( element, propertyName, out var property ) ) {
479 return defaultValue;
480 }
481 if ( property.ValueKind == JsonValueKind.Number && property.TryGetSingle( out float value ) ) {
482 return value;
483 }
484 throw new Exception( $"Binding property '{propertyName}' must be a number." );
485 }

487 private static bool GetOptionalBoolean( JsonElement element, string propertyName, bool defaultValue ) {
488 if ( !TryGetProperty( element, propertyName, out var property ) ) {
489 return defaultValue;
490 }
491 if ( property.ValueKind is JsonValueKind.True or JsonValueKind.False ) {
492 return property.GetBoolean();
493 }
494 throw new Exception( $"Binding property '{propertyName}' must be a boolean." );
495 }

497 private static TEnum GetOptionalEnum<TEnum>( JsonElement element, string propertyName, TEnum defaultValue )
498 where TEnum : struct, Enum
499 {
500 if ( !TryGetProperty( element, propertyName, out var property ) ) {
501 return defaultValue;
502 }
503 if ( property.ValueKind != JsonValueKind.String ) {
504 throw new Exception( $"Binding property '{propertyName}' must be a string." );
505 }
506 return ParseEnum<TEnum>( property.GetString()!, propertyName );
507 }
508
509 private static TEnum ParseEnum<TEnum>( string value, string propertyName )
510 where TEnum : struct, Enum
511 {
512 if ( Enum.TryParse<TEnum>( value, true, out var result ) ) {
513 return result;
514 }
515 throw new Exception( $"Invalid {propertyName} '{value}' in bindings file." );
516 }
517
518 private static bool TryGetProperty( JsonElement element, string propertyName, out JsonElement propertyValue ) {
519 if ( element.ValueKind == JsonValueKind.Object ) {
520 foreach ( var property in element.EnumerateObject() ) {
521 if ( property.Name.Equals( propertyName, StringComparison.OrdinalIgnoreCase ) ) {
522 propertyValue = property.Value;
523 return true;
524 }
525 }
526 }
527 propertyValue = default;
528 return false;
529 }
