54 public SubscriptionGroup( string name, IGameEventRegistryService eventRegistry ) {
55 ArgumentGuard.ThrowIfNullOrEmpty( name, nameof( name ) );
56 _name = name;
57 _eventFactory = eventRegistry;
58 }
59
60 /*
61 ===============
62 Dispose
63 ===============
64 */
68 public void Dispose() {
69 if ( !_isDisposed ) {
70 UnsubscribeAll();
71 _eventFactory.RemoveGroup( this );
72 }
73 GC.SuppressFinalize( this );
74 _isDisposed = true;
75 }
76
77 /*
78 ===============
79 Add
80 ===============
81 */
88 public void Add<TArgs>( IGameEvent<TArgs> gameEvent, EventCallback<TArgs> callback )
89 where TArgs : struct
90 {
91 _subscriptions.Add( gameEvent.Subscribe( callback ) );
92 }
93
94 /*

102 public void UnsubscribeAll() {
103 _subscriptions.Clear();
104 }
