94 public ICVar<T> Register<T>( in CVarCreateInfo<T> createInfo ) {
95 var cvar = _repository.AddCVar( in createInfo, _eventFactory );
96 if ( GetCVarGroup( createInfo.Group ?? "Default", out var group ) ) {
97 group.AddCVar( cvar );

100 _logger.PrintLine( $"CVarSystem.Register: registered CVar '{createInfo.Name}' with default value {createInfo.DefaultValue} and flags {createInfo.Flags}." );

126 public void AddGroup( string groupName ) {
127 ArgumentGuard.ThrowIfNullOrEmpty( groupName );
128
129 if ( _groups.ContainsKey( groupName ) ) {
130 throw new InvalidOperationException( $"CVarGroup {groupName} added twice" );
131 }
132 _groups.TryAdd( groupName, new CVarGroup( groupName, _logger ) );
133
134 _logger.PrintLine( $"CVarSystem.AddGroup: Added CVar group {groupName}." );
135 }

147 public void Save( IFileSystem fileSystem, string configFile ) {
148 ArgumentGuard.ThrowIfNullOrEmpty( configFile );

156 /*
157 ===============
158 Load
159 ===============
160 */
166 public void Load( IFileSystem fileSystem, string configFile ) {
167 ArgumentGuard.ThrowIfNullOrEmpty( configFile );
168
169 // ensure we block all access
170 lock ( _repository ) {
171 IniLoader reader = new IniLoader( configFile, _logger, fileSystem );
172
173 foreach ( var group in _groups ) {
174 foreach ( var cvar in group.Value.CVars ) {
175 string name = $"{group.Key}:{cvar.Name}";
176 switch ( cvar.Type ) {
177 case CVarType.Int:
178 if ( reader.LoadConfigValue( name, out int intValue ) ) {
179 cvar.SetIntegerValue( intValue );
180 }
181 break;
182 case CVarType.UInt:
183 if ( reader.LoadConfigValue( name, out uint uintValue ) ) {
184 cvar.SetUIntegerValue( uintValue );
185 }
186 break;
187 case CVarType.Decimal:
188 if ( reader.LoadConfigValue( name, out float floatValue ) ) {
189 cvar.SetDecimalValue( floatValue );
190 }
191 break;
192 case CVarType.String:
193 if ( reader.LoadConfigValue( name, out string strValue ) ) {
194 cvar.SetStringValue( strValue );
195 }
196 break;
197 case CVarType.Boolean:
198 if ( reader.LoadConfigValue( name, out bool boolValue ) ) {
199 cvar.SetBooleanValue( boolValue );
200 }
201 break;
202 }
203 }
204 }
205 }
206 }

381 private bool GetCVarGroup( string name, out ICVarGroup? group ) {
382 group = null;
383 foreach ( var value in _groups ) {
384 if ( string.Equals( value.Key, name, StringComparison.CurrentCultureIgnoreCase ) ) {
385 group = value.Value;

387 }
