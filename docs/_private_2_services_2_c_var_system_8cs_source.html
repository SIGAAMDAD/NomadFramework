49 /*
50 ===============
51 CVarSystem
52 ===============
53 */
60 public CVarSystem( IGameEventRegistryService eventFactory, ILoggerService logger ) {
61 ArgumentGuard.ThrowIfNull( logger, nameof( logger ) );
62
63 _eventFactory = eventFactory ?? throw new ArgumentNullException( nameof( eventFactory ) );
64 _repository = new CVarRepository( logger, eventFactory );
65
66 _category = logger.CreateCategory( nameof( CVarSystem ), LogLevel.Info, true );
67
68 AddGroup( "Default" );
69 }

79 public void Dispose() {

82 _groups.Clear();

84 GC.SuppressFinalize( this );
85 _isDisposed = true;
86 }

99 public ICVar<T> Register<T>( CVarCreateInfo<T> createInfo ) {
100 var cvar = _repository.AddCVar( createInfo, _eventFactory );
101 if ( !GetCVarGroup( createInfo.Group ?? "Default", out var group ) ) {
102 var newGroup = new CVarGroup( createInfo.Group, _category );
103 _groups.TryAdd( newGroup.Name, newGroup );
104 group = newGroup;
105 }
106 group.AddCVar( cvar );
107
108 _category.PrintLine( $"CVarSystem.Register: registered CVar '{createInfo.Name}' with default value {createInfo.DefaultValue} and flags {createInfo.Flags}." );
109 return cvar;
110 }

135 public void AddGroup( string groupName ) {
136 ArgumentGuard.ThrowIfNullOrEmpty( groupName );
137
138 if ( _groups.ContainsKey( groupName ) ) {
139 throw new InvalidOperationException( $"CVarGroup {groupName} added twice" );
140 }
141 _groups.TryAdd( groupName, new CVarGroup( groupName, _category ) );
142
143 _category.PrintLine( $"CVarSystem.AddGroup: Added CVar group {groupName}." );
144 }

165 /*
166 ===============
167 Load
168 ===============
169 */
175 public void Load( IFileSystem fileSystem, string configFile ) {
176 ArgumentGuard.ThrowIfNullOrEmpty( configFile );
177
178 // ensure we block all access
179 lock ( _repository ) {
180 _category.PrintLine( "Loading cvar configuration..." );
181 IniLoader reader = new IniLoader( configFile, _category, fileSystem );
182
183 foreach ( var group in _groups ) {
184 foreach ( var cvar in group.Value.CVars ) {
185 string name = $"{group.Key}:{cvar.Name}";
186 _category.PrintLine( $"Loading CVar '{cvar.Name}'" );
187 switch ( cvar.Type ) {
188 case CVarType.Int:
189 if ( reader.LoadConfigValue( name, out int intValue ) ) {
190 cvar.SetIntegerValue( intValue );
191 }
192 break;
193 case CVarType.UInt:
194 if ( reader.LoadConfigValue( name, out uint uintValue ) ) {
195 cvar.SetUIntegerValue( uintValue );
196 }
197 break;
198 case CVarType.Decimal:
199 if ( reader.LoadConfigValue( name, out float floatValue ) ) {
200 cvar.SetDecimalValue( floatValue );
201 }
202 break;
203 case CVarType.String:
204 if ( reader.LoadConfigValue( name, out string strValue ) ) {
205 cvar.SetStringValue( strValue );
206 }
207 break;
208 case CVarType.Boolean:
209 if ( reader.LoadConfigValue( name, out bool boolValue ) ) {
210 cvar.SetBooleanValue( boolValue );
211 }
212 break;
213 }
214 }
215 }
216 }

392 private bool GetCVarGroup( string name, out ICVarGroup? group ) {
393 group = null;
394 foreach ( var value in _groups ) {
