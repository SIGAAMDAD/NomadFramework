123 internal CVar( IGameEventRegistryService eventFactory, CVarCreateInfo<T> createInfo ) {
124 ArgumentGuard.ThrowIfNullOrEmpty( createInfo.Name );
125 ArgumentGuard.ThrowIfNull( createInfo.Description );
126
127 _validator = new CVarValidator<T>( createInfo.Validator );
128 if ( !CVarValidator<T>.ValidateCVarType() ) {
129 throw new InvalidCastException( "Invalid CVar type!" );
130 }
131
132 _metadata = new CVarMetadata(
133 createInfo.Name,
134 createInfo.Description,
135 createInfo.Flags,
136 typeof( T ).GetCVarType()
137 );
138
139 _converter = new CVarConverter<T>( _metadata.Type, createInfo.DefaultValue );
140 _defaultValue = createInfo.DefaultValue;
141
142 _valueChanged = eventFactory.GetEvent<CVarValueChangedEventArgs<T>>( $"{createInfo.Name}:{CVarValueChangedEventArgs.Name}", CVarValueChangedEventArgs.NameSpace );
143 }

168 [MethodImpl( MethodImplOptions.AggressiveInlining )]
169 public void SetFromString( string value ) {
170 ArgumentGuard.ThrowIfNull( value );
171 if ( !CVarStringConverter.TryParse( value, typeof( T ), out object? convertedValue ) ) {
172 throw new ArgumentException( $"Failed to convert cvar value '{value}' to type {typeof( T ).Name}" );
173 }
174 Value = (T)convertedValue!;
175 }

177 /*
178 ===============
179 Set
180 ===============

186 public void Set( T value ) {
187 if ( IsReadOnly || !_validator.ValidateValue( value ) || EqualityComparer<T>.Default.Equals( _converter.Value, value ) ) {
188 return;
189 }
190 T old = _converter.Value;
191 _converter.Value = value;
192 _valueChanged.Publish( new CVarValueChangedEventArgs<T>( old, value ) );
193 }
