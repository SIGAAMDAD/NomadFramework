48 public FileSink( ICVarSystemService cvarSystem, IFileSystem fileSystem ) {
49 var logfile = cvarSystem.Register(
50 new CVarCreateInfo<string> {
51 Name = "console.LogFile",
52 DefaultValue = "debug.log",
53 Description = "The path to the console's logging file.",
54 Flags = CVarFlags.Archive | CVarFlags.Developer,
55 Validator = file => file.Length > 0
56 }
57 );
58
59 try {
60 _writer = fileSystem.OpenWrite(
61 new FileWriteConfig {
62 FilePath = logfile.Value,
63 Append = false
64 }
65 ) as IFileWriteStream;
66 } catch ( Exception e ) {
67 _writer?.Close();
68 Console.WriteLine( $"FileSink: failed to create log file {logfile.Value} - {e.Message}" );
69 throw;
70 }
71 Console.WriteLine( $"Sucessfully opened logfile at {DateTime.Now}." );
72 }
73

77 ===============
78 */
82 protected override void Dispose( bool disposing ) {
83 if ( !isDisposed ) {

85 }
86 base.Dispose( disposing );
