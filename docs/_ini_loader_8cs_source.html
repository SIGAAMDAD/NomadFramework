50 public IniLoader( string configFile, ILoggerService logger, IFileSystem fileSystem ) {
51 ArgumentGuard.ThrowIfNullOrEmpty( configFile );
52 ArgumentGuard.ThrowIfNull( logger );
53
54 _iniData = null;
55
56 try {
57 using var fileStream = new FileStream( $"{fileSystem.GetConfigPath()}/{configFile}", FileMode.Open, FileAccess.Read );
58
59 _iniData = IniStreamConfigurationProvider.Read( fileStream );
60 if ( _iniData == null ) {
61 logger.PrintError( $"IniLoader: error parsing ini data from configuration file '{configFile}'" );
62 }
63 } catch ( Exception e ) {
64 logger.PrintError( $"IniLoader: Error opening configuration file '{configFile}: {e.Message}" );
65 throw;
66 }
67 }

148 public bool LoadConfigValue( string name, out bool value ) {
149 ArgumentGuard.ThrowIfNull( _iniData );
150 ArgumentGuard.ThrowIfNullOrEmpty( name );
151
152 if ( _iniData.TryGetValue( name, out string? data ) ) {
153 ArgumentGuard.ThrowIfNullOrEmpty( data );
154
155 string str = data!.Trim().ToLower();
156 if ( str == "1" || str == "true" || str == "yes" || str == "on" ) {
157 value = true;
158 } else if ( str == "0" || str == "false" || str == "no" || str == "off" ) {
159 value = false;
160 } else {
161 value = bool.TryParse( data, out bool result ) && result;
162 }
163 return true;
164 }
165 value = false;
166 return false;
167 }
