51 public IniLoader( string configFile, ILoggerCategory logger, IFileSystem fileSystem ) {
52 ArgumentGuard.ThrowIfNullOrEmpty( configFile );
53 ArgumentGuard.ThrowIfNull( logger );
54
55 _iniData = null;
56
57 logger.PrintLine( $"Loading configuration file '{configFile}'..." );
58
59 try {
60 string resolvedPath = ResolveConfigPath( configFile, fileSystem );
61 IBufferHandle? fileBuffer = fileSystem.LoadFile( resolvedPath ) ??
62 throw new FileNotFoundException( $"Could not find file '{resolvedPath}'.", resolvedPath );
63
64 using ( fileBuffer )
65 using ( var fileStream = fileBuffer.AsStream( 0, fileBuffer.Length ) ) {
66 _iniData = IniStreamConfigurationProvider.Read( fileStream );
67 if ( _iniData == null ) {
68 logger.PrintError( $"IniLoader: error parsing ini data from configuration file '{resolvedPath}'" );
69 }
70 }
71 } catch ( Exception e ) {
72 logger.PrintError( $"IniLoader: Error opening configuration file '{configFile}: {e.Message}" );
73 throw;
74 }
75 }

77 private static string ResolveConfigPath( string configFile, IFileSystem fileSystem ) {
78 ArgumentGuard.ThrowIfNull( fileSystem );
79
80 if ( Path.IsPathRooted( configFile ) || !string.IsNullOrEmpty( Path.GetDirectoryName( configFile ) ) ) {
81 return configFile;
82 }
83
84 string configDirectoryCandidate = Path.Combine( fileSystem.GetConfigPath(), configFile );
85 return fileSystem.FileExists( configDirectoryCandidate ) ? configDirectoryCandidate : configFile;
86 }

167 public bool LoadConfigValue( string name, out bool value ) {
168 ArgumentGuard.ThrowIfNull( _iniData );
169 ArgumentGuard.ThrowIfNullOrEmpty( name );
170
171 if ( _iniData.TryGetValue( name, out string? data ) ) {
172 ArgumentGuard.ThrowIfNullOrEmpty( data );
173
174 string str = data!.Trim().ToLower();
175 if ( str == "1" || str == "true" || str == "yes" || str == "on" ) {
176 value = true;
177 } else if ( str == "0" || str == "false" || str == "no" || str == "off" ) {
178 value = false;
179 } else {
180 value = bool.TryParse( data, out bool result ) && result;
181 }
182 return true;
183 }
184 value = false;
185 return false;
186 }
