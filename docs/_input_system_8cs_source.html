87 public InputSystem( IFileSystem fileSystem, ICVarSystemService cvarSystem, ILoggerService logger, IGameEventRegistryService eventFactory, IServiceRegistry registry ) {
88 ArgumentGuard.ThrowIfNull( eventFactory, nameof( eventFactory ) );
89 ArgumentGuard.ThrowIfNull( cvarSystem, nameof( cvarSystem ) );
90 ArgumentGuard.ThrowIfNull( registry, nameof( registry ) );
91
92 InputCVarRegistry.RegisterCVars( cvarSystem );
93
94 _bindRepository = new BindRepository( fileSystem, cvarSystem, logger );
95 _compiledBindings = new CompiledBindingRepository();
96 _compilerService = new BindingCompilerService( _compiledBindings );
97
98 _stateService = new InputStateService();
99 _dispatchService = new InputDispatchService( eventFactory );
100 _matcherService = new BindingMatcherService( _compiledBindings, _stateService );
101 _actionResolverService = new ActionResolverService( _compiledBindings, _stateService );
102
103 _compilerService.CompileIntoRepository( _bindRepository.GetAllBindings() );
104
105 _bindResolver = new BindResolver( _bindRepository, RecompileBindings );
106 _rebindService = new InputRebindService( _bindRepository, _compilerService, eventFactory );
107
108 registry.AddSingleton( _bindResolver );
109 registry.AddSingleton( _rebindService );
110 registry.AddSingleton<IInputSnapshotService>( _stateService );
111
112 _pauseStateChanged = eventFactory
113 .GetEvent<PauseStateChangedEventArgs>( PauseStateChangedEventArgs.Name, PauseStateChangedEventArgs.NameSpace )
114 .Subscribe( OnPauseStateChanged );
115
116 _keyboardEvent = eventFactory
117 .GetEvent<KeyboardEventArgs>( KeyboardEventArgs.Name, KeyboardEventArgs.NameSpace )
118 .Subscribe( OnKeyboardEventTriggered );
119
120 _mouseButtonEvent = eventFactory
121 .GetEvent<MouseButtonEventArgs>( MouseButtonEventArgs.Name, MouseButtonEventArgs.NameSpace )
122 .Subscribe( OnMouseButtonEventTriggered );
123
124 _mouseMotionEvent = eventFactory
125 .GetEvent<MouseMotionEventArgs>( MouseMotionEventArgs.Name, MouseMotionEventArgs.NameSpace )
126 .Subscribe( OnMouseMotionEventTriggered );
127
128 _mousePositionChangedEvent = eventFactory
129 .GetEvent<MousePositionChangedEventArgs>( MousePositionChangedEventArgs.Name, MousePositionChangedEventArgs.NameSpace )
130 .Subscribe( OnMousePositionChangedEventTriggered );
131
132 _gamepadAxisEvent = eventFactory
133 .GetEvent<GamepadAxisEventArgs>( GamepadAxisEventArgs.Name, GamepadAxisEventArgs.NameSpace )
134 .Subscribe( OnGamepadAxisEventTriggered );
135
136 _gamepadButtonEvent = eventFactory
137 .GetEvent<GamepadButtonEventArgs>( GamepadButtonEventArgs.Name, GamepadButtonEventArgs.NameSpace )
138 .Subscribe( OnGamepadButtonEventTriggered );
139 }
140
141 public void Dispose() {
142 if ( _isDisposed ) {
143 return;
144 }
145
146 _pauseStateChanged.Dispose();
147 _mousePositionChangedEvent.Dispose();
148 _mouseMotionEvent.Dispose();
149 _mouseButtonEvent.Dispose();
150 _gamepadAxisEvent.Dispose();
151 _gamepadButtonEvent.Dispose();
152 _keyboardEvent.Dispose();
153
154 ( _rebindService as IDisposable ).Dispose();
155 _bindRepository.Dispose();
156 _stateService.Dispose();
157
158 _isDisposed = true;
159 GC.SuppressFinalize( this );
160 }

171 public void PushKeyboardEvent( in KeyboardEventArgs keyEvent ) {
172 _stateService.SetPressed( InputDeviceSlot.Keyboard, keyEvent.KeyNum.ToControlId(), keyEvent.Pressed );
173
174 CompiledBindingGraph graph = _compiledBindings.Current;
175 BindingMatchSet matches = _matcherService.MatchKeyboard( graph, in keyEvent, _contextMask, _mode );
176
177 DispatchResolved( graph, _actionResolverService.ResolveMatchesNonAlloc( graph, matches, keyEvent.TimeStamp ) );
178 DispatchResolved( graph, _actionResolverService.ResolveKeyboardCompositesNonAlloc( graph, _contextMask, _mode, keyEvent.TimeStamp ) );
179 }

285 private void DispatchResolved( CompiledBindingGraph graph, ReadOnlySpan<ResolvedAction> actions ) {
286 for ( int i = 0; i < actions.Length; i++ ) {
287 _dispatchService.Dispatch( graph, in actions[i] );

314 private static InputDeviceSlot GetGamepadDeviceSlot( int deviceId ) {
315 return deviceId switch {
316 0 => InputDeviceSlot.Gamepad0,
317 1 => InputDeviceSlot.Gamepad1,
318 2 => InputDeviceSlot.Gamepad2,
319 3 => InputDeviceSlot.Gamepad3,
320 _ => throw new ArgumentOutOfRangeException( nameof( deviceId ) )
321 };
