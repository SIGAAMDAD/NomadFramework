96 internal GameEvent( InternString nameSpace, InternString name, ILoggerService logger, EventFlags flags ) {
97 ArgumentGuard.ThrowIfNullOrEmpty( name );
98
99 _nameSpace = nameSpace;
100 _name = name;
101 _hashCode = HashCode.Combine( _name.GetHashCode(), nameSpace.GetHashCode(), typeof( TArgs ).TypeHandle.ToString() );
102
103 bool isSynchronous = flags.HasFlag( EventFlags.Synchronous );
104 bool isAsync = flags.HasFlag( EventFlags.Asynchronous );
105 bool lockFree = flags.HasFlag( EventFlags.NoLock );
106
107 if ( lockFree && isAsync ) {
108 throw new NotSupportedException( "Cannot have an event that is both lock free and asynchronous!" );
109 }
110
111 if ( lockFree ) {
112 _subscriptions = new LockFreeSubscriptionSet<TArgs>( this, logger );
113 } else if ( isAsync ) {
114 _subscriptions = new SubscriptionSet<TArgs>( this, logger );
115 }
116 }
