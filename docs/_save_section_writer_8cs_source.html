71 public SaveSectionWriter( SaveConfig config, ILoggerCategory category, string name, IMemoryFileWriteStream writer ) {

81 Dispose
82 ===============
83 */
87 public void Dispose() {
88 lock ( _disposeLock ) {
89 if ( !_isDisposed ) {
90 long offset = _writer.Position;
91 {
92 var header = new SectionHeader( _name, 0, FieldCount, Checksum.Empty );
93 header.Save( _writer );
94 }
95
96 long start = offset + SectionHeader.HEADER_CHECKSUM_OFFSET;
97 foreach ( var field in _fields ) {
98 SaveField.Write( _name, field.Value, _writer );
99 }
100
101 long position = _writer.Position;
102 long length = position - start;
103
104 _writer.Seek( offset, System.IO.SeekOrigin.Begin );
105 {
106 var header = new SectionHeader( _name, (int)length, FieldCount, Checksum.Compute( _writer.Buffer!.GetSlice( (int)start, (int)length ) ) );
107 header.Save( _writer );
108
109 if ( _config.LogSerializationTree ) {
110 _category.PrintLine( "Finalized section data:" );
111 _category.PrintLine( $"\tName: {header.Name}" );
112 _category.PrintLine( $"\tByteLength: {header.ByteLength}" );
113 _category.PrintLine( $"\tFieldCount: {header.FieldCount}" );
114 _category.PrintLine( $"\tChecksum64: {header.Checksum}" );
115 }
116 }
117 _writer.Seek( position, System.IO.SeekOrigin.Begin );
118 _fields.Clear();
119
120 _isDisposed = true;
121 }
122 }
123 GC.SuppressFinalize( this );
124 }

137 public void AddField<T>( string fieldId, T value ) {
138 StateGuard.ThrowIfDisposed( _isDisposed, this );
139 ArgumentGuard.ThrowIfNullOrEmpty( fieldId );
140
141 if ( _fields.ContainsKey( fieldId ) ) {
142 throw new DuplicateFieldException( _name, fieldId );
143 }
144
145 var type = Any.GetType<T>();
146 var fieldValue = Any.From( value );
147
148 _fields[fieldId] = new SaveField(
149 fieldId,
150 type,
151 fieldValue
152 );
153 if ( _config.LogSerializationTree ) {
154 _category.PrintLine( $"\t\t[Field] (NAME) {fieldId}, (TYPE) {type}, (VALUE) {fieldValue}" );
155 }
156 }
