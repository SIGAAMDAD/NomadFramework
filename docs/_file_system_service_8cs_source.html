131 public void ReplaceFile( string sourcePath, string destPath, string destBackupPath ) {
132 //
133 // NOTE: File.Replace does not work reliably with the VFS nor does it play very nice with temporary files. So this is here
134 // because it is the only way to reliably replace files across platforms. In testing, the temporary file existed, and File.Replace
135 // was given a valid file path (File.Exists succeeded on it), yet File.Replace threw FileNotFoundException each time it was attempted,
136 // even when the source file exists and is accessible, likely due to VFS path translation issues. No fix has been found yet, and this
137 // works for now.
138 //
139 // It mimics the exact same behavior as File.Replace, but within the VFS. If there is a way to make File.Replace behave, then
140 // do tell and open a PR, but until then, this is what we've got.
141 //
142
143 // TODO: this copies only the contents, not the metadata, of the file. Do find a way of copying the metadata in the long run.
144
145 ArgumentGuard.ThrowIfNullOrEmpty( sourcePath );
146 ArgumentGuard.ThrowIfNullOrEmpty( destPath );
147
148 if ( sourcePath.Equals( destPath, StringComparison.OrdinalIgnoreCase ) ) {
149 return; // nothing to do
150 }
151
152 // ensure the source file exists
153 if ( !FileExists( sourcePath ) ) {
154 throw new FileNotFoundException( "Source file not found", sourcePath );
155 }
156
157 string destDir = Path.GetDirectoryName( destPath );
158 string tempFileName = Path.GetRandomFileName();
159 string tempPath = Path.Combine( destDir, tempFileName );
160
161 try {
162 using ( var sourceStream = OpenRead( new FileReadConfig { FilePath = sourcePath } ) )
163 using ( var tempStream = OpenWrite( new FileWriteConfig { FilePath = tempPath } ) ) {
164 sourceStream.WriteToStream( tempStream );
165 }
166 if ( !string.IsNullOrEmpty( destBackupPath ) && FileExists( destPath ) ) {
167 // if the backup already exists delete it, File.Replace does this already
168 if ( FileExists( destBackupPath ) ) {
169 DeleteFile( destBackupPath );
170 }
171 MoveFile( destPath, destBackupPath );
172 }
173 if ( FileExists( destPath ) ) {
174 DeleteFile( destPath );
175 }
176
177 MoveFile( tempPath, destPath );
178 } catch {
179 // cleanup the temp file if it still exists
180 if ( FileExists( tempPath ) ) {
181 DeleteFile( tempPath );
182 }
183 throw;
184 }
185 }

390 ===============
391 OpenRead
392 ===============
393 */
399 public IReadStream? OpenRead( IReadConfig config ) {
400 ArgumentGuard.ThrowIfNull( config );
401
402 switch ( config.Type ) {
403 case StreamType.File: {
404 var fileConfig = config as FileReadConfig ?? throw new InvalidCastException();
405 var fullPath = _searchHelper.FindFile( fileConfig.FilePath );
406 if ( fullPath == null ) {
407 _category.PrintLine( $"FileSystemService.OpenRead: couldn't find file '{fileConfig.FilePath}'" );
408 return null;
409 }
410 fileConfig = fileConfig with { FilePath = fullPath };
411 return new FileReadStream( fileConfig );
412 }
413 case StreamType.MemoryFile: {
414 var memoryFileConfig = config as MemoryFileReadConfig ?? throw new InvalidCastException();
415 var fullPath = _searchHelper.FindFile( memoryFileConfig.FilePath );
416 if ( fullPath == null ) {
417 _category.PrintLine( $"FileSystemService.OpenRead: couldn't find file '{memoryFileConfig.FilePath}'" );
418 return null;
419 }
420 memoryFileConfig = memoryFileConfig with { FilePath = fullPath };
421 return new MemoryFileReadStream( memoryFileConfig );
422 }
423 case StreamType.Memory:
424 return new MemoryReadStream(
425 config is MemoryReadConfig memoryConfig ? memoryConfig : throw new InvalidCastException()
426 );
427 default:
428 throw new ArgumentOutOfRangeException( nameof( config ) );
429 }
430 }

520 public IBufferHandle? LoadFile( string path ) {
521 var fullPath = _searchHelper.FindFile( path );
522 if ( fullPath == null ) {
523 return null;
524 }
525 using var stream = OpenRead( new FileReadConfig { FilePath = fullPath } ) ?? throw new IOException();
526
527 var handle = new PooledBufferHandle( (int)stream.Length );
528 stream.Read( handle.Span, 0, handle.Length );
529
530 return handle;
531 }
