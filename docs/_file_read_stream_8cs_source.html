193 public byte[] ReadToEnd() {
194 StateGuard.ThrowIfDisposed( isDisposed, this );
195
196 long remaining = fileStream.Length - fileStream.Position;
197 if ( remaining > int.MaxValue ) {
198 throw new InvalidOperationException( "File is too large to read into a single array." );
199 }
200 byte[] buffer = new byte[remaining];
201 int bytesRead = fileStream.Read( buffer, 0, (int)remaining );
202 if ( bytesRead != remaining ) {
203 throw new IOException( $"FileStream failed to read exactly {remaining} bytes!" );
204 }
205 return buffer;
206 }

322 public int Read7BitEncodedInt() {
323 int value = 0;
324 int shift = 0;
325 byte b;
326
327 do {
328 b = ReadUInt8();
329 value |= (b & 0x7F) << shift;
330 shift += 7;
331 if ( shift > 35 ) {
332 throw new FormatException( "Invalid 7-bit encoded integer formatting in file." );
333 }
334 } while ( (b & 0x80) != 0 );
335
336 return value;
337 }
