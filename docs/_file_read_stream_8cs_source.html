67 public FileReadStream( FileReadConfig config )
68 : base( config.FilePath!, FileMode.Open, FileAccess.Read )
69 {
70 ArgumentGuard.ThrowIfNull( fileStream );
71 _format = config.Format;
72 switch ( _format ) {
73 case StreamFormat.Binary:
74 _streamReader = new FileReader( new BinaryReader( fileStream ) );
75 break;
76 case StreamFormat.Utf8:
77 _streamReader = new FileReader( new StreamReader( fileStream ) );
78 break;
79 default:
80 Dispose();
81 throw new ArgumentOutOfRangeException( nameof( config ) );
82 }

225 public async ValueTask<byte[]> ReadToEndAsync( CancellationToken ct = default ) {
226 StateGuard.ThrowIfDisposed( isDisposed, this );
227
228 ct.ThrowIfCancellationRequested();
229
230 long remaining = fileStream.Length - fileStream.Position;
231 if ( remaining > int.MaxValue ) {
232 throw new InvalidOperationException( "File is too large to read into a single array." );
233 }
234 byte[] buffer = new byte[remaining];
235 int bytesRead = await fileStream.ReadAsync( buffer.AsMemory( 0, (int)remaining ), ct );
236 if ( bytesRead < remaining ) {
237 Array.Resize( ref buffer, bytesRead );
238 }
239 return buffer;
240 }

251 public void WriteToStream( IWriteStream stream ) {
252 StateGuard.ThrowIfDisposed( isDisposed, this );
253
254 long position = _streamReader.Position;
255 _streamReader.Position = 0;
256
257 int length = (int)Length;
258 byte[] buffer = ArrayPool<byte>.Shared.Rent( length );
259 Read( buffer, 0, length );
260 stream.Write( buffer, 0, length );
261 ArrayPool<byte>.Shared.Return( buffer );
262 _streamReader.Position = position;

329 public int Read7BitEncodedInt() {
330 int value = 0;
331 int shift = 0;
332 byte b;
333
334 do {
335 b = ReadUInt8();
336 value |= (b & 0x7F) << shift;
337 shift += 7;
338 if ( shift > 35 ) {
339 throw new FormatException( "Invalid 7-bit encoded integer formatting in file." );
340 }
341 } while ( (b & 0x80) != 0 );
342
343 return value;
344 }
