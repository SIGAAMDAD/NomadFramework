103 protected static void AppendPropertyMembers(StringBuilder builder, TemplateGenerationModel model)
104 {
105 foreach (var propertyImplementation in model.PropertyImplementations)
106 {
107 AppendDocumentationComment(builder, " ", propertyImplementation.DocumentationLines);
108 builder.Append(" public ");
109 if (propertyImplementation.RequiresNewKeyword)
110 {
111 builder.Append("new ");
112 }
113
114 builder.Append(propertyImplementation.PropertyTypeName);
115 builder.Append(' ');
116 builder.Append(propertyImplementation.PropertyName);
117 builder.AppendLine();
118 builder.AppendLine(" {");
119
120 if (propertyImplementation.GetterExpression is not null)
121 {
122 AppendAccessor(builder, "get", propertyImplementation.GetterExpression);
123 }
124
125 if (propertyImplementation.SetterExpression is not null)
126 {
127 AppendAccessor(builder, "set", propertyImplementation.SetterExpression);
128 }
129
130 builder.AppendLine(" }");
131
132 if (propertyImplementation.BackingFieldDeclaration is not null)
133 {
134 builder.AppendLine(" " + propertyImplementation.BackingFieldDeclaration);
135 }
136
137 builder.AppendLine();
138 }

141 private static void AppendAccessor(StringBuilder builder, string accessorKeyword, string accessorExpression)
142 {
143 if (!RequiresBlockAccessor(accessorExpression))
144 {
145 builder.AppendLine(" " + accessorKeyword + " => " + accessorExpression + ";");
146 return;
147 }
148
149 builder.AppendLine(" " + accessorKeyword);
150 builder.AppendLine(" {");
151
152 foreach (var line in accessorExpression.Split(new[] { "\r\n", "\n" }, System.StringSplitOptions.None))
153 {
154 if (line.Length == 0)
155 {
156 builder.AppendLine();
157 continue;
158 }
159
160 builder.AppendLine(" " + line);
161 }
162
163 builder.AppendLine(" }");
164 }

166 private static bool RequiresBlockAccessor(string accessorExpression)
167 {
168 var trimmedExpression = accessorExpression.TrimStart();
169 return trimmedExpression.StartsWith("if ", System.StringComparison.Ordinal) ||
170 trimmedExpression.StartsWith("if(", System.StringComparison.Ordinal) ||
171 trimmedExpression.StartsWith("switch ", System.StringComparison.Ordinal) ||
172 trimmedExpression.StartsWith("switch(", System.StringComparison.Ordinal) ||
173 trimmedExpression.StartsWith("for ", System.StringComparison.Ordinal) ||
174 trimmedExpression.StartsWith("for(", System.StringComparison.Ordinal) ||
175 trimmedExpression.StartsWith("foreach ", System.StringComparison.Ordinal) ||
176 trimmedExpression.StartsWith("while ", System.StringComparison.Ordinal) ||
177 trimmedExpression.StartsWith("while(", System.StringComparison.Ordinal) ||
178 trimmedExpression.StartsWith("{", System.StringComparison.Ordinal);
179 }

404 private static string FormatAccessibility(Accessibility accessibility)
405 => accessibility switch
406 {
407 Accessibility.Public => "public",
408 Accessibility.Internal => "public",
409 Accessibility.Private => "private",
410 Accessibility.Protected => "protected",
411 Accessibility.ProtectedAndInternal => "private protected",
412 Accessibility.ProtectedOrInternal => "protected internal",
413 _ => "internal"
414 };

420 internal sealed class GodotTemplateRenderer : EngineTemplateRendererBase
421 {
425 public override string Render(TemplateGenerationModel model)
426 {
427 var builder = new StringBuilder();
428 AppendFileHeader(builder);
429 AppendNamespaceStart(builder, model);
430
431 if (model.RequiresSpacingThemeConstant)
432 {
433 builder.AppendLine(" private static readonly global::Godot.StringName SeparationThemeConstantName = \"separation\";");
434 builder.AppendLine();
435 }
436
437 AppendConstantMembers(builder, model);
438 AppendPropertyMembers(builder, model);
439
440 if (model.UsesGameObjectAdapter)
441 {
442 builder.AppendLine(" private readonly global::Nomad.EngineUtils.GodotGameObject _impl;");
443 builder.AppendLine(" private bool _isDisposed;");
444 builder.AppendLine();
445 }
446
447 AppendEventMembers(builder, model);
448
449 if (model.UsesGameObjectAdapter)
450 {
451 AppendSummaryDocumentation(builder, " ", "Initializes the generated engine wrapper.");
452 builder.AppendLine(" public " + model.GeneratedClassName + "()");
453 builder.AppendLine(" {");
454 builder.AppendLine(" _impl = new global::Nomad.EngineUtils.GodotGameObject(this);");
455 builder.AppendLine(" Ready += () => _Ready();");
456 builder.AppendLine(" TreeExited += () => _ExitTree();");
457 builder.AppendLine(" }");
458 builder.AppendLine();
459
460 AppendSummaryDocumentation(builder, " ", "Initializes generated events and invokes startup hooks.");
461 builder.AppendLine(" public sealed override void _Ready()");
462 builder.AppendLine(" {");
463 builder.AppendLine(" base._Ready();");
464 builder.AppendLine();
465 builder.AppendLine(" GetTree().ProcessFrame += () => _Process(GetProcessDeltaTime());");
466 builder.AppendLine(" GetTree().PhysicsFrame += () => _PhysicsProcess(GetPhysicsProcessDeltaTime());");
467 builder.AppendLine();
468
469 AppendEventInitializationWithInterpolation(builder, model, " ");
470 if (model.EventImplementations.Length > 0)
471 {
472 builder.AppendLine();
473 }
474
475 AppendEventHooks(builder, model, " ");
476 if (model.EventImplementations.Length > 0)
477 {
478 builder.AppendLine();
479 }
480
481 builder.AppendLine(" _impl.OnInit();");
482 builder.AppendLine(" OnInit();");
483 builder.AppendLine(" }");
484 builder.AppendLine();
485
486 AppendSummaryDocumentation(builder, " ", "Invokes the generated frame update pipeline.");
487 builder.AppendLine(" public sealed override void _Process(double delta)");
488 builder.AppendLine(" {");
489 builder.AppendLine(" base._Process(delta);");
490 builder.AppendLine();
491 builder.AppendLine(" float deltaTime = (float)delta;");
492 builder.AppendLine(" _impl.OnUpdate(deltaTime);");
493 builder.AppendLine(" OnUpdate(deltaTime);");
494 builder.AppendLine(" }");
495 builder.AppendLine();
496
497 AppendSummaryDocumentation(builder, " ", "Invokes the generated physics update pipeline.");
498 builder.AppendLine(" public sealed override void _PhysicsProcess(double delta)");
499 builder.AppendLine(" {");
500 builder.AppendLine(" base._PhysicsProcess(delta);");
501 builder.AppendLine();
502 builder.AppendLine(" float deltaTime = (float)delta;");
503 builder.AppendLine(" _impl.OnPhysicsUpdate(deltaTime);");
504 builder.AppendLine(" OnPhysicsUpdate(deltaTime);");
505 builder.AppendLine(" }");
506 builder.AppendLine();
507
508 AppendSummaryDocumentation(builder, " ", "Invokes shutdown hooks and disposes generated events.");
509 builder.AppendLine(" public sealed override void _ExitTree()");
510 builder.AppendLine(" {");
511 builder.AppendLine(" base._ExitTree();");
512 builder.AppendLine();
513 builder.AppendLine(" DisposeCore();");
514 builder.AppendLine(" }");
515 builder.AppendLine();
516
517 AppendSummaryDocumentation(builder, " ", "Releases generated engine resources for the wrapper.");
518 builder.AppendLine(" public new void Dispose()");
519 builder.AppendLine(" {");
520 builder.AppendLine(" DisposeCore();");
521 builder.AppendLine(" base.Dispose();");
522 builder.AppendLine(" }");
523 builder.AppendLine();
524
525 AppendSummaryDocumentation(builder, " ", "Performs one-time disposal for the generated engine wrapper.");
526 builder.AppendLine(" private void DisposeCore()");
527 builder.AppendLine(" {");
528 builder.AppendLine(" if (_isDisposed)");
529 builder.AppendLine(" {");
530 builder.AppendLine(" return;");
531 builder.AppendLine(" }");
532 builder.AppendLine();
533 builder.AppendLine(" _isDisposed = true;");
534 builder.AppendLine();
535 builder.AppendLine(" _impl.OnShutdown();");
536 builder.AppendLine(" OnShutdown();");
537 if (model.EventImplementations.Length > 0)
538 {
539 builder.AppendLine();
540 AppendEventDisposal(builder, model, " ");
541 }
542
543 builder.AppendLine(" }");
544 builder.AppendLine();
545
546 AppendGameObjectLifecycleHooks(builder);
547 }
548 else if (model.IsAsset)
549 {
550 AppendSummaryDocumentation(builder, " ", "Disposes the generated asset wrapper and releases generated event bindings.");
551 builder.AppendLine(" public new void Dispose()");
552 builder.AppendLine(" {");
553 builder.AppendLine(" OnDispose();");
554 if (model.EventImplementations.Length > 0)
555 {
556 builder.AppendLine();
557 AppendEventDisposal(builder, model, " ");
558 }
559
560 builder.AppendLine(" base.Dispose();");
561 builder.AppendLine(" }");
562 builder.AppendLine();
563
564 AppendAssetDisposeHook(builder);
565 }
566
567 AppendMethodMembers(builder, model);
568 AppendNamespaceEnd(builder);
569 return builder.ToString();
570 }

576 internal sealed class UnityTemplateRenderer : EngineTemplateRendererBase
577 {
581 public override string Render(TemplateGenerationModel model)
582 {
583 var builder = new StringBuilder();
584 AppendFileHeader(builder);
585 AppendNamespaceStart(builder, model);
586
587 AppendConstantMembers(builder, model);
588 AppendPropertyMembers(builder, model);
589
590 if (model.UsesGameObjectAdapter)
591 {
592 builder.AppendLine(" private global::Nomad.EngineUtils.UnityGameObject _impl = default!;");
593 builder.AppendLine(" private bool _isDisposed;");
594 builder.AppendLine();
595 builder.AppendLine(" private TUnityComponent? GetUnityComponent<TUnityComponent>()");
596 builder.AppendLine(" where TUnityComponent : global::UnityEngine.Component");
597 builder.AppendLine(" {");
598 builder.AppendLine(" return gameObject.GetComponent<TUnityComponent>();");
599 builder.AppendLine(" }");
600 builder.AppendLine();
601 }
602
603 AppendEventMembers(builder, model);
604
605 if (model.UsesGameObjectAdapter)
606 {
607 AppendSummaryDocumentation(builder, " ", "Creates the generated engine adapter and initializes generated events.");
608 builder.AppendLine(" private void Awake()");
609 builder.AppendLine(" {");
610 builder.AppendLine(" _impl = new global::Nomad.EngineUtils.UnityGameObject(gameObject);");
611 if (model.EventImplementations.Length > 0)
612 {
613 builder.AppendLine();
614 AppendEventInitializationWithInterpolation(builder, model, " ");
615 }
616
617 builder.AppendLine(" }");
618 builder.AppendLine();
619
620 AppendSummaryDocumentation(builder, " ", "Invokes startup hooks for the generated engine wrapper.");
621 builder.AppendLine(" private void Start()");
622 builder.AppendLine(" {");
623 builder.AppendLine(" _impl.OnInit();");
624 builder.AppendLine(" OnInit();");
625 builder.AppendLine(" }");
626 builder.AppendLine();
627
628 AppendSummaryDocumentation(builder, " ", "Invokes the generated frame update pipeline.");
629 builder.AppendLine(" private void Update()");
630 builder.AppendLine(" {");
631 builder.AppendLine(" float deltaTime = global::UnityEngine.Time.deltaTime;");
632 builder.AppendLine(" _impl.OnUpdate(deltaTime);");
633 builder.AppendLine(" OnUpdate(deltaTime);");
634 builder.AppendLine(" }");
635 builder.AppendLine();
636
637 AppendSummaryDocumentation(builder, " ", "Invokes the generated physics update pipeline.");
638 builder.AppendLine(" private void FixedUpdate()");
639 builder.AppendLine(" {");
640 builder.AppendLine(" float deltaTime = global::UnityEngine.Time.fixedDeltaTime;");
641 builder.AppendLine(" _impl.OnPhysicsUpdate(deltaTime);");
642 builder.AppendLine(" OnPhysicsUpdate(deltaTime);");
643 builder.AppendLine(" }");
644 builder.AppendLine();
645
646 AppendSummaryDocumentation(builder, " ", "Invokes shutdown hooks and releases generated event bindings.");
647 builder.AppendLine(" private void OnDestroy()");
648 builder.AppendLine(" {");
649 builder.AppendLine(" DisposeCore();");

656 builder.AppendLine(" DisposeCore();");
657 builder.AppendLine(" global::UnityEngine.Object.Destroy(this);");

692 builder.AppendLine(" public void Dispose()");
