160 public T? FindChild<T>(string childName)
161 where T : class, IGameObject
162 {
163 if (string.IsNullOrEmpty(childName))
164 {
165 return null;
166 }
167
168 // TODO: write a zero-allocation version of this
169 ReadOnlySpan<string> segments = childName.Split('/');
170 var current = _node;
171
172 for (int i = 0; i < segments.Length; i++)
173 {
174 var segment = segments[i];
175 if (string.IsNullOrEmpty(segment))
176 {
177 continue;
178 }
179 bool found = false;
180 foreach (var child in current.GetChildren())
181 {
182 if (child.Name == segment)
183 {
184 current = child;
185 found = true;
186 break;
187 }
188 }
189 if (!found)
190 {
191 return null;
192 }
193 }
194 return (T)WrapNode(current);
195 }
