41 internal sealed class History : IHistory {
42 private const string CONSOLE_HISTORY_FILE = "user://history.txt";
43 private const int CONSOLE_HISTORY_MAX = 32;
44
45 private readonly Queue<string> _consoleHistory = new Queue<string>();
46 private readonly string _historyPath;

58 /*
59 ===============
60 History
61 ===============
62 */
70 public History( ICommandBuilder builder, IEngineService engineService, ILoggerService logger, IGameEventRegistryService eventFactory ) {
71 ArgumentGuard.ThrowIfNull( builder );
72 ArgumentGuard.ThrowIfNull( eventFactory );
73 ArgumentGuard.ThrowIfNull( logger );
74 ArgumentGuard.ThrowIfNull( engineService );
75
76 _logger = logger;
77 _historyPath = engineService.GetStoragePath( "history.txt", StorageScope.UserData );
78
79 builder.TextEntered.Subscribe( OnTextEntered );
80
81 eventFactory.GetEvent<EmptyEventArgs>( Constants.Events.Console.HISTORY_PREV_EVENT, Constants.Events.Console.NAMESPACE ).Subscribe( OnHistoryPrev );
82 eventFactory.GetEvent<EmptyEventArgs>( Constants.Events.Console.HISTORY_NEXT_EVENT, Constants.Events.Console.NAMESPACE ).Subscribe( OnHistoryNext );
83
84 _historyPrev = eventFactory.GetEvent<HistoryPrevEventArgs>( Constants.Events.Console.HISTORY_PREV_EVENT, Constants.Events.Console.NAMESPACE );
85 _historyNext = eventFactory.GetEvent<HistoryNextEventArgs>( Constants.Events.Console.HISTORY_NEXT_EVENT, Constants.Events.Console.NAMESPACE );
86
87 LoadHistory();
88 }

113 public void OnTextEntered( in TextEnteredEventArgs args ) {

115 SaveHistory();

118 /*
119 ===============
120 SaveHistory
121 ===============
122 */
126 private void SaveHistory() {
127 _logger?.PrintLine( "History.SaveHistory: writing command line history to disk..." );
128 try {
129 using StreamWriter writer = new StreamWriter( _historyPath );
130 foreach ( var history in _consoleHistory ) {
131 writer.WriteLine( history );
132 }
133 } catch ( Exception e ) {
134 _logger?.PrintError( $"History.SaveHistory: couldn't write console command history data to file '{_historyPath}'! Exception: {e}" );
135 throw;
136 }
137 }
138
139 /*
140 ===============
141 OnHistoryPrev
142 ===============
143 */
147 private void OnHistoryPrev( in EmptyEventArgs args ) {
148 if ( _historyIndex > 0 ) {
149 _historyIndex--;
150 if ( _historyIndex >= 0 ) {
151 HistoryPrev.Publish( new HistoryPrevEventArgs( _consoleHistory.ElementAt( _historyIndex ) ) );
152 }
153 }
154 }
155
156 /*
157 ===============
158 OnHistoryNext
159 ===============
160 */
164 private void OnHistoryNext( in EmptyEventArgs args ) {
165 if ( _historyIndex < _consoleHistory.Count ) {
166 _historyIndex++;
167 HistoryNext.Publish( new HistoryNextEventArgs( _historyIndex < _consoleHistory.Count, _consoleHistory.ElementAt( _historyIndex ) ) );
168 }
169 }
170
171 /*
172 ===============
173 LoadHistory
174 ===============
175 */
176 private void LoadHistory() {
177 try {
178 using StreamReader reader = new StreamReader( _historyPath );
179
180 string? text;
181 while ( (text = reader.ReadLine()) != null ) {
182 AddInputHistory( text );
183 }
184 } catch ( Exception ) {
185 }
186 }
187
188 /*
189 ===============
190 AddInputHistory
191 ===============
192 */
197 private void AddInputHistory( string text ) {
198 if ( _consoleHistory.Count >= CONSOLE_HISTORY_MAX ) {
199 _consoleHistory.Dequeue();
200 }
201 _consoleHistory.Enqueue( text );
202 _historyIndex = _consoleHistory.Count;
203 }
204 };
