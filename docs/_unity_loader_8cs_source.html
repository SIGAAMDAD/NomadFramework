35 internal sealed class UnityLoader : IResourceLoader {
43 public Result<TResource> Load<TResource, TId>( TId id ) {
44 if ( id is not string path ) {
45 throw new InvalidCastException();
46 }
47
48 UnityEngine.Object resource = Resources.Load( path, typeof( TResource ) );
49 if ( resource == null ) {
50 return Result<TResource>.Failure( InternalError.Create( "Error loading unity resource '" + path + "'" ) );
51 }
52
53 if ( resource is TResource loadedResource ) {
54 return Result<TResource>.Success( loadedResource );
55 }
56
57 throw new InvalidCastException();
58 }

73 public async Task<Result<TResource>> LoadAsync<TResource, TId>( TId id, CancellationToken ct = default ) {
74 ct.ThrowIfCancellationRequested();
75
76 if ( id is not string path ) {
77 throw new InvalidCastException();
78 }
79
80 ResourceRequest request = Resources.LoadAsync( path, typeof( TResource ) );
81 if ( request == null ) {
82 return Result<TResource>.Failure( InternalError.Create( "Error loading unity resource '" + path + "'" ) );
83 }
84
85 UnityEngine.Object resource = await request;
86 if ( resource is TResource loadedResource ) {
87 return Result<TResource>.Success( loadedResource );
88 }
89
90 throw new InvalidCastException();
91 }
