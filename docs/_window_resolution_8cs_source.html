159 static WindowResolutionExtensions()
160 {
161 _displayStringMap = new Dictionary<WindowResolution, InternString>(_resolutionData.Length);
162 _sizeMap = new Dictionary<WindowResolution, WindowSize>(_resolutionData.Length);
163 _parseMap = new Dictionary<InternString, WindowResolution>(_resolutionData.Length);
164
165 foreach (var data in _resolutionData)
166 {
167 // For Native resolution, there is no valid size; skip adding to _sizeMap
168 if (data.resolution != WindowResolution.Res_Native)
169 {
170 _sizeMap[data.resolution] = new WindowSize(data.width, data.height);
171 }
172
173 _displayStringMap[data.resolution] = data.display;
174 _parseMap[data.display] = data.resolution;
175 }
176 }
177
184 public static InternString ToDisplayString(this WindowResolution resolution)
185 {
186 if (_displayStringMap.TryGetValue(resolution, out var display))
187 {
188 return display;
189 }
190 throw new ArgumentOutOfRangeException(nameof(resolution), resolution, null);
191 }
192
199 public static WindowSize GetSize(this WindowResolution resolution)
200 {
201 if (_sizeMap.TryGetValue(resolution, out var size))
202 {
203 return size;
204 }
205 throw new ArgumentOutOfRangeException(nameof(resolution), resolution, null);
206 }

214 public static bool TryParse(InternString resolutionString, out WindowResolution resolution)
215 {
216 if (resolutionString == InternString.Empty)
217 {
218 resolution = WindowResolution.Default;
219 return false;
220 }
221 if (_parseMap.TryGetValue(resolutionString, out resolution))
222 {
223 return true;
224 }
225 resolution = WindowResolution.Default;
226 return false;
227 }
