55 public ReadOnlySpan<ResolvedAction> ResolveMatchesNonAlloc( CompiledBindingGraph graph, BindingMatchSet matches, long timeStamp ) {
56 BeginPass( graph.Actions.Length );
57
58 for ( int i = 0; i < matches.Length; i++ ) {
59 int bindingIndex = matches.BindingIndices[i];
60 ref readonly var binding = ref graph.Bindings[bindingIndex];
61
62 if ( !TryEvaluateBinding( in binding, out var valueType, out var buttonValue, out var floatValue, out var vector2Value, out var magnitude ) ) {
63 continue;
64 }
65
66 StoreBestCandidate(
67 actionSlot: binding.ActionIndex,
68 bindingIndex: bindingIndex,
69 score: matches.Scores[i],
70 valueType: valueType,
71 buttonValue: buttonValue,
72 floatValue: floatValue,
73 vector2Value: vector2Value,
74 magnitude: magnitude
75 );
76 }
77
78 return MaterializeResolvedActions( graph, timeStamp );
79 }

81 public ReadOnlySpan<ResolvedAction> ResolveKeyboardCompositesNonAlloc( CompiledBindingGraph graph, uint activeContextMask, InputScheme? activeScheme, long timeStamp ) {
82 BeginPass( graph.Actions.Length );
83
84 bool hasActiveScheme = activeScheme.HasValue;
85 InputScheme activeSchemeValue = activeScheme.GetValueOrDefault();
86
87 ReadOnlySpan<int> composite1D = CompiledBindingRepository.GetComposite1DBindingIndices( graph );
88 for ( int i = 0; i < composite1D.Length; i++ ) {
89 int bindingIndex = composite1D[i];
90 ref readonly var binding = ref graph.Bindings[bindingIndex];
91
92 if ( !ContextMatches( binding.ContextMask, activeContextMask ) ) {
93 continue;
94 }
95 if ( hasActiveScheme && binding.Scheme != activeSchemeValue ) {
96 continue;
97 }
98
99 float value = ResolveComposite1D( in binding.Axis1DComposite );
100 StoreBestCandidate(
101 actionSlot: binding.ActionIndex,
102 bindingIndex: bindingIndex,
103 score: binding.ScoreBase,
104 valueType: InputValueType.Float,
105 buttonValue: false,
106 floatValue: value,
107 vector2Value: default,
108 magnitude: MathF.Abs( value )
109 );
110 }
111
112 ReadOnlySpan<int> composite2D = CompiledBindingRepository.GetComposite2DBindingIndices( graph );
113 for ( int i = 0; i < composite2D.Length; i++ ) {
114 int bindingIndex = composite2D[i];
115 ref readonly var binding = ref graph.Bindings[bindingIndex];
116
117 if ( !ContextMatches( binding.ContextMask, activeContextMask ) ) {
118 continue;
119 }
120 if ( hasActiveScheme && binding.Scheme != activeSchemeValue ) {
121 continue;
122 }
123
124 Vector2 value = ResolveComposite2D( in binding.Axis2DComposite );
125 StoreBestCandidate(
126 actionSlot: binding.ActionIndex,
127 bindingIndex: bindingIndex,
128 score: binding.ScoreBase,
129 valueType: InputValueType.Vector2,
130 buttonValue: false,
131 floatValue: 0.0f,
132 vector2Value: value,
133 magnitude: value.LengthSquared()
134 );
135 }
136
137 return MaterializeResolvedActions( graph, timeStamp );
138 }

152 [MethodImpl( MethodImplOptions.AggressiveInlining )]
153 private void StoreBestCandidate(
154 int actionSlot,
155 int bindingIndex,
156 int score,
157 InputValueType valueType,
158 bool buttonValue,
159 float floatValue,
160 Vector2 vector2Value,
161 float magnitude
162 ) {
163 if ( _slotGeneration[actionSlot] != _generation ) {
164 _slotGeneration[actionSlot] = _generation;
165 _touchedSlots[_touchedCount++] = actionSlot;
166
167 _bestBindingIndex[actionSlot] = bindingIndex;
168 _bestScore[actionSlot] = score;
169 _bestMagnitude[actionSlot] = magnitude;
170 _bestValueType[actionSlot] = valueType;
171 _bestButtonValue[actionSlot] = buttonValue;
172 _bestFloatValue[actionSlot] = floatValue;
173 _bestVector2Value[actionSlot] = vector2Value;
174 return;
175 }
176
177 if ( score < _bestScore[actionSlot] ) {
178 return;
179 }
180 if ( score == _bestScore[actionSlot] && magnitude <= _bestMagnitude[actionSlot] ) {
181 return;
182 }
183
184 _bestBindingIndex[actionSlot] = bindingIndex;
185 _bestScore[actionSlot] = score;
186 _bestMagnitude[actionSlot] = magnitude;
187 _bestValueType[actionSlot] = valueType;
188 _bestButtonValue[actionSlot] = buttonValue;
189 _bestFloatValue[actionSlot] = floatValue;
190 _bestVector2Value[actionSlot] = vector2Value;
191 }

193 private ReadOnlySpan<ResolvedAction> MaterializeResolvedActions( CompiledBindingGraph graph, long timeStamp ) {
194 ResolvedAction[] resolvedActionBuffer = GetResolvedActionBuffer( _touchedCount );
195
196 int actionCount = 0;
197 for ( int i = 0; i < _touchedCount; i++ ) {
198 int slot = _touchedSlots[i];
199 InputValueType valueType = _bestValueType[slot];
200
201 bool isActive = valueType switch {
202 InputValueType.Button => _bestButtonValue[slot],
203 InputValueType.Float => MathF.Abs( _bestFloatValue[slot] ) > 0.0001f,
204 InputValueType.Vector2 => _bestVector2Value[slot].LengthSquared() > 0.0001f,
205 _ => false
206 };
207
208 if ( !TryResolvePhase( slot, isActive, out var phase ) ) {
209 continue;
210 }
211
212 resolvedActionBuffer[actionCount++] = new ResolvedAction(
213 actionId: graph.Actions[slot].ActionId,
214 actionIndex: slot,
215 valueType: valueType,
216 phase: phase,
217 timeStamp: timeStamp,
218 buttonValue: _bestButtonValue[slot],
219 floatValue: _bestFloatValue[slot],
220 vector2Value: _bestVector2Value[slot]
221 );
222 }
223
224 return resolvedActionBuffer.AsSpan( 0, actionCount );
225 }

227 [MethodImpl( MethodImplOptions.AggressiveInlining )]
228 private bool TryResolvePhase( int actionSlot, bool isActive, out InputActionPhase phase ) {
229 InputActionPhase previousPhase = _phaseByAction[actionSlot];
230 bool wasActive = previousPhase is InputActionPhase.Started or InputActionPhase.Performed;
231
232 if ( isActive ) {
233 phase = wasActive ? InputActionPhase.Performed : InputActionPhase.Started;
234 _phaseByAction[actionSlot] = phase;
235 return true;
236 }
237
238 if ( wasActive ) {
239 phase = InputActionPhase.Canceled;
240 _phaseByAction[actionSlot] = phase;
241 return true;
242 }
243
244 phase = default;
245 return false;
246 }

248 private bool TryEvaluateBinding( in CompiledBinding binding, out InputValueType valueType, out bool buttonValue, out float floatValue, out Vector2 vector2Value, out float magnitude ) {
249 switch ( binding.Kind ) {
250 case InputBindingKind.Button: {
251 bool pressed = _stateService.IsPressed( binding.Button.DeviceId, binding.Button.ControlId );
252 valueType = InputValueType.Button;
253 buttonValue = pressed;
254 floatValue = 0.0f;
255 vector2Value = default;
256 magnitude = pressed ? 1.0f : 0.0f;
257 return true;
258 }
259
260 case InputBindingKind.Axis1D: {
261 float value = ApplyAxis1DProcessors(
262 _stateService.GetAxis1D( binding.Axis1D.DeviceId, binding.Axis1D.ControlId ),
263 in binding.Axis1D
264 );
265
266 valueType = InputValueType.Float;
267 buttonValue = false;
268 floatValue = value;
269 vector2Value = default;
270 magnitude = MathF.Abs( value );
271 return true;
272 }
273
274 case InputBindingKind.Axis2D: {
275 Vector2 value = ApplyAxis2DProcessors(
276 _stateService.GetAxis2D( binding.Axis2D.DeviceId, binding.Axis2D.ControlId ),
277 in binding.Axis2D
278 );
279
280 valueType = InputValueType.Vector2;
281 buttonValue = false;
282 floatValue = 0.0f;
283 vector2Value = value;
284 magnitude = value.LengthSquared();
285 return true;
286 }
287
288 case InputBindingKind.Delta2D: {
289 Vector2 value = ApplyDelta2DProcessors( _stateService.MouseDelta, in binding.Delta2D );
290
291 valueType = InputValueType.Vector2;
292 buttonValue = false;
293 floatValue = 0.0f;
294 vector2Value = value;
295 magnitude = value.LengthSquared();
296 return true;
297 }
298
299 default:
300 valueType = default;
301 buttonValue = default;
302 floatValue = default;
303 vector2Value = default;
304 magnitude = default;
305 return false;
306 }
307 }

309 [MethodImpl( MethodImplOptions.AggressiveInlining )]
310 private float ResolveComposite1D( in Axis1DCompositeBinding binding ) {
311 float negative = _stateService.IsPressed( InputDeviceSlot.Keyboard, binding.Negative ) ? 1.0f : 0.0f;
312 float positive = _stateService.IsPressed( InputDeviceSlot.Keyboard, binding.Positive ) ? 1.0f : 0.0f;
313 float value = (positive - negative) * binding.Scale;
314
315 if ( binding.Normalize ) {
316 value = Math.Clamp( value, -1.0f, 1.0f );
317 }
318
319 return value;
320 }

322 [MethodImpl( MethodImplOptions.AggressiveInlining )]
323 private Vector2 ResolveComposite2D( in Axis2DCompositeBinding binding ) {
324 float up = _stateService.IsPressed( InputDeviceSlot.Keyboard, binding.Up ) ? 1.0f : 0.0f;
325 float down = _stateService.IsPressed( InputDeviceSlot.Keyboard, binding.Down ) ? 1.0f : 0.0f;
326 float left = _stateService.IsPressed( InputDeviceSlot.Keyboard, binding.Left ) ? 1.0f : 0.0f;
327 float right = _stateService.IsPressed( InputDeviceSlot.Keyboard, binding.Right ) ? 1.0f : 0.0f;
328
329 Vector2 value = new(
330 (right - left) * binding.ScaleX,
331 (up - down) * binding.ScaleY
332 );
333
334 if ( binding.Normalize && value != Vector2.Zero ) {
335 value = Vector2.Normalize( value );
336 }
337
338 return value;
339 }

341 [MethodImpl( MethodImplOptions.AggressiveInlining )]
342 private static float ApplyAxis1DProcessors( float value, in Axis1DBinding binding ) {
343 if ( MathF.Abs( value ) < binding.Deadzone ) {
344 value = 0.0f;
345 }
346 if ( binding.Invert ) {
347 value = -value;
348 }
349
350 return value * binding.Sensitivity * binding.Scale;
351 }
352
353 [MethodImpl( MethodImplOptions.AggressiveInlining )]
354 private static Vector2 ApplyAxis2DProcessors( Vector2 value, in Axis2DBinding binding ) {
355 float deadzoneSq = binding.Deadzone * binding.Deadzone;
356 if ( value.LengthSquared() < deadzoneSq ) {
357 return Vector2.Zero;
358 }
359 if ( binding.InvertX ) {
360 value.X = -value.X;
361 }
362 if ( binding.InvertY ) {
363 value.Y = -value.Y;
364 }
365
366 value.X *= binding.ScaleX * binding.Sensitivity;
367 value.Y *= binding.ScaleY * binding.Sensitivity;
368 return value;
369 }
370
371 [MethodImpl( MethodImplOptions.AggressiveInlining )]
372 private static Vector2 ApplyDelta2DProcessors( Vector2 value, in Delta2DBinding binding ) {
373 if ( binding.InvertX ) {
374 value.X = -value.X;
375 }
376 if ( binding.InvertY ) {
377 value.Y = -value.Y;
378 }
379
380 value.X *= binding.ScaleX * binding.Sensitivity;
381 value.Y *= binding.ScaleY * binding.Sensitivity;
382 return value;
383 }

390 private void EnsureActionCapacity( int actionCount ) {
391 if ( _phaseByAction.Length >= actionCount ) {
392 return;
393 }
394
395 int previousLength = _phaseByAction.Length;
396 Array.Resize( ref _phaseByAction, actionCount );
397 Array.Fill( _phaseByAction, InputActionPhase.Count, previousLength, actionCount - previousLength );
398 Array.Resize( ref _slotGeneration, actionCount );
399 Array.Resize( ref _touchedSlots, actionCount );
400
401 Array.Resize( ref _bestBindingIndex, actionCount );
402 Array.Resize( ref _bestScore, actionCount );
403 Array.Resize( ref _bestMagnitude, actionCount );
404 Array.Resize( ref _bestButtonValue, actionCount );
405 Array.Resize( ref _bestFloatValue, actionCount );
406 Array.Resize( ref _bestVector2Value, actionCount );
407 Array.Resize( ref _bestValueType, actionCount );
408 }

410 [MethodImpl( MethodImplOptions.AggressiveInlining )]
411 private ResolvedAction[] GetResolvedActionBuffer( int requiredCapacity ) {
412 if ( _resolvedActionBuffers.Length == 0 ) {
413 _resolvedActionBuffers = new ResolvedAction[ResolvedActionBufferRingSize][];
414 Array.Fill( _resolvedActionBuffers, Array.Empty<ResolvedAction>() );
415 }
416
417 int bufferIndex = _resolvedActionBufferCursor;
418 _resolvedActionBufferCursor = (_resolvedActionBufferCursor + 1) % _resolvedActionBuffers.Length;
419
420 ResolvedAction[] buffer = _resolvedActionBuffers[bufferIndex];
421 if ( buffer.Length >= requiredCapacity ) {
422 return buffer;
423 }
424
425 int newCapacity = Math.Max( requiredCapacity, buffer.Length == 0 ? 8 : buffer.Length * 2 );
426 Array.Resize( ref buffer, newCapacity );
427 _resolvedActionBuffers[bufferIndex] = buffer;
428 return buffer;
429 }
