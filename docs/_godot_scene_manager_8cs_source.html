92 public IScene LoadScene( string path, LoadSceneMode mode = LoadSceneMode.Single ) {
93 _sceneCache.GetCached( path ).Get( out var resource );
94 var scene = WalkSceneTree( resource );
95 switch ( mode ) {
96 case LoadSceneMode.Additive:
97 _sceneTree.Root.AddChild( ( scene.Root as GodotGameObject ).Node );
98 _additiveScenes.Add( scene );
99 break;
100 case LoadSceneMode.Single:
101 SetActiveScene( scene );
102 break;
103 }
104 return scene;
105 }

112 public void SetActiveScene( IScene scene ) {
113 ArgumentGuard.ThrowIfNull( scene );
114 if ( scene is not GodotScene godotScene ) {
115 throw new InvalidCastException();
116 }
117
118 // clear any existing scenes
119 UnloadAllScenes();
120 _sceneTree.CallDeferred( SceneTree.MethodName.ChangeSceneToNode, ( godotScene.Root as GodotGameObject ).Node );
121 _activeScene = godotScene;
122 }

128 public void UnloadScene( IScene scene ) {
129 if ( scene is not GodotScene godotScene ) {
130 throw new InvalidCastException();
131 }
132 if ( _additiveScenes.Contains( godotScene ) ) {
133 _sceneTree.Root.CallDeferred( Node.MethodName.RemoveChild, ( godotScene.Root as GodotGameObject ).Node );
134 }
135 _additiveScenes.Remove( godotScene );
136 _sceneCache.Unload( godotScene.Path );
137 }
138
142 public void UnloadAllScenes() {
143 UnloadCurrentScene();
144 for ( int i = 0; i < _additiveScenes.Count; i++ ) {
145 _sceneTree.Root.CallDeferred( Node.MethodName.RemoveChild, ( _additiveScenes[i].Root as GodotGameObject ).Node );
146 _sceneCache.Unload( _additiveScenes[i].Path );
147 }
148 _additiveScenes.Clear();
149 }
150
154 public void UnloadCurrentScene() {
155 if ( _activeScene == null ) {
156 return;
157 }
158 _sceneTree.CurrentScene = null;
159
160 _sceneTree.Root.CallDeferred( Node.MethodName.RemoveChild, ( _activeScene.Root as GodotGameObject ).Node );
161 _sceneCache.Unload( _activeScene.Path );
162 _activeScene = null;
163 }

170 private GodotScene WalkSceneTree( PackedScene resource ) {
171 var root = new GodotGameObject( resource.Instantiate() );
172 _gameObjects[root.Node] = root;
173 WalkNode( root.Node );
174
175 return new GodotScene( root );
176 }
177
182 private void WalkNode( Node parentObject ) {
183 var children = parentObject.GetChildren();
184 for ( int i = 0; i < children.Count; i++ ) {
185 var childNode = children[i];
186 _gameObjects[childNode] = new GodotGameObject( childNode );
187 WalkNode( childNode );
188 }
189 }
