158 public virtual void ClearUnused()
159 {
160#if DEBUG
161 _logger.PrintDebug($"BaseCache{typeof(TResource)}.ClearUnused: removing unused entries...");
162#endif
163 var toRemove = new List<TId>();
164 foreach (KeyValuePair<TId, CacheEntry<TResource, TId>> kvp in _cache)
165 {
166 if (kvp.Value.ReferenceCount == 0)
167 {
168 toRemove.Add(kvp.Key);
169 }
170 }
171
172 int removedCount = 0;
173 _cacheLock.EnterWriteLock();
174 try
175 {
176 for (int i = 0; i < toRemove.Count; i++)
177 {
178 TId id = toRemove[i];
179 if (_cache.Remove(id, out CacheEntry<TResource, TId>? entry) && entry.ReferenceCount <= 0)
180 {
181 removedCount++;
182 _currentMemorySize -= entry.MemorySize;
183 ResourceUnloaded.Publish(new ResourceUnloadedEventArgs<TId>(id, entry.MemorySize, UnloadReason.ReferenceCountZero));
184 entry.Dispose();
185 }
186 else
187 {
188 _logger.PrintError($"BaseCache{typeof(TResource)}.ClearUnused: Dictionary.Remove failed!");
189 }
190 }
191 }
192 finally
193 {
194 _cacheLock.ExitWriteLock();
195 }
196
197 Interlocked.Add(ref _totalUnloaded, removedCount);
198 }

211 public virtual ICacheEntry<TResource, TId> GetCached(TId id)
212 {
213 _cacheLock.EnterUpgradeableReadLock();
214 try
215 {
216 if (_cache.TryGetValue(id, out CacheEntry<TResource, TId>? entry))
217 {
218 Interlocked.Increment(ref _cacheHits);
219 entry.UpdateAccessStats();
220 return entry;
221 }
222 }
223 finally
224 {
225 _cacheLock.ExitUpgradeableReadLock();
226 }
227
228 Interlocked.Increment(ref _cacheMisses);
229 ICacheEntry<TResource, TId>? cacheEntry = LoadAndCacheResource(id);
230 if (cacheEntry == null)
231 {
232 _logger.PrintError($"BaseCache{typeof(TResource)}.GetCached: failed to load resource '{id}'");
233 }
234 return cacheEntry;
235 }

366 public virtual void ReleaseReference(TId id)
367 {
368 bool shouldUnload = false;
369
370 _cacheLock.EnterUpgradeableReadLock();
371 try
372 {
373 if (_cache.TryGetValue(id, out CacheEntry<TResource, TId>? entry))
374 {
375 int newCount = Interlocked.Decrement(ref entry.ReferenceCount);
376 if (newCount == 0 && (DateTime.UtcNow - entry.AccessStats.LastAccessTime) > _policy.UnloadUnusedAfter)
377 {
378 shouldUnload = true;
379 }
380 }
381 else
382 {
383 throw new KeyNotFoundException(id.ToString());
384 }
385 }
386 finally
387 {
388 _cacheLock.ExitUpgradeableReadLock();
389 }
390 if (shouldUnload)
391 {
392 Unload(id);
393 }

535 public virtual void EvictIfNeeded()
536 {
537 if (_policy.EvictionPolicy == EvictionPolicy.Never || (_currentMemorySize <= _policy.MaxMemorySize && _cache.Count <= _policy.MaxResourceCount))
538 {
539 return;
540 }
541
542 _cacheLock.EnterWriteLock();
543 try
544 {
545 var candidates = new Dictionary<TId, CacheEntry<TResource, TId>>();
546 foreach (var entry in _cache)
547 {
548 if (entry.Value.ReferenceCount == 0)
549 {
550 candidates.Add(entry.Key, entry.Value);
551 }
552 }
553 if (candidates.Count == 0)
554 {
555 return;
556 }
557
558 IEnumerable<KeyValuePair<TId, CacheEntry<TResource, TId>>> sortedCandidates = _policy.EvictionPolicy switch
559 {
560 EvictionPolicy.LRU => candidates.OrderBy(kvp => kvp.Value.AccessStats.LastAccessTime),
561 EvictionPolicy.LFU => candidates.OrderBy(kvp => kvp.Value.AccessStats.AccessCount),
562 EvictionPolicy.SizeBased => candidates.OrderByDescending(kvp => kvp.Value.MemorySize),
563 _ => candidates.OrderByDescending(kvp => kvp.Value.AccessStats.LastAccessTime)
564 };
565
566 int removedCount = 0;
567 foreach (KeyValuePair<TId, CacheEntry<TResource, TId>> candidate in sortedCandidates)
568 {
569 if (_currentMemorySize <= _policy.MaxMemorySize && _cache.Count <= _policy.MaxResourceCount)
570 {
571 break;
572 }
573 if (_cache.Remove(candidate.Key, out CacheEntry<TResource, TId>? entry))
574 {
575 removedCount++;
576 _currentMemorySize -= entry.MemorySize;
577 ResourceUnloaded.Publish(new ResourceUnloadedEventArgs<TId>(
578 candidate.Key, entry.MemorySize, UnloadReason.CacheFull
579 ));
580 entry.Dispose();
581 }
582 }
583 Interlocked.Add(ref _totalUnloaded, removedCount);
584 }
585 finally
586 {
587 _cacheLock.ExitWriteLock();
588 }
589 }

596 private ICacheEntry<TResource, TId>? LoadAndCacheResource(TId id)
597 {
598 var loadTimer = Stopwatch.StartNew();
599 try
600 {
601 Result<TResource> result = _loader.Load<TResource, TId>(id);
602 if (result.IsFailure)
603 {
604 _logger.PrintError($"BaseCache.LoadAndCacheResource: failed to load resource '{id}'");
605 return null;
606 }
607
608 int memorySize = CalculateMemorySize(result.Value);
609 return CacheResource(id, result.Value, memorySize, loadTimer.Elapsed);
610 }
611 catch (Exception e)
612 {
613 _logger.PrintError($"BaseCache{typeof(TResource)}.LoadAndCacheResource: exception thrown while loading resource '{id}' - {e}");
614 loadTimer.Stop();
615 throw;
616 }
617 }

700 private ICacheEntry<TResource, TId> CacheResource(TId id, TResource resource, int memorySize, TimeSpan loadTime)
701 {
702 _cacheLock.EnterWriteLock();
703
704 var entry = new CacheEntry<TResource, TId>(
705 this, id, resource, memorySize, loadTime, ResourceLoadState.Complete
706 );
707 try
708 {
709 _cache[id] = entry;
710 _currentMemorySize += memorySize;
711 _totalLoaded++;
712 _totalLoadTime += loadTime;
713
714 ResourceLoaded.Publish(new ResourceLoadedEventArgs<TId>(id, loadTime, memorySize));
715 }
716 finally
717 {
718 _cacheLock.ExitWriteLock();
719 }
720
721 // flush the cache if needed
722 EvictIfNeeded();
723
724 return entry;
725 }
