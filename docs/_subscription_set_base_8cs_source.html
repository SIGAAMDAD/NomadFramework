210 PumpAsyncSnapshot
211 ===============
212 */
222 protected async Task PumpAsyncSnapshot( AsyncEventCallback<TArgs>[] subscriptions, int subscriptionCount, TArgs args, CancellationToken ct ) {
223 ct.ThrowIfCancellationRequested();
224
225 if ( subscriptionCount == 0 ) {
226 IncrementPublishCount();
227 return;
228 }
229
230 if ( subscriptionCount == 1 ) {
231 await PumpSingleAsync( subscriptions[0], 0, args, ct ).ConfigureAwait( false );
232 IncrementPublishCount();
233 return;
234 }
235
236 Task[] tasks = ArrayPool<Task>.Shared.Rent( subscriptionCount );
237 int taskCount = 0;
238 List<EventHandlerException>? failures = null;
239
240 try {
241 for ( int i = 0; i < subscriptionCount; i++ ) {
242 ct.ThrowIfCancellationRequested();
243
244 try {
245 Task task = subscriptions[i].Invoke( args, ct ) ?? Task.CompletedTask;
246 tasks[taskCount++] = task;
247 } catch ( Exception ex ) {
248 failures ??= new List<EventHandlerException>();
249 failures.Add( CreateAsyncFailure( subscriptions[i], i, ex ) );
250
251 if ( ExceptionPolicy == EventExceptionPolicy.Propagate ) {
252 break;
253 }
254 }
255 }
256
257 if ( taskCount > 0 ) {
258#if NET10_0_OR_GREATER
259 Task whenAll = Task.WhenAll( tasks.AsSpan( 0, taskCount ) );
260#else
261 Task whenAll = Task.WhenAll( tasks.Take( taskCount ) );
262#endif
263
264 try {
265 await whenAll.ConfigureAwait( false );
266 } catch {
267 failures ??= new List<EventHandlerException>();
268
269 for ( int i = 0; i < taskCount; i++ ) {
270 Task task = tasks[i];
271
272 if ( !task.IsFaulted || task.Exception == null ) {
273 continue;
274 }
275
276 foreach ( Exception ex in task.Exception.InnerExceptions ) {
277 failures.Add( CreateAsyncFailure( subscriptions[i], i, ex ) );
278 }
279 }
280 }
281 }
282
283 if ( failures is { Count: > 0 } ) {
284 foreach ( EventHandlerException failure in failures ) {
285 Logger?.PrintError( failure.ToString() );
286 }
287
288 if ( ExceptionPolicy != EventExceptionPolicy.ReportAndContinue ) {
289 throw new EventPublishException(
290 EventData.DebugName,
291 typeof( TArgs ),
292 failures
293 );
294 }
295 }
296
297 IncrementPublishCount();
298 } finally {
299 Array.Clear( tasks, 0, taskCount );
300 ArrayPool<Task>.Shared.Return( tasks );
301 }
302 }
