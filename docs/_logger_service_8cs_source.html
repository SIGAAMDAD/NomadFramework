37 internal sealed class LoggerService : ILoggerService {
38 private ICVar<LogLevel>? _logDepth;
39 private readonly List<ILoggerSink> _sinks = new List<ILoggerSink>();
40
41 private readonly ILoggerCategory _defaultCategory;
42 private readonly ConcurrentDictionary<string, LoggerCategory> _categories;
43 private readonly MessageBuilder _messageBuilder = new MessageBuilder();
44
45 private bool _isDisposed = false;
46
47 /*
48 ===============
49 LoggerService
50 ===============
51 */
55 public LoggerService() {
56 _defaultCategory = new LoggerCategory( "Logger", LogLevel.Info, true, _messageBuilder );
57 _categories = new ConcurrentDictionary<string, LoggerCategory> {
58 [ "Default" ] = _defaultCategory as LoggerCategory
59 };
60 }
61
62 /*
63 ===============
64 Dispose
65 ===============
66 */
70 public void Dispose() {
71 if ( !_isDisposed ) {
72 for ( int i = 0; i < _sinks.Count; i++ ) {
73 _sinks[ i ]?.Dispose();
74 }
75 _sinks?.Clear();
76 foreach ( var category in _categories ) {
77 category.Value?.Dispose();
78 }
79 _categories?.Clear();
80 _defaultCategory?.Dispose();
81 }
82 GC.SuppressFinalize( this );
83 _isDisposed = true;
84 }
85
86 /*
87 ===============
88 CreateCategory
89 ===============
90 */
98 public ILoggerCategory CreateCategory( string name, LogLevel level, bool enabled ) {
99 var category = new LoggerCategory( name, level, enabled, _messageBuilder );
100 for ( int i = 0; i < _sinks.Count; i++ ) {
101 category.AddSink( _sinks[ i ] );
102 }
103 return category;
104 }
105
106 /*
107 ===============
108 InitConfig
109 ===============
110 */
115 public void InitConfig( ICVarSystemService cvarSystem ) {
116 _logDepth = cvarSystem.Register(
117 new CVarCreateInfo<LogLevel> {
118 Name = Constants.CVars.Console.CONSOLE_LOG_LEVEL,
119 DefaultValue = LogLevel.Info,
120 Description = "The verbosity of the logger.",
121 Flags = CVarFlags.Archive,
122 Validator = value => value >= LogLevel.Error && value < LogLevel.Count
123 }
124 );
125 }
126
127 /*
128 ===============
129 AddSink
130 ===============
131 */
136 public void AddSink( ILoggerSink sink ) {
137 ArgumentGuard.ThrowIfNull( sink );
138 _sinks.Add( sink );
139
140 foreach ( var category in _categories ) {
141 category.Value.AddSink( sink );
142 }
143 }
144
145 /*
146 ===============
147 PrintMessage
148 ===============
149 */
156 [MethodImpl( MethodImplOptions.AggressiveInlining )]
157 private void PrintMessage( ILoggerCategory category, LogLevel level, string message ) {
158 if ( level < LogLevel.Info || level >= _logDepth?.Value ) {
159 return;
160 }
161 category.PrintLine( message );
162 }
163
164 /*
165 ===============
166 PrintLine
167 ===============
168 */
173 public void PrintLine( string message ) {
174 PrintMessage( _defaultCategory, LogLevel.Info, message );
175 }
176
177 /*

186 public void PrintDebug( string message ) {

189
190 /*

192 PrintWarning
193 ===============
194 */
199 public void PrintWarning( string message ) {
200 PrintMessage( _defaultCategory, LogLevel.Warning, message );
201 }
202
203 /*
204 ===============
205 PrintError
206 ===============
207 */
212 public void PrintError( string message ) {
213 PrintMessage( _defaultCategory, LogLevel.Error, message );
214 }
215
216 /*
217 ===============
218 Clear
219 ===============
220 */
224 public void Clear() {
225 for ( int i = 0; i < _sinks.Count; i++ ) {
226 _sinks[ i ].Clear();
227 }
228 }
229 };
