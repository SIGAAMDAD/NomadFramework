56 private static string FormatButton(in ButtonBinding binding)
57 {
58 if (binding.ControlId == InputControlId.None)
59 {
60 return UNBOUND;
61 }
62
63 string control = FormatControl(binding.ControlId);
64 if (binding.Modifiers.IsDefaultOrEmpty)
65 {
66 return PrefixWithDevice(binding.DeviceId, control);
67 }
68
69 string[] parts = new string[binding.Modifiers.Length + 1];
70 for (int i = 0; i < binding.Modifiers.Length; i++)
71 {
72 parts[i] = FormatControl(binding.Modifiers[i]);
73 }
74
75 parts[parts.Length - 1] = control;
76 return PrefixWithDevice(binding.DeviceId, string.Join(" + ", parts));
77 }

191 private static string SplitPascalCase(string value)
192 {
193 if (string.IsNullOrEmpty(value))
194 {
195 return string.Empty;
196 }
197
198 var builder = new StringBuilder(value.Length + 8);
199 builder.Append(value[0]);
200
201 for (int i = 1; i < value.Length; i++)
202 {
203 char current = value[i];
204 char previous = value[i - 1];
205
206 if ((char.IsUpper(current) && !char.IsUpper(previous))
207 || (char.IsDigit(current) && !char.IsDigit(previous)))
208 {
209 builder.Append(' ');
210 }
211 builder.Append(current);
212 }
213 return builder.ToString();
214 }
