172 public bool TryAdd(T item)
173 {
174 var node = new Node(item);
175
176 if (_allowMultipleProducers)
177 {
178 // Multi-producer: use interlocked operations
179 Node? oldTail, oldNext;
180
181 while (true)
182 {
183 oldTail = _tail;
184 oldNext = oldTail.Next;
185
186 // Check if tail is still consistent
187 if (oldTail == _tail)
188 {
189 if (oldNext == null)
190 {
191 // Try to link new node at end
192 if (Interlocked.CompareExchange(ref oldTail.Next, node, null) == null)
193 {
194 // Try to move tail forward
195 Interlocked.CompareExchange(ref _tail, node, oldTail);
196 IncrementCount();
197 return true;
198 }
199 }
200 else
201 {
202 // Help other thread by moving tail forward
203 Interlocked.CompareExchange(ref _tail, oldNext, oldTail);
204 }
205 }
206 }
207 }
208 else
209 {
210 // Single-producer: simpler algorithm
211 Node tail = _tail;
212 tail.Next = node;
213 _tail = node;
214 IncrementCount();
215 return true;
216 }
217 }

222 public bool TryTake(out T item)
223 {
224 if (_allowMultipleConsumers)
225 {
226 // Multi-consumer: use interlocked operations
227 while (true)
228 {
229 Node head = _head;
230 Node tail = _tail;
231 Node? next = head.Next;
232
233 if (head == _head)
234 { // Check consistency
235 if (head == tail)
236 {
237 // Queue might be empty or tail needs updating
238 if (next == null)
239 {
240 item = default!;
241 return false; // Queue is empty
242 }
243
244 // Help by moving tail forward
245 Interlocked.CompareExchange(ref _tail, next, tail);
246 }
247 else
248 {
249 // Read value before CAS (important!)
250 item = next!.Value!;
251
252 // Try to move head forward
253 if (Interlocked.CompareExchange(ref _head, next, head) == head)
254 {
255 DecrementCount();
256 // Optional: set old head's value to default for GC
257 head.Value = default;
258 return true;
259 }
260 }
261 }
262 }
263 }
264 else
265 {
266 // Single-consumer: simpler algorithm
267 Node head = _head;
268 Node? next = head.Next;
269
270 if (next == null)
271 {
272 item = default!;
273 return false;
274 }
275
276 item = next!.Value!;
277 _head = next;
278 DecrementCount();
279 head.Value = default; // Help GC
280 return true;
281 }
282 }
