86 public override bool AddSubscription( EventCallback<TArgs> callback ) {
87 ThrowIfDisposed();
88 ArgumentGuard.ThrowIfNull( callback );
89
90 while ( true ) {
91 EventCallback<TArgs>[] current = Volatile.Read( ref _genericSubscriptions );
92
93 if ( IndexOf( current, callback ) >= 0 ) {
94 Logger?.PrintWarning( $"AtomicSubscriptionSet.AddSubscription: subscription for callback '{callback.Method.Name}' already exists!" );
95 return false;
96 }
97
98 EventCallback<TArgs>[] updated = new EventCallback<TArgs>[current.Length + 1];
99 Array.Copy( current, updated, current.Length );
100 updated[current.Length] = callback;
101
102 if ( ReferenceEquals( Interlocked.CompareExchange( ref _genericSubscriptions, updated, current ), current ) ) {
103 IncrementSubscriberCount();
104 return true;
105 }
106 }
107 }

213
214 EventCallback<TArgs>[] subscriptions = Volatile.Read( ref _genericSubscriptions );
215 List<EventHandlerException>? failures = null;
216
217 for ( int i = 0; i < subscriptions.Length; i++ ) {
218 var callback = subscriptions[i];
219 var ex = InvokeCallback( callback, i, in args );
220 if ( ex != null ) {
221 failures ??= new();
222 failures.Add( ex );
223 }
224 }
225
226 IncrementPublishCount();

294 private static int IndexOf<TCallback>( TCallback[] subscriptions, TCallback callback ) {
295 for ( int i = 0; i < subscriptions.Length; i++ ) {
296 if ( Equals( subscriptions[i], callback ) ) {
297 return i;
298 }
299 }
300 return -1;
301 }

305 RemoveAt
306 ===============
307 */
315 private static TCallback[] RemoveAt<TCallback>( TCallback[] subscriptions, int index ) {
316 if ( subscriptions.Length == 1 ) {
317 return Array.Empty<TCallback>();
318 }
319
320 TCallback[] updated = new TCallback[subscriptions.Length - 1];
321 if ( index > 0 ) {
322 Array.Copy( subscriptions, 0, updated, 0, index );
323 }
324
325 int tailLength = subscriptions.Length - index - 1;
326 if ( tailLength > 0 ) {
327 Array.Copy( subscriptions, index + 1, updated, index, tailLength );
328 }
329
330 return updated;
331 }
