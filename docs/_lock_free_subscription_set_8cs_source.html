82 public override bool AddSubscription( EventCallback<TArgs> callback ) {
83 ThrowIfDisposed();
84 ArgumentGuard.ThrowIfNull( callback );
85
86 if ( ContainsCallback( callback, out _ ) ) {
87 Logger?.PrintWarning( $"LockFreeSubscriptionSet.AddSubscription: subscription for callback '{callback.Method.Name}' already exists!" );
88 return false;
89 }

91 _genericSubscriptions.Add( callback );

93 return true;
94 }

114 */
120 public override bool RemoveSubscription( EventCallback<TArgs> callback ) {
121 ThrowIfDisposed();
122 ArgumentGuard.ThrowIfNull( callback );
123
124 if ( !ContainsCallback( callback, out int index ) ) {
125 Logger?.PrintWarning( $"LockFreeSubscriptionSet.RemoveSubscription: no such existing subscription for callback '{callback.Method.Name}'." );
126 return false;
127 }
128
129 _genericSubscriptions.RemoveAt( index );

131 return true;
132 }

148 /*
149 ===============
150 Pump
151 ===============
152 */
157 public override void Pump( in TArgs args ) {
158 ThrowIfDisposed();
159
160 List<EventHandlerException>? failures = null;
161
162 for ( int i = 0; i < _genericSubscriptions.Count; i++ ) {
163 var callback = _genericSubscriptions[i];
164 var ex = InvokeCallback( callback, i, in args );
165 if ( ex != null ) {
166 failures ??= new();
167 failures.Add( ex );
168 }
169 }
170
171 IncrementPublishCount();
172 CheckAggregateException( failures );
173 }
174

188 throw new NotSupportedException( $"{nameof( LockFreeSubscriptionSet<TArgs> )} does not support async pumping." );
189 }
